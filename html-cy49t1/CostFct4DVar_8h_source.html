90 template <typename MODEL>

136 template <typename MODEL>
137 void CostFct4DVar<MODEL>::runNL(CtrlVar_ &xx,
138  PostProcessor<State_> &post) const {
139  ASSERT(xx.state().checkStatesNumber(1));
140  ASSERT(xx.state()[0].validTime() == windowBegin_);
141  CostFct_::getModel().forecast(xx.state()[0], xx.modVar(), windowLength_,
142  post);
143  ASSERT(xx.state()[0].validTime() == windowEnd_);
144 }
