51 void TLML95::setTrajectory(StateL95 & xx,
52  const ModelL95 & lrmodel,
53  const ModelBias & bias) {
54  ASSERT(traj_.find(xx.validTime()) == traj_.end());
55  ModelTrajectory * traj = new ModelTrajectory();
56 // Interpolate xx to xlr here
57  FieldL95 zz(xx.getField());
58  lrmodel.stepRK(zz, bias, *traj);
59  traj_[xx.validTime()] = traj;
60 }

78 void TLML95::stepTL(IncrementL95 & xx,
79  const ModelBiasCorrection & bias) const {
80  FieldL95 dx(xx.getField(), false);
81  FieldL95 zz(xx.getField(), false);
82  FieldL95 dz(xx.getField(), false);
83  const ModelTrajectory * traj = this->getTrajectory(xx.validTime());
84 
85  zz = xx.getField();
86  this->tendenciesTL(zz, bias.bias(), traj->get(1), dz);
87  dx = dz;
88 
89  zz = xx.getField();
90  zz.axpy(0.5, dz);
91  this->tendenciesTL(zz, bias.bias(), traj->get(2), dz);
92  dx.axpy(2.0, dz);
93 
94  zz = xx.getField();
95  zz.axpy(0.5, dz);
96  this->tendenciesTL(zz, bias.bias(), traj->get(3), dz);
97  dx.axpy(2.0, dz);
98 
99  zz = xx.getField();
100  zz += dz;
101  this->tendenciesTL(zz, bias.bias(), traj->get(4), dz);
102  dx += dz;
103 
104  const double zt = 1.0/6.0;
105  xx.getField().axpy(zt, dx);
106  xx.validTime() += tstep_;
107 }

109 void TLML95::stepAD(IncrementL95 & xx,
110  ModelBiasCorrection & bias) const {
111  FieldL95 dx(xx.getField(), false);
112  FieldL95 zz(xx.getField(), false);
113  FieldL95 dz(xx.getField(), false);
114 
115  xx.validTime() -= tstep_;
116  const ModelTrajectory * traj = this->getTrajectory(xx.validTime());
117 
118  const double zt = 1.0/6.0;
119  dx = xx.getField();
120  dx *= zt;
121 
122  dz = dx;
123  this->tendenciesAD(zz, bias.bias(), traj->get(4), dz);
124  xx.getField() += zz;
125  dz = zz;
126 
127  dz.axpy(2.0, dx);
128  this->tendenciesAD(zz, bias.bias(), traj->get(3), dz);
129  xx.getField() += zz;
130  dz = zz;
131  dz *= 0.5;
132 
133  dz.axpy(2.0, dx);
134  this->tendenciesAD(zz, bias.bias(), traj->get(2), dz);
135  xx.getField() += zz;
136  dz = zz;
137  dz *= 0.5;
138 
139  dz += dx;
140  this->tendenciesAD(zz, bias.bias(), traj->get(1), dz);
141  xx.getField() += zz;
142 }

144 void TLML95::tendenciesTL(const FieldL95 & xx, const double & bias,
145  const FieldL95 & xtraj, FieldL95 & dx) const {
146  const int nn = resol_.npoints();
147  for (int jj = 0; jj < nn; ++jj) {
148  int jm2 = jj - 2;
149  int jm1 = jj - 1;
150  int jp1 = jj + 1;
151  if (jm2 < 0) jm2 += nn;
152  if (jm1 < 0) jm1 += nn;
153  if (jp1 >= nn) jp1 -= nn;
154  const double dxdt = - xx[jm2] * xtraj[jm1] - xtraj[jm2] * xx[jm1]
155  + xx[jm1] * xtraj[jp1] + xtraj[jm1] * xx[jp1]
156  - xx[jj] - bias;
157  dx[jj] = dt_ * dxdt;
158  }
159 }

161 void TLML95::tendenciesAD(FieldL95 & xx, double & bias,
162  const FieldL95 & xtraj, const FieldL95 & dx) const {
163  const int nn = resol_.npoints();
164  xx.zero();
165  for (int jj = 0; jj < nn; ++jj) {
166  int jm2 = jj - 2;
167  int jm1 = jj - 1;
168  int jp1 = jj + 1;
169  if (jm2 < 0) jm2 += nn;
170  if (jm1 < 0) jm1 += nn;
171  if (jp1 >= nn) jp1 -= nn;
172  const double dxdt = dt_ * dx[jj];
173  xx[jm2] -= dxdt * xtraj[jm1];
174  xx[jm1] -= dxdt * xtraj[jm2];
175  xx[jm1] += dxdt * xtraj[jp1];
176  xx[jp1] += dxdt * xtraj[jm1];
177  xx[jj] -= dxdt;
178  bias -= dxdt;
179  }
180 }
