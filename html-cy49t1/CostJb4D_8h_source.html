144 template <typename MODEL>
145 CostJb4D<MODEL>::CostJb4D(const eckit::Configuration &config,
146  const Geometry_ &resolouter,
147  const Variables_ &ctlvars, const util::Duration &,
148  const State4D_ &xb)
149  : B4D_(), controlvars_(ctlvars), resol_(), times_() {
150  // Create one row of blocks of the whole BMatrix, one object for each
151  // subwindow. Each object stands for all blocks in the same column of the B
152  // Matrix.
153  // It can be from any concrete class of ModelSpaceCovarianceBase,
154  // according to the configuration file.
155  const eckit::LocalConfiguration covar(config, "Covariance");
156  B4D_.reset(
157  Covariance4DFactory<MODEL>::create(covar, resolouter, controlvars_, xb));
158  times_.clear();
159 
160  if (covar.has("envar_times")) {
161  std::vector<eckit::LocalConfiguration> envar_times;
162  covar.get("envar_times", envar_times);
163  for (size_t jsub = 0; jsub < envar_times.size(); ++jsub) {
164  times_.push_back(util::DateTime(envar_times[jsub].getString("reference_time") ));
165  }
166  }
167  else {
168  for (unsigned jsub = 0; jsub < xb.statesNumber(); ++jsub) {
169  times_.push_back(xb[jsub].validTime());
170  }
171  }
172  Log::trace() << "CostJb4D contructed." << std::endl;
173 }

187  Increment4D_ &dx) const {
188  if (xb.statesNumber() == 1) {
189  dx.zero();
