118 template <typename MODEL>
119 double SQRTBPCGMinimizer<MODEL>::solve(
120  CtrlVec_ &dv, CtrlVec_ &rr, const Jbmat_ &Jb, const UtHtRinvHU_ &UtHtRinvHU,
121  const bool &linfoconv, const size_t &maxiter, const size_t &miniter,
122  const double &tolerance, const double &costJ0Jb, const double &costJ0JoJc) {
123  // dv control vector
124  // rr (sum dv_i^{b} +) U^T G^T R^{-1} d
125 
126  CtrlVec_ qq(dv);
127  CtrlVec_ pp(dv);
128  CtrlVec_ pr(dv);
129  CtrlVec_ jbpp(dv);
130  CtrlVec_ jbdv(dv);
131  CtrlVec_ zz(dv);
132  CtrlVec_ r0(dv);
133  CtrlVec_ ww(dv);
134 
135  // vectors for re-orthogonalization
136  std::vector<CtrlVec_> vvecs;
137  std::vector<CtrlVec_> zvecs;
138  std::vector<double> scals;
139 
140  // setup calculation of the quadratic cost function
141  this->setupQuadCost(costJ0Jb, costJ0JoJc);
142 
143  // change resolution of LMP vectors
144  lmp_.changeResolution();
145 
146  // r_{0}
147  r0 = rr;
148 
149  // r_{0}^T r_{0}
150  Log::info() << "normr0 " << dot_product(rr, rr) << std::endl;
151 
152  // z_{0} = LMP r_{0}
153  lmp_.inverseMultiply(rr, zz);
154  // p_{0} = z_{0}
155  pp = zz;
156 
157  // r_{0}^T z_{0}
158  double dotZZ0 = dot_product(rr, zz);
159  Log::info() << "normr0z0 " << dotZZ0 << std::endl;
160  double normReduction = 1.0;
161  double zdots = dotZZ0;
162  double zdots_old = 0.0;
163 
164  vvecs.push_back(rr);
165  zvecs.push_back(zz);
166  scals.push_back(1.0 / dotZZ0);
167 
168  Log::info() << std::endl;
169  for (size_t jiter = 0; jiter < maxiter; ++jiter) {
170  Log::info() << " SQRTPCG Starting Iteration " << jiter + 1 << std::endl;
171 
172  util::LogbookWriter<int> log("InnerLoop", jiter);
173 
174  if (jiter > 0) {
175  // beta_{i} = z_{i+1}^T z_{i+1} / z_{i}^T z_{i}
176  double beta = zdots / zdots_old;
177 
178  // p_{i+1} = z_{i+1} + beta*p_{i}
179  pp *= beta;
180  pp += zz;
181  }
182 
183  // q_{i} = ( P^-1/2 Jb P^-T/2 + U^T H^T R^{-1} H U ) p_{i}
184  UtHtRinvHU.multiply(pp, qq);
185  Jb.multiply(pp, jbpp); // U^T B^-1 U != I for VarBC
186  qq += jbpp;
187 
188  // alpha_{i} = z_{i}^T z_{i} / q_{i}^T p_{i}
189  double rho = dot_product(pp, qq);
190  double alpha = zdots / rho;
191 
192  // dv_{i+1} = dv_{i} + alpha * p_{i}
193  dv.axpy(alpha, pp);
194 
195  // jbdv_{i+1} = jbdv_{i} + alpha * jbh_{i}
196  jbdv.axpy(alpha, jbpp);
197 
198  // r_{i+1} = r_{i} - alpha * q_{i}
199  rr.axpy(-alpha, qq);
200 
201  // Compute the quadratic cost function
202  this->calcQuadCost(dv, jbdv, r0);
203 
204  // Re-orthogonalization
205  for (size_t jj = 0; jj < jiter; ++jj) {
206  double proj = scals[jj] * dot_product(rr, zvecs[jj]);
207  rr.axpy(-proj, vvecs[jj]);
208  }
209 
210  lmp_.inverseMultiply(rr, zz);
211 
212  // z_{i}^T z_{i}
213  zdots_old = zdots;
214  // z_{i+1}^T z_{i+1}
215  zdots = dot_product(rr, zz);
216 
217  Log::info() << "zdots " << zdots << " iteration " << jiter
218  << std::endl;
219 
220  // z_{i+1}^T z_{i+1} / z_{0}^T z_{0}
221  normReduction = sqrt(zdots / dotZZ0);
222 
223  util::LogbookWriter<double> log_norm("NormReduction", normReduction);
224 
225  Log::info() << "SQRTPCG end of iteration " << jiter + 1 << std::endl
226  << " Norm reduction (" << std::setw(2) << jiter + 1
227  << ") = " << util::full_precision(normReduction) << std::endl;
228 
229  this->printQuadCost(dv, jiter);
230 
231  if (normReduction < tolerance && jiter >= miniter) {
232  Log::info() << "SQRTPCG: Achieved required reduction in residual norm."
233  << std::endl;
234  break;
235  }
236 
237  vvecs.push_back(rr);
238  zvecs.push_back(zz);
239  scals.push_back(1.0 / zdots);
240 
241  lmp_.push(pp, zz, qq, rho);
242  }
243  lmp_.update();
244 
245  return normReduction;
246 }

250 template <typename MODEL>
251 void SQRTBPCGMinimizer<MODEL>::setupQuadCost(const double &costJ0Jb,
252  const double &costJ0JoJc) {
253  // J0
254  costJ0Jb_ = costJ0Jb;
255  costJ0JoJc_ = costJ0JoJc;
256 
257  // J0
258  costJ0_ = costJ0Jb_ + costJ0JoJc_;

260  // reset

263  costJoJc_ = 0;

268 template <typename MODEL>
269 void SQRTBPCGMinimizer<MODEL>::calcQuadCost(const CtrlVec_ &dv,
270  const CtrlVec_ &jbdv,
271  const CtrlVec_ &r0) {
272  // Compute the quadratic cost function
273  // J[dv_{i}] = J[0] - 0.5 dv_{i}^T r_{0}
274  costJ_ = costJ0_ - 0.5 * dot_product(dv, r0);
275  Log::info() << "dvtro " << util::full_precision(dot_product(dv, r0))
276  << std::endl;
277  // Jb[dv_{i}] = 0.5 dv_{i}^T Jb.dv_{i}
278  costJb_ = costJ0Jb_ + 0.5 * dot_product(dv, jbdv);
279  // Jo[dv_{i}] + Jc[dv_{i}] = J[dv_{i}] - Jb[dv_{i}]
280  costJoJc_ = costJ_ - costJb_;
281 }
