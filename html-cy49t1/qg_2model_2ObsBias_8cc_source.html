39 ObsBias::ObsBias(const ObsSpaceQG &, const eckit::Configuration & conf)
40  : active_(false), bias_(ntypes, 0.0), dummyObsSpaceKey_(0), dummyKey_(0)
41 {
42  Log::info() << "ObsBias: conf = " << conf << std::endl;
43  active_ = conf.has("stream") || conf.has("uwind") ||
44  conf.has("vwind") || conf.has("wspeed");
45  if (active_) {

47  if (conf.has("uwind")) bias_[1] = conf.getDouble("uwind");
48  if (conf.has("vwind")) bias_[2] = conf.getDouble("vwind");

90 void ObsBias::read(const eckit::Configuration & conf) {
91  std::string filename = conf.getString("filename");
92  Log::info() << " ObsBias::read opening" << filename << std::endl;
93  eckit::PathName path = eckit::LocalPathName::cwd() + "/" + filename;
94  std::ifstream ifs(path.localPath());
95  if (!ifs)
96  throw eckit::CantOpenFile(path.localPath(), Here());
97  for (size_t ii = 0; ii < ntypes; ++ii) {
98  if (!ifs.bad()) {
99  std::string bias;
100  ifs >> bias;
101  if (ifs.good()) {
102  bias_[ii] = std::stod(bias);
103  } else if (ifs.eof()) {
104  throw eckit::ReadError("Failed to read entire ObsBias", Here());
105  } else if (ifs.fail()) {
106  throw eckit::ReadError("Failed to read ObsBias", Here());
107  }
108  }
109  }
110  Log::info() << " DEBUG ObsBias::read " << util::full_precision(bias_) << std::endl;
111 }

129 double ObsBias::norm() const {
130  double zz = 0.0;
131  if (active_) {
132  for (unsigned int jj = 0; jj < ntypes; ++jj) zz += bias_[jj]*bias_[jj];
133  zz = std::sqrt(zz/3.0);
134  }
135  return zz;
136 }
