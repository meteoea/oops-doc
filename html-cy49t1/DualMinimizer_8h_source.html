74 template <typename MODEL>
75 ControlIncrement<MODEL> *DualMinimizer<MODEL>::doMinimize(
76  const eckit::Configuration &config) {
77  int ninner = config.getInt("ninner");
78  double gnreduc = config.getDouble("gradient_norm_reduction");
79 
80  bool lauxMinDiags = false;
81  if (config.has("onlineDiagnostics")) {
82  const eckit::LocalConfiguration onlineDiag(config, "onlineDiagnostics");
83  lauxMinDiags = onlineDiag.getBool("auxMinDiags");
84  }
85 
86  if (gradJb_ == 0) {
87  gradJb_.reset(new CtrlInc_(J_.jb()));
88  } else {
89  gradJb_.reset(new CtrlInc_(J_.jb().resolution(), *gradJb_));
90  }
91 
92  Log::info() << std::endl;
93  Log::info() << classname() << ": max iter = " << ninner
94  << ", requested norm reduction = " << gnreduc << std::endl;
95 
96  // Define the matrices
97  const Bmat_ B(J_);
98  const HBHt_ HBHt(J_, lauxMinDiags);
99  const Rinv_ Rinv(J_);
100  const HMatrix<MODEL> H(J_);
101  const HtMatrix<MODEL> Ht(J_);
102 
103  // Define minimisation starting point in dual space
104  Dual_ vv;
105  for (unsigned jj = 0; jj < J_.nterms(); ++jj) {
106  vv.append(J_.jterm(jj).newDualVector());
107  }
108  double vvp = 0.0;
109 
110  // Get R^{-1} d
111  Dual_ rr;
112  for (unsigned jj = 0; jj < J_.nterms(); ++jj) {
113  rr.append(J_.jterm(jj).newGradientFG());
114  }
115 
116  // Update rr if initial dx in model space is not a zero vector
117  // rr = rr - Rinv H dx0
118 
119  // Compute a = H (xb - xk)
120  Dual_ dy;
121  CtrlInc_ hdxfg(J_.jb().getFirstGuessIncrement());
122  H.multiply(hdxfg, dy);
123  dy *= -1.0;
124 
125  // Compute sigma = (xb - xk)^T Binv (xb - xk)
126  CtrlInc_ g0(J_.jb());
127  J_.jb().addGradientFG(g0, *gradJb_);
128 
129  double sigma = -1.0 * dot_product(J_.jb().getFirstGuessIncrement(), g0);
130 
131  // Solve the linear system
132  double reduc =
133  this->solve(vv, vvp, rr, HBHt, Rinv, ninner, gnreduc, dy, sigma);
134 
135  Log::test() << classname() << ": reduction in residual norm = " << reduc
136  << std::endl;
137 
138  // Recover solution in primal space
139  CtrlInc_ dh(J_.jb());
140  J_.zeroAD(dh);
141  Ht.multiply(vv, dh);
142  dh.axpy(vvp, g0);
143 
144  CtrlInc_ *dx = new CtrlInc_(J_.jb());
145  B.multiply(dh, *dx); // BHtaug vvaug
146 
147  Log::info() << classname()
148  << ": Estimated Final Jb = " << 0.5 * dot_product(*dx, dh)
149  << std::endl;
150  Log::info() << classname() << " output" << *dx << std::endl;
151 
152  // Update gradient Jb
153  *gradJb_ += dh;
154 
155  return dx;
156 }
