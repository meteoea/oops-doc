162 template <typename MODEL>
163 void LinearModel<MODEL>::forecastTL(Increment_ &dx, const ModelAuxIncr_ &mctl,
164  const util::Duration &len,
165  PostProcessor<Increment_> post,
166  PostProcessorTL<Increment_> cost,
167  const bool idmodel) const {
175  Log::trace() << "LinearModel<MODEL>::forecastTL starting" << std::endl;
176  util::Timer timer(classname(), "forecastTL");
177 
178  const util::DateTime end(dx.validTime() + len);
179  const util::Duration tstep(tlm_->timeResolution());
180  Log::info() << "LinearModel<MODEL>::forecastTL: Starting " << dx << std::endl;
181  this->initializeTL(dx, mctl);
182  cost.initializeTL(dx, end, tstep);
183  post.initialize(dx, end, tstep);
184  if (idmodel) {
185  while (dx.validTime() < end) {
186  dx.updateTime(tstep);
187  cost.processTL(dx);
188  post.process(dx);
189  }
190  } else {
191  while (dx.validTime() < end) {
192  this->stepTL(dx, mctl);
193  cost.processTL(dx);
194  post.process(dx);
195  }
196  }
197  cost.finalizeTL(dx);
198  post.finalize(dx);
199  this->finalizeTL(dx);
200  Log::info() << "LinearModel<MODEL>::forecastTL: Finished " << dx << std::endl;
201  ASSERT(dx.validTime() == end);
202 
203  Log::trace() << "LinearModel<MODEL>::forecastTL done" << std::endl;
204 }

208 template <typename MODEL>
209 void LinearModel<MODEL>::forecastAD(Increment_ &dx, ModelAuxIncr_ &mctl,
210  const util::Duration &len,
211  PostProcessor<Increment_> post,
212  PostProcessorAD<Increment_> cost,
213  const bool idmodel) const {
221  Log::trace() << "LinearModel<MODEL>::forecastAD starting" << std::endl;
222  util::Timer timer(classname(), "forecastAD");
223 
224  const util::DateTime bgn(dx.validTime() - len);
225  const util::Duration tstep(tlm_->timeResolution());
226  Log::info() << "LinearModel<MODEL>::forecastAD: Starting " << dx << std::endl;
227  this->initializeAD(dx, mctl);
228  post.initialize(dx, bgn, tstep);
229  cost.initializeAD(dx, bgn, tstep);
230  if (idmodel) {
231  while (dx.validTime() > bgn) {
232  cost.processAD(dx);
233  dx.updateTime(-tstep);
234  post.process(dx);
235  }
236  } else {
237  while (dx.validTime() > bgn) {
238  cost.processAD(dx);
239  this->stepAD(dx, mctl);
240  post.process(dx);
241  }
242  }
243  cost.finalizeAD(dx);
244  post.finalize(dx);
245  this->finalizeAD(dx);
246  Log::info() << "LinearModel<MODEL>::forecastAD: Finished " << dx << std::endl;
247  ASSERT(dx.validTime() == bgn);
248 
249  Log::trace() << "LinearModel<MODEL>::forecastAD done" << std::endl;
250 }
