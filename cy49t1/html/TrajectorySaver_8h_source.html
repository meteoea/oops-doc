186 template <typename MODEL>
187 void TrajectorySaver<MODEL>::recordTrajectory() {
189  for (size_t jsub = 0; jsub < subwin_; ++jsub) {
190  ASSERT(subtlm_[jsub] != nullptr);
192  for (std::shared_ptr<PostBase_> jp : processors_)
193  jp->doInitialize(*forcing_[jsub][0], end_[jsub], step_[jsub]);
195  forcing_[jsub][0].reset();
197  for (size_t jj = 1; jj < forcing_[jsub].size(); ++jj) {
199  xlr_->forceWith(*forcing_[jsub][jj], varsForced_);
201  forcing_[jsub][jj].reset();
203  for (std::shared_ptr<PostBase_> jp : processors_) jp->doProcessing(*xlr_);
205  subtlm_[jsub]->setTrajectory(*xlr_, lrBias_);
206  }
208  for (std::shared_ptr<PostBase_> jp : processors_) jp->doFinalize(*xlr_);
209  }
210  tlm_.swap(subtlm_);
211 }
