37 void RestartHelper::start() {
38  Log::info() << "RestartHelper::start() Start Restart Logbook recording"
39  << std::endl;
40 
41  if (getInstance().path_.exists()) {
42  getInstance().isReadIn_ = true;
43  Log::info() << "RestartHelper::RestartHelper Path : " << getInstance().path_
44  << " exists." << std::endl;
45  if (mpi::comm().rank() == getInstance().root_) {
46  eckit::YAMLConfiguration yaml(getInstance().path_);
47  getInstance().logbook_.reset(new Logbook(yaml));
48  } else {
49  getInstance().logbook_.reset(new Logbook());
50  }
51  getInstance().logbook_->broadcast(getInstance().root_);
52  } else {
53  Log::info() << "RestartHelper::RestartHelper Path : " << getInstance().path_
54  << " does not exist." << std::endl;
55  getInstance().logbook_.reset(new Logbook());
56  }
57  getInstance().on_ = true;
58 
59  Log::info() << *getInstance().logbook_ << std::endl;
60 }

72 void RestartHelper::write() {
73  if (mpi::comm().rank() == getInstance().root_) {
74  eckit::PathName uniquepath = eckit::PathName::unique(getInstance().path_);
75  std::ofstream ofs(uniquepath.localPath());
76  if (!ofs) throw eckit::CantOpenFile(uniquepath.localPath(), Here());
77  eckit::JSON j(ofs);
78  j << *(getInstance().logbook_);
79  ofs.close();
80  eckit::PathName::rename(uniquepath, getInstance().path_);
81  }
82 }
