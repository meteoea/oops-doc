115 template<typename MODEL>
116 void AdvectorHandler<MODEL>::inv_mult_ensemble() const {
117  for (unsigned k = 0; k < advector_inv_.size(); ++k) {
118  if (EnsemblesCollection_::getInstance().has(time_[k])) {
119  Log::info()
120  << "AdvectorHandler inv_mult_ensemble() called for time = "
121  << time_[k] << std::endl;
122  EnsemblePtr_ ens = EnsemblesCollection_::getInstance()[time_[k]];
123  for (unsigned l = 0; l < ens->size(); ++l) {
124  advector_inv_[k]->advect((*ens)[l], *param_adv_, (*ens)[l]);
125  }
126  }
127  }
128 }

146 // -----------------------------------------------------------------------------
147 template<typename MODEL>
148 void AdvectorHandler<MODEL>::doFinalize(const State_ &) {
149 
150  for (unsigned k = 1; k < xx_.size(); ++k) {
151  time_.push_back(xx_[k]->validTime());
152  advector_.emplace_back(
153  new Advector_(*(xx_[k]), config_, false));
154  advector_inv_.emplace_back(
155  new Advector_(*(xx_[k]), config_, false));
156  advector_ad_.emplace_back(
157  new Advector_(*(xx_[k]), config_, true));
158 
159  advector_[k - 1]->set_wind(xx_);
160  advector_inv_[k - 1]->set_wind_inv(xx_);
161  *advector_ad_[k - 1] = *advector_[k - 1];
162  }
163  xx_.clear();
164  inv_mult_ensemble();
165 }
