102 template <typename MODEL, typename INCR>
103 void ObserverAD<MODEL, INCR>::doFirstAD(INCR &dx, const util::DateTime &bgn,
104  const util::Duration &tstep) {
105  if (hslot_ == util::Duration(0)) hslot_ = tstep / 2;
106  tstep_ = tstep;
107  const util::DateTime end(dx.validTime());
108  if (subwindows_) {
109  if (bgn == end) {
110  bgn_ = bgn - hslot_;
111  end_ = end + hslot_;
112  } else {
113  bgn_ = bgn;
114  end_ = end;
115  }
116  }
117  if (bgn_ < winbgn_) bgn_ = winbgn_;
118  if (end_ > winend_) end_ = winend_;
119  gvalsIter_ it = gvals_.find(bgn_);
120  if (it == gvals_.end()) {
121  std::vector<std::unique_ptr<GeoVaLs_> > gvalsvec;
122  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
123  gvalsvec.emplace_back(
124  new GeoVaLs_(obspaces_[jj], hoptlad_.variables(jj), dx, bgn_, end_));
125  }
126  gvals_.emplace(bgn_, std::move(gvalsvec));
127  it = gvals_.find(bgn_);
128  ASSERT(it != gvals_.end());
129  }
130  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
131  // [H(dx) P(dB)]^T
132  hoptlad_[jj].obsEquivAD(*it->second[jj], (*ydep_)[jj].obsvector(),
133  (*dybias_)[jj]);
134  // P(dB)^T
135  if ((*ydep_)[jj].hasBias())
136  hoptlad_[jj].obsBiasEquivAD(*it->second[jj], (*ydep_)[jj].biasvector(),
137  (*dybias_)[jj]);
138  }
139 }

141 template <typename MODEL, typename INCR>
142 void ObserverAD<MODEL, INCR>::doProcessingAD(INCR &dx) {
143  util::DateTime xxtime(bgn_);
144  int ts = hslot_.toSeconds()*2;
145 
146  if (ts == tstep_.toSeconds()) {
148  xxtime = dx.validTime();
149  }
150  else {
152  util::Duration term(dx.validTime()-bgn_);
153  int vt = term.toSeconds();
154  if ( (vt % ts) > tstep_.toSeconds()/2 ) return;
155 
156  // xxtime : validTime of timeslot to use
157  util::Duration slot((int)std::round(((double)vt)/((double)ts))*ts);
158  xxtime += slot;
159  }
160 
162  util::DateTime t1(dx.validTime() - hslot_);
163  util::DateTime t2(dx.validTime() + hslot_);
164  if (t1 < bgn_) t1 = bgn_;
165  if (t2 > end_) t2 = end_;
166 
167  //Log::info() << "ObserverAD doProcessing for : t1=" << t1 << " t2=" << t2 << std::endl;
168 
169  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
171  Locations_ locs(obspaces_[jj], t1, t2);
172 
174  gvalsIter_ it = gvals_.find(bgn_);
175  ASSERT(it != gvals_.end());
176  dx.interpolateAD(locs, *it->second[jj]);
177  }
178 }

182  gvalsIter_ it = gvals_.find(bgn_);
183  ASSERT(it != gvals_.end());
184  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
185  it->second[jj]->zero();
186  }
