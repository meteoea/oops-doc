40 std::map<util::DateTime, double> DolphChebyshev::setWeights(
41  const util::DateTime& bgn, const util::DateTime& end,
42  const util::Duration& dt) {
43  constexpr double pi = 3.14159265358979323846;
44  const util::Duration window(end - bgn);
45  const int nstep = window.toSeconds() / dt.toSeconds();
46  ASSERT(window.toSeconds() == dt.toSeconds() * nstep);
47 
48  const int M = nstep / 2;
49  const int N = 2 * M + 1;
50  Log::info() << "DolphChebyshev M = " << M << ", N = " << N << std::endl;
51 
52  const int tt = tau_.toSeconds() / dt.toSeconds();
53  ASSERT(tau_.toSeconds() == dt.toSeconds() * tt);
54  ASSERT(tt > 1);
55  double thetas = 0.;
56  if (window_) {
57  thetas = 2.0 * pi / M;
58  } else {
59  thetas = 2.0 * pi / tt;
60  }
61  const double x0 = 1.0 / std::cos(thetas / 2.0);
62  Log::info() << "DolphChebyshev tt = " << tt << " thetas = " << thetas
63  << ", x0 = " << x0 << std::endl;
64 
65  std::vector<double> w(M + 1);
66  double rr = 0.0;
67  for (int n = 0; n <= M; ++n) {
68  double tn = 2.0 * pi * n / N;
69  double sum = 0.0;
70  for (int m = 0; m <= M; ++m) {
71  double xx = x0 * std::cos(pi * m / N);
72  double t2m = 0.0;
73  double tnm2 = 1.0;
74  double tnm1 = xx;
75  for (int kk = 2; kk <= 2 * M; ++kk) {
76  t2m = 2.0 * xx * tnm1 - tnm2;
77  tnm2 = tnm1;
78  tnm1 = t2m;
79  }
80  if (n == 0 && m == 0) {
81  rr = 1. / t2m;
82  Log::info() << "DolphChebyshev 2.rr = " << 2.0 * rr << std::endl;
83  } else if (m > 0) {
84  sum += t2m * std::cos(m * tn);
85  }
86  }
87  w[n] = (1.0 + 2.0 * rr * sum) / N;
88  }
89  // Check the sum
90  double checksum = 0.0;
91  for (int jj = -M; jj <= M; ++jj) {
92  int n = std::abs(jj);
93  checksum += w[n];
94  }
95  Log::debug() << "DolphChebyshev checksum = " << checksum << std::endl;
96  ASSERT(std::abs(checksum - 1.0) < 1.0e-8);
97  // Combining with simple filter
98  double sum = 0.0;
99  std::vector<double> ws(2 * M + 1);
100  if (window_) {
101  for (int jj = -M; jj <= M; ++jj) {
102  int n = std::abs(jj);
103  if (jj == 0) {
104  ws[jj + M] = w[n] * thetas / pi;
105  } else {
106  ws[jj + M] = w[n] * std::sin(jj * thetas) / (jj * pi);
107  }
108  sum += ws[jj + M];
109  }
110  }
111  std::map<util::DateTime, double> weights;
112  util::DateTime now(bgn);
113  for (int jj = -M; jj <= M; ++jj) {
114  int n = std::abs(jj);
115  Log::debug() << "DolphChebyshev now = " << now << ", n = " << n
116  << ", w = " << w[n] << std::endl;
117  if (window_) {
118  weights[now] = ws[jj + M] / sum;
119  } else {
120  weights[now] = w[n];
121  }
122  now += dt;
123  }
124  ASSERT(now == end + dt);
125 
126  return weights;
127 }
