94 template <typename MODEL>
95 double DRGMRESRMinimizer<MODEL>::solve(CtrlInc_ &xx, CtrlInc_ &xh, CtrlInc_ &rr,
96  const Bmat_ &B, const HtRinvH_ &HtRinvH,
97  const double costJ0Jb,
98  const double costJ0JoJc,
99  const int maxiter,
100  const double tolerance) {
101  IdentityMatrix<CtrlInc_> precond;
102  std::vector<CtrlInc_> c;
103  std::vector<CtrlInc_> u;
104  std::vector<CtrlInc_> uh;
105  CtrlInc_ cc(xh);
106  CtrlInc_ zz(xh);
107  CtrlInc_ zh(xh);
108 
109  double dotRr0 = dot_product(rr, rr);
110  double normReduction = 1.0;
111 
112  Log::info() << std::endl;
113  for (int jiter = 0; jiter < maxiter; ++jiter) {
114  Log::info() << " DRGMRESR Starting Iteration " << jiter + 1 << std::endl;
115 
116  precond.multiply(rr, zh); // returns zh as approxmate solve of (BA)zh = r
117  B.multiply(zh, zz); // x=B zh
118  HtRinvH.multiply(zz, cc);
119  cc += zh; // cc = zh+Cz = Az
120 
121  for (int jj = 0; jj < jiter; ++jj) {
122  double alpha = -dot_product(c[jj], cc);
123  cc.axpy(alpha, c[jj]); // cc = cc - <c[jj],cc>*c[jj];
124  zz.axpy(alpha, u[jj]); // zz = zz - <c[jj],cc>*u[jj];
125  zh.axpy(alpha, uh[jj]); // zh = zh - <c[jj],cc>*uh[jj];
126  }
127 
128  double ccnorm = sqrt(dot_product(cc, cc));
129  c.push_back(cc); // c[jiter]=cc
130  c[jiter] *= 1.0 / ccnorm;
131  u.push_back(zz); // u[jiter]=z
132  u[jiter] *= 1.0 / ccnorm;
133  uh.push_back(zh); // uh[jiter]=zh
134  uh[jiter] *= 1.0 / ccnorm;
135 
136  double cdotr = dot_product(c[jiter], rr);
137  xx.axpy(cdotr, u[jiter]);
138  xh.axpy(cdotr, uh[jiter]);
139  rr.axpy(-cdotr, c[jiter]);
140 
141  normReduction = sqrt(dot_product(rr, rr) / dotRr0);
142  Log::info() << "DRGMRESR end of iteration " << jiter + 1
143  << ". Norm reduction= " << util::full_precision(normReduction)
144  << std::endl
145  << std::endl;
146 
147  if (normReduction < tolerance) {
148  Log::info() << "DRGMRESR: Achieved required reduction in residual norm."
149  << std::endl;
150  break;
151  }
152  }
153  return normReduction;
154 }
