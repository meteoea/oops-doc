139 template <typename MODEL>
140 CostJbJq<MODEL>::CostJbJq(const eckit::Configuration &config,
141  const Geometry_ &resolouter,
142  const Variables_ &ctlvars, const util::Duration &len,
143  const State4D_ &xb, const bool forcing)
144  : B_(0),
145  subwin_(len),
146  forcing_(forcing),
147  controlvars_(ctlvars),
148  resol_(),
149  times_() {
150  std::vector<eckit::LocalConfiguration> confs;
151  config.get("Covariance", confs);
152  for (size_t jsub = 0; jsub < confs.size(); ++jsub) {
153  B_.emplace_back(CovarianceFactory<MODEL>::create(confs[jsub], resolouter,
154  ctlvars, xb[jsub]));
155  }
156  Log::trace() << "CostJbJq contructed." << std::endl;
157 }

161 template <typename MODEL>
162 void CostJbJq<MODEL>::linearize(const State4D_ &fg,
163  const Geometry_ &resolinner,
164  const eckit::Configuration & innerConf) {
165  ASSERT(fg.checkStatesNumber(B_.size()));
166  resol_.reset(new Geometry_(resolinner));
167  times_.clear();
168  for (unsigned jsub = 0; jsub < B_.size(); ++jsub) {
169  times_.push_back(fg[jsub].validTime());
170  B_[jsub]->linearize(fg[jsub], *resol_, innerConf);
171  }
172 }

186 template <typename MODEL>
187 void CostJbJq<MODEL>::addGradient(const Increment4D_ &dxFG, Increment4D_ &grad,
188  Increment4D_ &gradJb) const {
189  // Jb from pre-computed gradient
190  grad[0] -= gradJb[0];
191 
192  // MC - the first formulation below would be valid if the state was updated
193  // just at the begining of the window and the forecast was ran from
194  // there throughout the entire window; this is not the case in the
195  // forcing formulation
196 
197  // Compute and add Jq gradient Qi^{-1} ( x_i - M(x_{i-1}) )
198  if (!forcing_) {
199  Increment4D_ gg(grad, false);
200  for (unsigned jsub = 1; jsub < B_.size(); ++jsub) {
201  B_[jsub]->inverseMultiply(dxFG[jsub], gg[jsub]);
202  grad[jsub] -= gg[jsub];
203  }
204  } else {
205  for (unsigned jsub = 1; jsub < B_.size(); ++jsub) {
206  grad[jsub] -= gradJb[jsub];
207  }
208  }
209 }

238 template <typename MODEL>
239 void CostJbJq<MODEL>::Bmult(const Increment4D_ &dxin,
240  Increment4D_ &dxout) const {
241  for (unsigned jsub = 0; jsub < B_.size(); ++jsub) {
242  B_[jsub]->multiply(dxin[jsub], dxout[jsub]);
243 
244  if (jsub == 0) {
245  Log::debug() << "CostJbJq:multiply Jb is "
246  << 0.5 * dot_product(dxin[0], dxout[0]) << std::endl;
247  } else {
248  Log::debug() << "CostJbJq:multiply Jq(" << jsub << ") is "
249  << 0.5 * dot_product(dxin[jsub], dxout[jsub]);
250  }
251  }
252 }
