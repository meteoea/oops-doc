171 template <typename MODEL>
172 ControlIncrement<MODEL> *SQRTMinimizer<MODEL>::doMinimize(
173  const eckit::Configuration &config) {
174  iter_++;
175  ASSERT(iter_ <= maxiter_);
176  Log::info() << std::endl;
177  Log::info() << classname() << ": starting" << std::endl;
178 
180  this->setupMinimization(config);
181 
183  const Jbmat_ Jb(J_);
184  const Bmat_ B(J_);
185  const UtHtRinvHU_ UtHtRinvHU(J_, config);
186 
189  CtrlInc_ fgrhs(J_.jb().getFirstGuessIncrement(), false);
190 
194  CtrlVec_ rhs(J_.jb());
195 
197  CtrlVec_ dv(J_.jb());
198 
199  double costJ0JoJc(0.0);
200  if (lclassicChaVar_) {
203 
207  if (gradJb_ == 0) {
208  gradJb_.reset(new CtrlVec_(J_.jb()));
209  } else {
210  gradJb_.reset(new CtrlVec_(J_.jb(), *gradJb_));
211  }
212  // Calculate first guess gradient
214  J_.computeGradientFG(fgrhs);
215 
217  B.multiplySqrtTrans(fgrhs, rhs);
218 
221  J_.jb().addGradientFG(rhs, *gradJb_);
222 
224  costJ0JoJc = J_.getCostJoJc();
225  } else {
228 
231  if (dv0_ == 0) {
232  dv0_.reset(new CtrlVec_(J_.jb()));
234  J_.computeGradientFG(fgrhs);
236  costJ0JoJc = J_.getCostJoJc();
237  } else {
238  dv0_.reset(new CtrlVec_(J_.jb(), *dv0_));
243  costJ0JoJc = J_.computeGradientFG(*dv0_, fgrhs);
244  }
247  B.multiplySqrtTrans(fgrhs, rhs);
258  CtrlVec_ jbdv0(J_.jb());
260  Jb.multiply(*dv0_, jbdv0);
261  rhs -= jbdv0;
263  dv = *dv0_;
264  }
265 
267  double reduc = this->solve(dv, rhs, Jb, UtHtRinvHU, linfoConv_, ninner_,
268  ninnerMin_, gnreduc_, costJ0Jb_, costJ0JoJc);
269 
272  this->enforceObsAuxFormulation(dv);
273 
275  CtrlInc_ *dx = new CtrlInc_(J_.jb().getFirstGuessIncrement(), false);
276  B.multiplySqrt(dv, *dx);
277 
278  // Print
279  if (linfoConv_) {
280  Log::test() << classname()
281  << ": final relative gain in information content = " << reduc
282  << std::endl;
283  } else {
284  Log::test() << classname() << ": reduction in residual norm = " << reduc
285  << std::endl;
286  }
287  Log::info() << classname() << " output increment:" << *dx << std::endl;
288 
290  CtrlVec_ jbdv(dv, false);
291  Jb.multiply(dv, jbdv);
292 
293  // Update Jb component of J[0]: 0.5 (x_i - x_b)^T B^-1 (x_i - x_b)
294  this->updateJbCostFunction(dv, jbdv);
295 
297  if (lclassicChaVar_) {
299 
301  *gradJb_ += dv;
302  } else {
304 
306  *dv0_ = dv;
308  const CtrlInc_ &dxfg = J_.jb().getFirstGuessIncrement();
309  *dx -= dxfg;
310  }
311 
312  Log::info() << classname() << ": done" << std::endl;
313 
314  return dx;
315 }

319 template <typename MODEL>
320 void SQRTMinimizer<MODEL>::setupSolver(const eckit::Configuration &conf) {
324  if (conf.has("classic_change_of_variables")) {
325  // true : dx = U dv
326  // false : dx = U dv + x_b - x_fg
327  lclassicChaVar_ = conf.getBool("classic_change_of_variables");
328  }
329 
331  if (conf.has("linfoconv")) linfoConv_ = conf.getBool("linfoconv");
332 
334  if (conf.has("preconditioner") || conf.has("obsAuxFormulations")) {
335  // Provide number of outer loops in order to skip the update of the
336  // preconditioner in the final minimization or/and to properly apply
337  // obs aux incremental formulation
338  ASSERT(conf.has("minimizations"));
339  maxiter_ = conf.getUnsigned("minimizations");
340  }
341 
344  if (conf.has("obsAuxFormulations")) {
345  Log::info()
346  << "SQRTMinimizer<MODEL>::setupSolver() set ObsAux* formulations"
347  << std::endl;
348  setObsAuxFormulation_ = true;
349  std::vector<eckit::LocalConfiguration> configs;
350  conf.get("obsAuxFormulations", configs);
351  for (size_t ii = 0; ii < configs.size(); ++ii) {
352  std::string key(configs[ii].getString("component"));
353  bool value(configs[ii].getBool("incremental"));
354  obsAuxFormulation_.insert(std::make_pair(key, value));
355  Log::info() << " ObsAuxComponent: " << key
356  << ", incremental formulation: " << value << std::endl;
357  }
358  }
359 }

387 template <typename MODEL>
388 void SQRTMinimizer<MODEL>::enforceObsAuxFormulation(CtrlVec_ &dv) const {
397  if (setObsAuxFormulation_ && iter_ < maxiter_) {
398  Log::info() << "SQRTMinimizer<MODEL>::enforceObsAuxFormulation(dv)"
399  << std::endl;
400  std::map<std::string, bool>::const_iterator icst =
401  obsAuxFormulation_.begin();
402  while (icst != obsAuxFormulation_.end()) {
403  if (!icst->second) {
404  for (std::size_t jobs = 0; jobs < dv.obsVar().size(); ++jobs) {
405  if (dv.obsVar()[jobs].size() > 0) {
406  dv.obsVar()[jobs].obsAuxCtlVec(icst->first).zero();
407  Log::info() << " set component of the control vector: "
408  << icst->first << " to zero" << std::endl;
409  }
410  }
411  }
412  ++icst;
413  }
414  }

419 template <typename MODEL>
420 void SQRTMinimizer<MODEL>::updateJbCostFunction(const CtrlVec_ &dv,
421  const CtrlVec_ &jbdv) {
424  if (lclassicChaVar_) {
425  dvh_.push_back(jbdv);
426 
427  // Update Jb component of J[0]: 0.5 (x_i - x_b)^T B^-1 (x_i - x_b)
428  costJ0Jb_ += 0.5 * dot_product(dv, jbdv);
429  costJ0JbJq_ += 0.5 * dot_product(dv.modVar(), jbdv.modVar());
430 
431  for (size_t jouter = 1; jouter < dvh_.size(); ++jouter) {
432  CtrlVec_ dvhtmp(J_.jb(), dvh_[jouter - 1]);
433  // MC temp
434  // costJ0Jb_ += dot_product(dv.state(), dvhtmp.state());
435  costJ0JbJq_ += dot_product(dv.modVar(), dvhtmp.modVar());
436  }
437  } else {
438  // Update Jb component of J[0]: 0.5 (x_i - x_b)^T B^-1 (x_i - x_b)
439  costJ0Jb_ = 0.5 * dot_product(dv, jbdv);
440  costJ0JbJq_ = 0.5 * dot_product(dv.modVar(), jbdv.modVar());
441  }
442 
443  // Print out costJ0Jb and its ObsAux components
444  Log::info() << "SQRTMinimizer: costJ0Jb : "
445  << util::full_precision(costJ0Jb_) << std::endl;
446  Log::info() << "SQRTMinimizer: costJ0JbJq : "
447  << util::full_precision(costJ0JbJq_) << std::endl;
448 }
