115 template <typename MODEL>
116 void Model<MODEL>::forecast(State_ &xx, const ModelAux_ &maux,
117  const util::Duration &len,
118  PostProcessor<State_> &post) const {
126  Log::trace() << "Model<MODEL>::forecast starting" << std::endl;
127  util::Timer timer(classname(), "forecast");
128  util::LogbookWriter<std::string> log("Forecast", "Nonlinear HR trajectory");
129  const util::DateTime end(xx.validTime() + len);
130  {
131  size_t totsteps = len / model_->timeResolution();
132  util::LogbookWriter<int> log_totsteps("traj_totsteps", totsteps);
133  }
134  Log::info() << "Model:forecast: forecast starting: " << xx << std::endl;
135  this->initialize(xx, maux);
136  post.initialize(xx, end, model_->timeResolution());
137  size_t step = 0;
138  while (xx.validTime() < end) {
139  this->step(xx, maux);
140  post.process(xx);
141  ++step;
142  util::LogbookWriter<int> log_step("traj_step", step);
143  }
144  post.finalize(xx);
145  this->finalize(xx);
146  Log::info() << "Model:forecast: forecast finished: " << xx << std::endl;
147  ASSERT(xx.validTime() == end);
148 
149  Log::trace() << "Model<MODEL>::forecast done" << std::endl;
150 }
