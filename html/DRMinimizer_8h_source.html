97 template <typename MODEL>
98 ControlIncrement<MODEL> *DRMinimizer<MODEL>::doMinimize(
99  const eckit::Configuration &config) {
100  // Config
101  int ninner = config.getInt("ninner");
102  double gnreduc = config.getDouble("gradient_norm_reduction");
103 
104  // Online Diagnostics
105  bool lauxMinDiags = false;
106  if (config.has("onlineDiagnostics")) {
107  const eckit::LocalConfiguration onlineDiag(config, "onlineDiagnostics");
108  lauxMinDiags = onlineDiag.getBool("auxMinDiags");
109  }
110 
111  // Change of variable
112  if (config.has("classic_change_of_variables")) {
113  // true : dx = B dx'
114  // false : dx = B dx' + x_b - x_fg
115  lclassicChaVar_ = config.getBool("classic_change_of_variables");
116  }
117 
118  Log::info() << std::endl;
119  Log::info() << classname() << ": max iter = " << ninner
120  << ", requested norm reduction = " << gnreduc << std::endl;
121 
122  // Define matrices
123  const Bmat_ B(J_);
124  const HtRinvH_ HtRinvH(J_, lauxMinDiags);
125 
126  // Allocate dx and dxh
127  // dx
128  CtrlInc_ *dx = new CtrlInc_(J_.jb());
129  // dxh = B^{-1} dx
130  CtrlInc_ dxh(J_.jb());
131 
132  // Compute H^T R^{-1} d
133  // dx_i = x_i - x_{i-1}; dx_1 = x_1 - x_b
134  CtrlInc_ rhs(J_.jb());
135  J_.computeGradientFG(rhs);
136 
137  if (lclassicChaVar_) {
138  // dx = Bdx'
139 
140  // Interpolate gradJb to the new inner loop resolution
141  if (gradJb_ == 0) {
142  gradJb_.reset(new CtrlInc_(J_.jb()));
143  } else {
144  gradJb_.reset(new CtrlInc_(J_.jb().resolution(), *gradJb_));
145  }
146  // Calculate first guess gradient
147  // -SUM dx' + H^T R^{-1} d
148  J_.jb().addGradientFG(rhs, *gradJb_);
149  } else {
150  // dx = Bdx' + x_b - x_fg
151 
152  // Interpolate the initial guess to the new inner loop resolution
153  if (dxh0_ == 0) {
154  dxh0_.reset(new CtrlInc_(J_.jb()));
155  } else {
156  dxh0_.reset(new CtrlInc_(J_.jb().resolution(), *dxh0_));
157  }
158  // Calculate first guess gradient
159  // H^T R^{-1} [d + H (x_b - x_fg)]
160  const HtRinvHdxFG_ HtRinvHdxFG(J_);
161  CtrlInc_ ht_rinv_h_dxfg(J_.jb());
162  HtRinvHdxFG.compute(ht_rinv_h_dxfg);
163  rhs += ht_rinv_h_dxfg;
164  // Set initial guess
165  dxh = *dxh0_;
166  B.multiply(dxh, *dx);
167  }
168 
169  Log::info() << classname() << " rhs" << rhs << std::endl;
170 
171  // Set J[0] = 0.5 (x_i - x_b)^T B^{-1} (x_i - x_b) + 0.5 d^T R^{-1} d
172  const double costJ0Jb = costJ0Jb_;
173  const double costJ0JoJc = J_.getCostJoJc();
174 
175  // Solve the linear system
176  double reduc = this->solve(*dx, dxh, rhs, B, HtRinvH, costJ0Jb, costJ0JoJc,
177  ninner, gnreduc);
178 
179  Log::test() << classname() << ": reduction in residual norm = " << reduc
180  << std::endl;
181  Log::info() << classname() << " output increment:" << *dx << std::endl;
182 
183  if (lclassicChaVar_) {
184  // dx = Bdx'
185 
186  // Update gradJb_
187  *gradJb_ += dxh;
188  dxh_.push_back(dxh);
189  // Update Jb component of J[0]: 0.5 (x_i - x_b)^T B^-1 (x_i - x_b)
190  costJ0Jb_ += 0.5 * dot_product(*dx, dxh);
191  for (unsigned int jouter = 1; jouter < dxh_.size(); ++jouter) {
192  CtrlInc_ dxhtmp(dx->geometry(), dxh_[jouter - 1]);
193  costJ0Jb_ += dot_product(*dx, dxhtmp);
194  }
195  } else {
196  // dx = Bdx' + x_b - x_fg
197 
198  // Save inital guess for next minimization
199  *dxh0_ = dxh;
200  // Update Jb component of J[0]: 0.5 (x_i - x_b)^T B^-1 (x_i - x_b)
201  costJ0Jb_ = 0.5 * dot_product(*dx, dxh);
202 
203  // Tranform dx_k to be defined w.r.t. the first guess
204  const CtrlInc_ &dxfg = J_.jb().getFirstGuessIncrement();
205  *dx -= dxfg;
206  }
207 
208  return dx;
209 }

213 template <typename MODEL>
214 void DRMinimizer<MODEL>::checkpoint(eckit::LocalConfiguration &conf) const {
215  Log::info() << classname() << "::checkpoint starting" << std::endl;
216 
217  // GradJb
218  ASSERT(conf.has("gradJb"));
219  eckit::LocalConfiguration gradjbconf(conf, "gradJb");
220  gradJb_->write(gradjbconf);
221  Log::info() << classname() << "::checkpoint gradJb: "
222  << util::full_precision(dot_product(*gradJb_, *gradJb_))
223  << std::endl;
224  // dxh
225  ASSERT(conf.has("dxh"));
226  std::vector<eckit::LocalConfiguration> dxhconfs;
227  conf.get("dxh", dxhconfs);
228  ASSERT(dxhconfs.size() == dxh_.size());
229  for (unsigned int jj = 0; jj < dxhconfs.size(); ++jj) {
230  dxh_[jj].write(dxhconfs[jj]);
231  Log::info() << classname() << "::checkpoint dxh[" << jj << "]: "
232  << util::full_precision(dot_product(dxh_[jj], dxh_[jj]))
233  << std::endl;
234  }
235  // costJ0Jb
236  conf.set("costJ0Jb", util::full_precision(costJ0Jb_));
237  Log::info() << classname()
238  << "::checkpoint costJ0Jb: " << util::full_precision(costJ0Jb_)
239  << std::endl;
240 
241  Log::info() << classname() << "::checkpoint done" << std::endl;
242 }

246 template <typename MODEL>
247 void DRMinimizer<MODEL>::restart(const eckit::Configuration &conf) {
248  Log::info() << classname() << "::restart starting" << std::endl;
249 
250  // GradJb
251  ASSERT(conf.has("gradJb"));
252  eckit::LocalConfiguration gradjbconf(conf, "gradJb");
253  if (gradJb_ == 0) {
254  gradJb_.reset(new CtrlInc_(J_.jb()));
255  }
256  gradJb_->read(gradjbconf);
257  Log::info() << classname() << "::restart gradJb: "
258  << util::full_precision(dot_product(*gradJb_, *gradJb_))
259  << std::endl;
260  // dxh
261  ASSERT(conf.has("dxh"));
262  std::vector<eckit::LocalConfiguration> dxhconfs;
263  conf.get("dxh", dxhconfs);
264  for (unsigned int jj = 0; jj < dxhconfs.size(); ++jj) {
265  CtrlInc_ dxh(J_.jb());
266  dxh.read(dxhconfs[jj]);
267  dxh_.push_back(dxh);
268  Log::info() << classname() << "::restart dxh[" << jj << "]: "
269  << util::full_precision(dot_product(dxh_[jj], dxh_[jj]))
270  << std::endl;
271  }
272  // costJ0Jb
273  ASSERT(conf.has("costJ0Jb"));
274  costJ0Jb_ = conf.getDouble("costJ0Jb");
275  Log::info() << classname()
276  << "::restart costJ0Jb: " << util::full_precision(costJ0Jb_)
277  << std::endl;
278 
279  Log::info() << classname() << "::restart done" << std::endl;
280 }
