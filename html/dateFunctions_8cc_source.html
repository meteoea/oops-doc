28 uint64_t dateToJulian(const int year, const int month, const int day) {
29  // Compute the Julian Day number applying the following formula
30  //
31  // julian_day = ( 1461 * ( y + 4800 + ( m - 14 ) / 12 ) ) / 4 +
32  // ( 367 * ( m - 2 - 12 * ( ( m - 14 ) / 12 ) ) ) / 12 -
33  // ( 3 * ( ( y + 4900 + ( m - 14 ) / 12 ) / 100 ) ) / 4 +
34  // d - 32075
35 
36  if (!util::datefunctions::validYYYYMMDD(year, month, day)) {
37  Log::error() << "year=" << year << " month=" << month << " day=" << day
38  << std::endl;
39  ABORT("Not a valid date");
40  }
41 
42  int m1 = (month - 14) / 12;
43  int a = (1461 * (year + 4800 + m1)) / 4;
44  int b = (367 * (month - 2 - (12 * m1))) / 12;
45  int m2 = (year + 4900 + m1) / 100;
46  int c = (3 * (m2)) / 4;
47  return static_cast<uint64_t>(a + b - c + day - 32075);
48 }

52 void julianToDate(const uint64_t julian, int& yy, int& mm, int& dd) {
53  uint64_t l = 0;
54  uint64_t n = 0;
55  uint64_t i = 0;
56  uint64_t j = 0;
57  uint64_t day = 0;
58  uint64_t month = 0;
59  uint64_t year = 0;
60 
61  // Modified Julian date
62 
63  l = julian + static_cast<uint64_t>(68569);
64  n = (4 * l) / 146097;
65  l = l - (146097 * n + 3) / 4;
66  i = (4000 * (l + 1)) / 1461001;
67  l = l - (1461 * i) / 4 + 31;
68  j = (80 * l) / 2447;
69  day = l - (2447 * j) / 80;
70  l = j / 11;
71  month = j + 2 - (12 * l);
72  year = 100 * (n - 49) + i + l;
73  if (year <= static_cast<uint64_t>(std::numeric_limits<int>::max())) {
74  dd = static_cast<int>(day);
75  mm = static_cast<int>(month);
76  yy = static_cast<int>(year);
77  } else {
78  Log::error() << "year=" << year << std::endl;
79  ABORT("year out of range");
80  }
81 }

85 int hmsToSeconds(const int hour, const int minute, const int second) {

87  Log::error() << "hour=" << hour << " minute=" << minute
88  << " second=" << second << std::endl;

96 void secondToHms(const int seconds, int& hh, int& mm, int& ss) {
97  const int secondsPerDay = 86400;
98 
99  if (seconds >= 0 && seconds <= secondsPerDay) {
100  int local_sec = seconds;
101  hh = local_sec / 3600;
102  local_sec %= 3600;
103  mm = local_sec / 60;
104  local_sec %= 60;
105  ss = local_sec;
106  } else {
107  Log::error() << "seconds=" << seconds << std::endl;
108  ABORT("seconds out of range");

127 bool validYYYYMMDD(const int year, const int month, const int day) {
128  bool good;
129  good = year >= 0 && year <= 9999 && month >= 1 && month <= 12 && day >= 1 &&
130  day <= 31;
131 
132  if (good) {
133  if (month == 9 || month == 4 || month == 6 || month == 11) {
134  good = (day <= 30);
135  } else if (month != 2) {
136  good = (day <= 31);
137  } else if (util::datefunctions::isLeapYear(year)) {
138  good = (day <= 29);
139  } else {
140  good = (day <= 28);
141  }
142  }
143 
144  return good;
145 }
