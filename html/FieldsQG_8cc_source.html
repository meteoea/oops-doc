167  int nx = -1;

171  qg_field_sizes_f90(keyFlds_, nx, ny, nf, nb);
172  os << std::endl << " Resolution = " << nx << ", " << ny
173  << ", Fields = " << nf << ", " << nb;
174  nf += nb;
175  std::vector<double> zstat(3*nf);
176  qg_field_gpnorm_f90(keyFlds_, nf, zstat[0]);
177  for (int jj = 0; jj < nf; ++jj) {
178  os << std::endl << " Min=" << zstat[3*jj]
179  << ", Max=" << zstat[3*jj+1] << ", RMS=" << zstat[3*jj+2];
180  }
