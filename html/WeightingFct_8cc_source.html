34 WeightFactory::WeightFactory(const std::string &name) {
35  if (!makers_) makers_ = new std::map<std::string, WeightFactory *>();
36 
37  if (makers_->find(name) != makers_->end()) {
38  Log::error() << name << " already registered in weight function factory."
39  << std::endl;
40  ABORT("Element already registered in WeightFactory.");
41  }
42  (*makers_)[name] = this;
43 }

47 WeightingFct *WeightFactory::create(const eckit::Configuration &config) {
48  if (!makers_) makers_ = new std::map<std::string, WeightFactory *>();
49 
50  std::string id = config.getString("type");
51  std::map<std::string, WeightFactory *>::iterator j = makers_->find(id);
52  if (j == makers_->end()) {
53  Log::error() << id << " does not exist in weight function factory."
54  << std::endl;
55  ABORT("Element does not exist in WeightFactory.");
56  }
57  return (*j).second->make(config);
