124 template <typename MODEL>
125 ObsAuxCtlVec<MODEL>::ObsAuxCtlVec(const ObsAuxCovariance_ &cov)
126  : size_(0), keys_(), ctlvecs_() {
127  Log::trace() << "ObsAuxCtlVec<MODEL>::ObsAuxCtlVec starting" << std::endl;
128  util::Timer timer(classname(), "ObsAuxCtlVec");
129 
130  std::vector<eckit::LocalConfiguration> covconfs;
131  cov.config().get("covarianceObsAux", covconfs);
132  size_ = covconfs.size();
133 
134  for (auto const &config : covconfs) {
135  std::string key(config.getString("type"));
136  std::unique_ptr<ObsAuxCtlVecBase<MODEL>> auxctl(
137  ObsAuxCtlVecFactory<MODEL>::create(cov.errorCovariance(key)));
138  ctlvecs_.emplace(key, std::move(auxctl));
139  keys_.push_back(key);
140  }
141  Log::trace() << "ObsAuxCtlVec<MODEL>::ObsAuxCtlVec done" << std::endl;
142 }
