115  : size_(0), ctls_() {

117  util::Timer timer(classname(), "ObsAuxControl");
118 
119  std::vector<eckit::LocalConfiguration> confs;
120  if (conf.has("stateObsAux")) conf.get("stateObsAux", confs);
121  size_ = confs.size();
122 
123  for (auto const &config : confs) {
124  std::string key(config.getString("type"));
125  std::unique_ptr<ObsAuxControlBase<MODEL>> auxctl(
126  ObsAuxControlFactory<MODEL>::create(os, config));
127  ctls_.emplace(key, std::move(auxctl));
