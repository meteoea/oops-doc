62 double IPCG(VECTOR& x, const VECTOR& b, const AMATRIX& A,
63  const PMATRIX& precond, const int maxiter, const double tolerance) {
64  VECTOR ap(x);
65  VECTOR p(x);
66  VECTOR r(x);
67  VECTOR s(x);
68  VECTOR oldr(x);
69  VECTOR w(x);
70  VECTOR v(x); // required for re-orthogonalization
71  VECTOR z(x); // required for re-orthogonalization
72 
73  std::vector<VECTOR> vVEC; // required for re-orthogonalization
74  std::vector<VECTOR> zVEC; // required for re-orthogonalization
75 
76  // Initial residual r = b - Ax
77  r = b;
78  double xnrm2 = dot_product(x, x);
79  if (xnrm2 != 0) {
80  A.multiply(x, s);
81  r -= s;
82  }
83 
84  // s = precond r
85  precond.multiply(r, s);
86 
87  double dotRr0 = dot_product(r, r);
88  double dotSr0 = dot_product(r, s);
89  double normReduction = 1.0;
90  double rdots_old = dotSr0;
91  double rdots = dotSr0;
92 
93  v = r;
94  v *= 1 / sqrt(dotSr0);
95  z = s;
96  z *= 1 / sqrt(dotSr0);
97 
98  vVEC.push_back(v);
99  zVEC.push_back(z);
100 
101  Log::info() << std::endl;
102  for (int jiter = 0; jiter < maxiter; ++jiter) {
103  Log::info() << " IPCG Starting Iteration " << jiter + 1 << std::endl;
104 
105  if (jiter == 0) {
106  p = s;
107  } else {
108  w = r;
109  w -= oldr; // w=r-oldr
110  double beta = dot_product(s, w) / rdots_old;
111  p *= beta;
112  p += s; // p = s + beta*p
113  }
114 
115  A.multiply(p, ap); // ap = Ap
116 
117  oldr = r;
118 
119  double alpha = rdots / dot_product(p, ap);
120 
121  x.axpy(alpha, p); // x = x + alpha*p;
122  r.axpy(-alpha, ap); // r = r - alpha*ap;
123 
124  // Re-orthogonalization
125  for (int iiter = 0; iiter < jiter; ++iiter) {
126  double proj = dot_product(r, zVEC[iiter]);
127  r.axpy(-proj, vVEC[iiter]);
128  }
129 
130  precond.multiply(r, s); // returns s as approximate solve of As=r
131 
132  rdots_old = rdots;
133  rdots = dot_product(r, s);
134 
135  v = r;
136  v *= 1 / sqrt(rdots);
137  z = s;
138  z *= 1 / sqrt(rdots);
139  vVEC.push_back(v);
140  zVEC.push_back(z);
141 
142  normReduction = sqrt(dot_product(r, r) / dotRr0);
143  Log::info() << "IPCG end of iteration " << jiter + 1
144  << ". Norm reduction= " << util::full_precision(normReduction)
145  << std::endl
146  << std::endl;
147 
148  if (normReduction < tolerance) {
149  Log::info() << "IPCG: Achieved required reduction in residual norm."
150  << std::endl;
151  break;
152  }
153  }
154 
155  Log::info() << "IPCG: end" << std::endl;
156 
157  return normReduction;
158 }
