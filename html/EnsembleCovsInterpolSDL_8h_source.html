98 template<typename MODEL>
99 void EnsembleCovsInterpolSDL<MODEL>::linearize(const State_ & xb,
100  const Geometry_ & resol, const eckit::Configuration &) {
101 
102  /*
103  * Build Geometry of the Ensemble
104  */
105  const eckit::LocalConfiguration confNewGeo(config_, "ensemble_resol");
106  resol_ens_.reset(new Geometry_(confNewGeo));
107 
108  /*
109  * Build Interpolation operators
110  */
111  const eckit::LocalConfiguration confInterpol(config_, "interpol");
112  intdir_.reset(new Interpolator_(xb, *resol_ens_, confInterpol, false));
113  intadj_.reset(new Interpolator_(xb, *resol_ens_, confInterpol, true));
114 
115  /*
116  * Compute the ensemble of perturbations at time of xb
117  */
118  ASSERT(xb.validTime() == time_);
119  std::vector < eckit::LocalConfiguration > confEnsemble;
120  config_.get("ensemble", confEnsemble);
121  EnsemblePtr_ ens_k(new Ensemble_(xb.validTime(), confEnsemble[0]));
122  ens_k->linearize(xb, *resol_ens_);
123  EnsemblesCollection_::getInstance().put(xb.validTime(), ens_k);
124 
125  /*
126  * Init Scale decomposition
127  */
128  const eckit::LocalConfiguration conf(config_, "localization");
129  const eckit::LocalConfiguration confScaleDecomposition(conf, "scale_decomposition");
130 
131  // What type of SDL is demanded
132  SDLtype_ = confScaleDecomposition.getInt("formulation");
133 
134  // How many horizontal bands do we have?
135  nhorizBand_ = confScaleDecomposition.getInt("nb_wave_band_peaks");
136 
137  // How many localization lenghts do we have?
138  std::vector < eckit::LocalConfiguration > confHorizBand;
139  confScaleDecomposition.get("horiz_band", confHorizBand);
140  int nband = confHorizBand.size();
141  // Make sure that the above two numbers are equal
142  ASSERT(nband == nhorizBand_);
143 
144  /*
145  * Build Localization
146  */
147  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
148  loc_.emplace_back(new Localization_(*resol_ens_, confHorizBand[horizBandIndex]));
149  }
150 
151  /*
152  * Performed a horizontal scale decomposition of the perturbations
153  */
154 
155  Log::info() << "Scale decomposition of the perturbation object" << std::endl;
156 
157  //Duplication de l'ensemble sur chacune des bandes
158  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[xb.validTime()];
159  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
160  EnsemblePtr_ ens(new Ensemble_(*e));
161  EnsembleKey k(xb.validTime(),horizBandIndex);
162  EnsemblesCollection_::getInstance().put(k, ens);
163  }
164 
165  HorizScaleDecomposition_ fltbg(confScaleDecomposition);
166  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
167  Log::info() << " >> calling horiz_scale_decomposition for ensemble member # " << memberIndex + 1 << std::endl;
168 
169  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
170  EnsembleKey k(xb.validTime(),horizBandIndex);
171  EnsemblePtr_ horizBandEns = EnsemblesCollection_::getInstance()[k];
172  fltbg.collect((*horizBandEns)[memberIndex], horizBandIndex);
173  }
174  fltbg.decompose((*e)[memberIndex], SDLtype_);
175  }
176 
177  Log::trace() << "EnsembleCovsInterpolSDL linearized." << std::endl;
178 }

182 template<typename MODEL>
183 void EnsembleCovsInterpolSDL<MODEL>::multiply(const Increment_ & dxi,
184  Increment_ & dxo) const
185 {
186 
187  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxi.validTime()];
188 
189 
190  // Adjoint interpolation to the resolution of the increment
191  Log::trace() << "Adjoint Interpolation (HR -> LR) of the increment" << std::endl ;
192  const eckit::LocalConfiguration confInterpol(config_, "interpol");
193  const Variables_ var(confInterpol);
194 
195  Increment_ dxi_lr(*resol_ens_, var, dxi.validTime());
196  dxi_lr.zero();
197 
198  intadj_->interpolate(dxi, var, dxi_lr);
199 
200  //
201  // Note on the scale-dependent localization (SDL) formulation...
202  // We use here a B^1/2 approach to save computational cost, since:
203  // With B^1/2, the cost of the SDL approach is proportional to the number of scale-bands
204  // With B , the cost of the SDL approach is proportional to the SQUARE of the number of scale-bands
205  //
206  // Tringlerie : création des IncrEnsCtlVec_ nécessaires
207  // structures intermédiaires entre BT/2 et B1/2
208  // Create one Ksi per ensemble member
209  std::vector<std::unique_ptr< IncrEnsCtlVec_ > > aksi;
210  unsigned ksiIndex;
211 
212  if (SDLtype_ == 1) {
213  // Create one Ksi per ensemble member
214  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
215  IncrEnsCtlVec_* ksi = new IncrEnsCtlVec_(*resol_ens_, *loc_[0]);
216  aksi.emplace_back(ksi);
217  }
218  } else {
219  // Create one Ksi per ensemble member and per horizontal waveband
220  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
221  for (unsigned l = 0; l < e->size(); ++l) {
222  IncrEnsCtlVec_* ksi = new IncrEnsCtlVec_(*resol_ens_, *loc_[0]);
223  aksi.emplace_back(ksi);
224  }
225  }
226  }
227 
228  //
229  // --- Apply B^T/2 ---
230  //
231 
232  Log::info() << "Aplying B^T/2" << std::endl;
233 
234  // loop on horizontal scale band
235  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
236  // loop on ensemble members
237  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
238  if (SDLtype_ == 1) {
239  ksiIndex = memberIndex;
240  } else {
241  ksiIndex = horizBandIndex * e->size() + memberIndex;
242  }
243  aksi[ksiIndex]->zero();
244 
245  Increment_ dx_lr_loc(dxi_lr);
246 
247  EnsembleKey k(dxi_lr.validTime(),horizBandIndex);
248  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
249  dx_lr_loc.schur_product_with((*e_t)[memberIndex]);
250 
251  loc_[horizBandIndex]->multiplySqrtTrans(dx_lr_loc, *(aksi[ksiIndex]));
252 
253  }
254  }
255 
256  //
257  // --- Apply B^1/2 ---
258  //
259 
260  Log::info() << "Aplying B^1/2" << std::endl;
261 
262  Increment_ dx_lr_loc(*resol_ens_, var, dxi.validTime());
263  Increment_ dxo_lr(dx_lr_loc);
264  dxo_lr.zero();
265 
266  // multiply
267  // loop on horizontal scale band
268  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
269  // loop on ensemble members
270  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
271  if (SDLtype_ == 1) {
272  ksiIndex = memberIndex;
273  } else {
274  ksiIndex = horizBandIndex * e->size() + memberIndex;
275  }
276  loc_[horizBandIndex]->multiplySqrt(*(aksi[ksiIndex]), dx_lr_loc);
277 
278  EnsembleKey k(dx_lr_loc.validTime(),horizBandIndex);
279  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
280  dx_lr_loc.schur_product_with((*e_t)[memberIndex]);
281  dxo_lr.axpy(1.0, dx_lr_loc, false);
282  }
283  }
284 
285  // Back to HR
286  Log::trace() << "Direct interpolation (LR -> HR) of the increment " << std::endl ;
287  intdir_->interpolate(dxo_lr, var , dxo);
288 
289 }

302 template<typename MODEL>
303 void EnsembleCovsInterpolSDL<MODEL>::multiplySqrt(const IncrCtlVec_ & dvi, Increment_ & dxo) const {
304 
305  // Interpolation to the resolution of the increment
306  const eckit::LocalConfiguration confInterpol(config_, "interpol");
307  const Variables_ var(confInterpol);
308  Increment_ dx_lr_loc(*resol_ens_, var, dxo.validTime());
309  Increment_ dxo_lr(dx_lr_loc);
310  dxo_lr.zero();
311 
312  Log::info() << "Aplying B^1/2" << std::endl;
313 
314  int nMembers = EnsemblesCollection_::getInstance()[dxo.validTime()]->size();
315 
316  // loop on horizontal scale band
317  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
318  // Get ensemble
319  EnsembleKey k(dx_lr_loc.validTime(),horizBandIndex);
320  EnsemblePtr_ ens_horizBandIndex = EnsemblesCollection_::getInstance()[k];
321  // loop on ensemble members
322  for (unsigned memberIndex = 0; memberIndex < nMembers; ++memberIndex) {
323  loc_[horizBandIndex]->multiplySqrt(dvi.ensCtlVec(memberIndex), dx_lr_loc);
324  dx_lr_loc.schur_product_with((*ens_horizBandIndex)[memberIndex]);
325  dxo_lr.axpy(1.0, dx_lr_loc, false);
326  }
327  }
328 
329  // Back to HR
330  Log::trace() << "Direct interpolation (LR -> HR) of the increment " << std::endl ;
331  intdir_->interpolate(dxo_lr, var , dxo);
332 }

338 
339 
340  // Adjoint interpolation to the resolution of the increment
341  Log::trace() << "Adjoint Interpolation (HR -> LR) of the increment" << std::endl ;
342 
343  const eckit::LocalConfiguration confInterpol(config_, "interpol");
344  const Variables_ var(confInterpol);
345  Increment_ dxi_lr(*resol_ens_, var, dxi.validTime());
346  dxi_lr.zero();
347 
348  intadj_->interpolate(dxi, var, dxi_lr);
349 
350 
351  Log::info() << "Aplying B^T/2" << std::endl;
352 
353  int nMembers = EnsemblesCollection_::getInstance()[dxi.validTime()]->size();
354 
355  // loop on ensemble members
356  for (unsigned memberIndex = 0; memberIndex < nMembers; ++memberIndex) {
357  // Initialization
358  dvo.ensCtlVec(memberIndex).zero();
359  // loop on horizontal scale band
360  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
361 
362  EnsembleKey k(dxi_lr.validTime(),horizBandIndex);

364  Increment_ dx_lr_loc(dxi_lr);
365  dx_lr_loc.schur_product_with((*ens_horizBandIndex)[memberIndex]);

368  }

374 template<typename MODEL>
375 void EnsembleCovsInterpolSDL<MODEL>::randomize(Increment_ & dxo) const {
376 
377  // Build control vector
378  EnsemblePtr_ e_k = EnsemblesCollection_::getInstance()[dxo.validTime()];
379  std::vector<std::unique_ptr< IncrEnsCtlVec_ > > aksi;
380  if (SDLtype_ == 1) {
381  // Create one Ksi per ensemble member
382  for (unsigned memberIndex = 0; memberIndex < e_k->size(); ++memberIndex) {
383  aksi.emplace_back(this->newIncrEnsCtlVec());
384  }
385  } else {
386  // Create one Ksi per ensemble member and per horizontal waveband
387  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
388  for (unsigned memberIndex = 0; memberIndex < e_k->size(); ++memberIndex) {
389  aksi.emplace_back(this->newIncrEnsCtlVec());
390  }
391  }
392  }
393 
394  // Initialize control vector
395  for (unsigned int ksi = 0; ksi < aksi.size(); ++ksi) {
396  aksi[ksi]->random();
397  }
398 
399  // dxo_lr = B^1/2 aksi
400 
401  //(a) prepare interpolation
402  const eckit::LocalConfiguration confInterpol(config_, "interpol");
403  const Variables_ var(confInterpol);
404  Increment_ dx_lr_loc(*resol_ens_, var, dxo.validTime());
405  Increment_ dxo_lr(dx_lr_loc);
406 
407  //(b) multiplication
408  dxo_lr.zero();
409  unsigned ksiIndex;
410  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
411  // loop on ensemble members
412  for (unsigned memberIndex = 0; memberIndex < e_k->size(); ++memberIndex) {
413  if (SDLtype_ == 1) {
414  ksiIndex = memberIndex;
415  } else {
416  ksiIndex = horizBandIndex * e_k->size() + memberIndex;
417  }
418  loc_[horizBandIndex]->multiplySqrt(*aksi[ksiIndex], dx_lr_loc);
419 
420 
421  EnsembleKey k(dxo_lr.validTime(),horizBandIndex);
422  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
423  dx_lr_loc.schur_product_with((*e_t)[memberIndex]);
424  dxo_lr.axpy(1.0, dx_lr_loc, false);
425  }
426  }
427 
428  // Pour test interpolation :
429  //dxo_lr.zero();
430  double product = dot_product(dxo_lr,dxo_lr);
431  Log::info() << "EnsembleCovInterpol<MODEL>::randomize : product=" << product << std::endl ;
432 
433  //(c) interpolation to HR
434  Log::trace() << "Direct interpolation (LR -> HR) of the increment " << std::endl;
435  intdir_->interpolate(dxo_lr, var, dxo);
436 
437  double productHR = dot_product(dxo,dxo);
438  Log::info() << "EnsembleCovInterpol<MODEL>::randomize : productHR=" << productHR << std::endl ;
439 }
