21 subroutine invent_state(flds,config)
22 
23 use qg_configs
24 use qg_fields
25 use iso_c_binding
26 use config_mod
27 use fckit_log_module, only : fckit_log
28 use qg_constants, only: u1,u2,bet,worog, domain_zonal, domain_meridional, &
29  & dlogtheta,f0,g,horog,scale_length,rossby_number
30 use kinds
31 
32 implicit none
33 
34 type(qg_field), intent(inout) :: flds
35 type(c_ptr), intent(in) :: config
36 
37 integer :: jx,jy,icentre,jcentre,ii,jj,ipert
38 real(kind=kind_real) :: distx,disty,d1,d2,f1,f2,rsmax,deltax0,deltay0,deltax,deltay,zz
39 real(kind=kind_real), allocatable :: pv(:,:,:), rs(:,:)
40 character(len=160) :: record
41 
42 ! ------------------------------------------------------------------------------
43 
44 d1 = config_get_real(config,"top_layer_depth")
45 d2 = config_get_real(config,"bottom_layer_depth")
46 
47 f1 = f0*f0*scale_length*scale_length/(g*dlogtheta*d1)
48 f2 = f0*f0*scale_length*scale_length/(g*dlogtheta*d2)
49 rsmax = horog/(rossby_number*d2)
50 deltax0 = domain_zonal/real(flds%nx,kind_real)
51 deltay0 = domain_meridional/real(flds%ny+1,kind_real)
52 deltax = deltax0/scale_length
53 deltay = deltay0/scale_length
54 
55 allocate(pv(flds%nx,flds%ny,2))
56 allocate(rs(flds%nx,flds%ny))
57 
58 !--- Uniform wind in each layer. A vertical shear outside the range
59 !--- -bet/F1 to bet/F2 should produce baroclinic instability.
60 
61 flds%x_south(1) = 0.0_kind_real
62 flds%x_south(2) = 0.0_kind_real
63 flds%x_north(1) = -real(flds%ny+1,kind_real)*deltay*u1
64 flds%x_north(2) = -real(flds%ny+1,kind_real)*deltay*u2
65 
66 do jy=1,flds%ny
67 do jx=1,flds%nx
68  flds%x(jx,jy,1) = -real(jy,kind_real)*deltay*u1
69  flds%x(jx,jy,2) = -real(jy,kind_real)*deltay*u2
70 enddo
71 enddo
72 
73 ipert = config_get_int(config,"perturb")
74 if (ipert/=0) then
75  write(record,*)"qg_invent_state_f90: Perturbing invented state by ",ipert,"%."
76  call fckit_log%info(record)
77  zz=real(ipert,kind_real)/100.0_kind_real
78  do jy=1,flds%ny
79  do jx=1,flds%nx
80  flds%x(jx,jy,1) = (1.0_kind_real+zz)*flds%x(jx,jy,1)
81  flds%x(jx,jy,2) = (1.0_kind_real-zz)*flds%x(jx,jy,2)
82  enddo
83  enddo
84 endif
85 
86 icentre=flds%nx/4
87 jcentre=3*flds%ny/4
88 do jj=1,flds%ny
89  do ii=1,flds%nx
90  distx = real(min(icentre-ii,flds%nx-(icentre-ii)),kind_real) * deltax0
91  disty = real(abs(jj-jcentre),kind_real) * deltay0
92  rs(ii,jj) = rsmax*exp(-(distx*distx+disty*disty)/(worog*worog))
93  enddo
94 enddo
95 
96 call calc_pv(flds%nx,flds%ny,pv,flds%x,flds%x_north,flds%x_south, &
97  & f1,f2,deltax,deltay,bet,rs)
98 
99 do jx=1,flds%nx
100  flds%q_south(jx,1) = 2.0_kind_real*pv(jx,1,1)-pv(jx,2,1)
101  flds%q_south(jx,2) = 2.0_kind_real*pv(jx,1,2)-pv(jx,2,2)
102 enddo
103 do jx=1,flds%nx
104  flds%q_north(jx,1) = 2.0_kind_real*pv(jx,flds%ny,1)-pv(jx,flds%ny-1,1)
105  flds%q_north(jx,2) = 2.0_kind_real*pv(jx,flds%ny,2)-pv(jx,flds%ny-1,2)
106 enddo
107 
108 deallocate(pv,rs)
109 ! ------------------------------------------------------------------------------
110 
111 return
112 end subroutine invent_state
