38 template <typename MODEL>
39 class DualConfigForecast : public Application {
40  using DualConfigGeometry_ = DualConfigGeometry<MODEL>;
41  using DualConfigModel_ = DualConfigModel<MODEL>;
42  using DualConfigModelAux_ = DualConfigModelAuxControl<MODEL>;
43  using DualConfigState_ = DualConfigState<MODEL>;
44  using Geometry_ = Geometry<MODEL>;
45  using Model_ = Model<MODEL>;
46  using ModelAux_ = ModelAuxControl<MODEL>;
47  using State_ = State<MODEL>;
48 
49  public:
50  // -----------------------------------------------------------------------------
51  DualConfigForecast() {}
52  // -----------------------------------------------------------------------------
53  virtual ~DualConfigForecast() {}
54  // -----------------------------------------------------------------------------
55  int execute(const eckit::Configuration& fullConfig) const {
56  // Setup resolution
57  const eckit::LocalConfiguration resolConfig(fullConfig, "resolution");
58  const DualConfigGeometry_ resol(resolConfig);
59 
60  // Setup Model
61  const eckit::LocalConfiguration modelConfig(fullConfig, "model");
62  const DualConfigModel_ model(resol, modelConfig);
63 
64  // Setup initial state
65  const eckit::LocalConfiguration initialConfig(fullConfig, "initial");
66  DualConfigState_ xx(resol, model, initialConfig);
67  Log::test() << "Initial state: " << xx.norm() << std::endl;
68 
69  // Setup augmented state
70  const DualConfigModelAux_ moderr(resol, model, initialConfig);
71 
72  // Setup times
73  const util::Duration fclength(fullConfig.getString("forecast_length"));
74  const util::DateTime bgndate(xx.validTime1());
75  const util::DateTime enddate(bgndate + fclength);
76  Log::info() << "Running dual-config forecast from " << bgndate << " to "
77  << enddate << std::endl;
78 
79  // Setup forecast outputs
80  DualConfigPostProcessor<State_> post;
81 
82  eckit::LocalConfiguration prtConfig;
83  eckit::LocalConfiguration prtConfig1;
84  eckit::LocalConfiguration prtConfig2;
85  if (fullConfig.has("prints")) {
86  prtConfig = eckit::LocalConfiguration(fullConfig, "prints");
87  prtConfig1 = eckit::LocalConfiguration(prtConfig, "prints1");
88  prtConfig2 = eckit::LocalConfiguration(prtConfig, "prints2");
89  }
90  post.post1().enrollProcessor(new StateInfo<State_>("fc", prtConfig1));
91  post.post2().enrollProcessor(new StateInfo<State_>("fc", prtConfig2));
92 
93  const eckit::LocalConfiguration outConfig(fullConfig, "output");
94  const eckit::LocalConfiguration outConfig1(outConfig, "output1");
95  const eckit::LocalConfiguration outConfig2(outConfig, "output2");
96  post.post1().enrollProcessor(new StateWriter<State_>(bgndate, outConfig1));
97  post.post2().enrollProcessor(new StateWriter<State_>(bgndate, outConfig2));
98 
99  // Run forecast
100  model.forecast(xx, moderr, fclength, post);
101 
102  Log::test() << "Final state: " << xx.norm() << std::endl;
103 
104  return 0;
105  }
106  // -----------------------------------------------------------------------------
107  private:
108  std::string appname() const {
109  return "oops::DualConfigForecast<" + MODEL::name() + ">";
110  }
111  // -----------------------------------------------------------------------------
112 };
