50 void ModelQG::step(StateQG & xx, const ModelBias & bias) const {
51  ASSERT(xx.fields().isForModel(true));
52  Log::debug() << "ModelQG::step fields in" << xx.fields() << std::endl;
53  qg_propagate_f90(keyConfig_, xx.fields().toFortran());
54 
55  if (bias.isActive()){
56  xx.fields() += bias.fields();
57  }
58  xx.validTime() += tstep_;

60 }
