144 template <typename MODEL>
145 ObsAuxIncrementBase<MODEL> *ObsAuxIncrementFactory<MODEL>::create(
146  const ObservationSpace<MODEL> &os, const eckit::Configuration &conf) {
147  const std::string id = conf.getString("type");
148  typename std::map<std::string, ObsAuxIncrementFactory<MODEL> *>::iterator
149  jloc = getMakers().find(id);
150  if (jloc == getMakers().end()) {
151  Log::error() << id << " does not exist in ObsAuxIncrementFactory."
152  << std::endl;
153  ABORT("Element does not exist in ObsAuxIncrementFactory.");
154  }
155  ObsAuxIncrementBase<MODEL> *ptr = jloc->second->make(os, conf);
156  return ptr;
157 }
