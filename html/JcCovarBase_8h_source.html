140 template <typename MODEL>
141 JcCovarBase<MODEL> *JcCovarFactory<MODEL>::create(
142  const State_ &state, const Geometry_ &geom,
143  const eckit::Configuration &conf) {
144  Log::trace() << "JcCovarBase<MODEL>::create starting" << std::endl;
145  const std::string id = conf.getString("version");
146  typename std::map<std::string, JcCovarFactory<MODEL> *>::iterator jerr =
147  getMakers().find(id);
148  if (jerr == getMakers().end()) {
149  Log::error() << id << " does not exist in the Jc covariances factory."
150  << std::endl;
151  ABORT("Element does not exist in JcCovarFactory.");
152  }
153  JcCovarBase<MODEL> *ptr = jerr->second->make(state, geom, conf);
154  Log::trace() << "JcCovarBase<MODEL>::create done" << std::endl;
155  return ptr;
156 }
