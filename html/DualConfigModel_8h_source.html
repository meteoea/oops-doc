130 template <typename MODEL>
131 void DualConfigModel<MODEL>::forecast(
132  DualConfigState_ &xx, const DualConfigModelAuxControl_ &maux,
133  const util::Duration &len, DualConfigPostProcessor<State_> &post) const {
134  Log::trace() << classname() << "::forecast starting" << std::endl;
135  util::Timer timer(classname(), "forecast");
136  util::LogbookWriter<std::string> log("Forecast", "Nonlinear HR trajectory");
137 
138  const util::DateTime end(xx.validTime1() + len);
139  Log::info() << classname() << "::forecast: starting 1: " << xx.state1()
140  << std::endl;
141  Log::info() << classname() << "::forecast: starting 2: " << xx.state2()
142  << std::endl;
143 
145  this->initialize(xx, maux);
146  post.post1().initialize(xx.state1(), end, model1_->timeResolution());
147  post.post2().initialize(xx.state2(), end, model2_->timeResolution());
148 
150  while (xx.validTime1() < end) {
151  Log::info() << classname() << "::step starting" << std::endl
152  << " t1 = " << xx.validTime1() << std::endl
153  << " t2 = " << xx.validTime2() << std::endl;
154 
156  if (xx.validTime2() == xx.validTime1()) {
158  xx.state2().forceWith(xx.state1(), varsForced2_);
160  xx.state1().forceWith(xx.state2(), varsForced1_);
161  }
162 
164  this->step(xx, maux);
165 
167  post.post1().process(xx.state1());
168  if (xx.validTime2() == xx.validTime1()) post.post2().process(xx.state2());
169 
170  Log::info() << classname() << "::step done" << std::endl;
171  }
172 
174  post.post1().finalize(xx.state1());
175  post.post2().finalize(xx.state2());
176  this->finalize(xx);
177 
178  Log::info() << classname() << "::forecast: forecast finished: " << xx
179  << std::endl;
180  ASSERT(xx.validTime1() == end);
181  ASSERT(xx.validTime2() == end);
182  Log::trace() << classname() << "::forecast done" << std::endl;
183 }
