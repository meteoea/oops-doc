78 template <typename MODEL>
79 DualVector<MODEL>::DualVector(const DualVector &other)
80  : dxjb_(),
81  dxjo_(),
82  dxjc_(),
83  ijo_(other.ijo_),
84  ijc_(other.ijc_),
85  size_(other.size_) {
86  if (other.dxjb_ != 0) {
87  dxjb_.reset(new CtrlInc_(*other.dxjb_));
88  }
89  for (unsigned jj = 0; jj < other.dxjo_.size(); ++jj) {
90  std::shared_ptr<Departures_> pd(new Departures_(*other.dxjo_[jj]));
91  dxjo_.push_back(pd);
92  }
93  for (unsigned jj = 0; jj < other.dxjc_.size(); ++jj) {
94  std::shared_ptr<Increment_> pi(new Increment_(*other.dxjc_[jj]));
95  dxjc_.push_back(pi);
96  }
97 }

109 template <typename MODEL>
110 void DualVector<MODEL>::append(GeneralizedDepartures *pv) {
111  // Since there is no duck-typing in C++, we do it manually.
112  Increment_ *pi = dynamic_cast<Increment_ *>(pv);
113  if (pi != 0) {
114  std::shared_ptr<Increment_> si(pi);
115  dxjc_.push_back(si);
116  ijc_.push_back(size_);
117  }
118  Departures_ *pd = dynamic_cast<Departures_ *>(pv);
119  if (pd != 0) {
120  std::shared_ptr<Departures_> sd(pd);
121  dxjo_.push_back(sd);
122  ijo_.push_back(size_);
123  }
124  ASSERT(pi != 0 || pd != 0);
125  ++size_;
126 }

129 std::shared_ptr<const GeneralizedDepartures> DualVector<MODEL>::getv(
130  const unsigned ii) const {
131  ASSERT(ii < size_);
132  std::shared_ptr<const GeneralizedDepartures> pv;
133  for (unsigned jj = 0; jj < ijo_.size(); ++jj) {
134  if (ijo_[jj] == ii) pv = dxjo_[jj];
135  }
136  for (unsigned jj = 0; jj < ijc_.size(); ++jj) {
137  if (ijc_[jj] == ii) pv = dxjc_[jj];
138  }
139  ASSERT(pv != 0);
140  return pv;
