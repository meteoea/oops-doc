115 template <typename VECTOR>
116 void SpectralLMP<VECTOR>::update(std::vector<std::unique_ptr<VECTOR> > &Zv,
117  std::vector<std::unique_ptr<VECTOR> > &Zhl,
118  std::vector<std::unique_ptr<VECTOR> > &Zl,
119  std::vector<double> &alphas,
120  std::vector<double> &betas) {
121  // If useoldpairs = false, use only current information
122  if (!useoldpairs_) {
123  eigvals_.clear();
124  omega_.clear();
125  X_.clear();
126  U_.clear();
127  usedpairIndx_.clear();
128  Zlast_.clear();
129  Zhlast_.clear();
130  }
131 
132  ++update_;
133  Log::debug() << "SpectralLMP size Zl = " << Zl.size() << std::endl;
134  Log::debug() << "SpectralLMP size Zhl = " << Zhl.size() << std::endl;
135  Log::debug() << "SpectralLMP size alpha = " << alphas.size() << std::endl;
136  Log::debug() << "SpectralLMP size beta = " << betas.size() << std::endl;
137  const unsigned nvec = Zl.size();
138  ASSERT(Zhl.size() == nvec);
139 
140  if (nvec > 0) {
141  Log::info() << std::endl;
142  Log::info() << "SpectralLMP: update " << update_ << std::endl;
143  ASSERT(alphas.size() == nvec - 1);
144  ASSERT(betas.size() == nvec - 1);
145  std::vector<double> evals;
146  std::vector<std::vector<double> > evecs;
147 
148  // Compute spectrum of tri-diagonal matrix
149  TriDiagSpectrum(alphas, betas, evals, evecs);
150 
151  // Determine the converged eigenvalues
152  std::vector<unsigned> convIndx;
153  unsigned oldpairIndx = 0;
154  for (unsigned jiter = 0; jiter < nvec - 1; ++jiter) {
155  double erritz = evecs[jiter][nvec - 2] * betas[nvec - 2];
156  double lambda = evals[jiter];
157  // Add new converged eigenvalues
158  if (std::abs(erritz) <= 0.001 * lambda) {
159  convIndx.push_back(jiter);
160  eigvals_.push_back(lambda);
161  omega_.push_back(erritz / lambda);
162  // If maximum number of pairs are exceeded, remove old information
163  // from
164  // the beginning
165  if (convIndx.size() > maxpairs_) {
166  convIndx.erase(convIndx.begin());
167  }
168  if (eigvals_.size() > maxpairs_) {
169  eigvals_.erase(eigvals_.begin());
170  omega_.erase(omega_.begin());
171  oldpairIndx += 1;
172  }
173  }
174  }
175  // Keep information on how many pairs are used at each minimization loop
176  usedpairIndx_.push_back(convIndx.size());
177  unsigned reduc = oldpairIndx;
178  for (unsigned kiter = 0; kiter < usedpairIndx_.size() - 1; ++kiter) {
179  if (reduc <= usedpairIndx_[kiter]) {
180  usedpairIndx_[kiter] -= reduc;
181  break;
182  } else {
183  reduc -= usedpairIndx_[kiter];
184  usedpairIndx_[kiter] = 0;
185  }
186  }
187  unsigned counter = 0;
188  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
189  counter += usedpairIndx_[kiter];
190  }
191  ASSERT(counter <= maxpairs_);
192  Log::info() << "SpectralLMP: Number of inner iterations : "
193  << Zl.size() - 1 << std::endl;
194  Log::info() << "SpectralLMP: Number of maximum pairs : "
195  << maxpairs_ << std::endl;
196  Log::info() << "SpectralLMP: Number of used total pairs : " << counter
197  << std::endl;
198  Log::info() << "SpectralLMP: Number of new converged pairs : "
199  << convIndx.size() << std::endl;
200  Log::info() << "SpectralLMP: Number of removed old pairs : "
201  << oldpairIndx << std::endl;
202  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
203  Log::info() << "SpectralLMP: Number of used pairs from outer loop "
204  << kiter + 1 << " : " << usedpairIndx_[kiter] << std::endl;
205  }
206 
207  // Calculate/Update the matrix X and U
208 
209  if (X_.size() != 0) {
210  // Remove the first oldpairIndx elements
211  unsigned minsize = std::min(oldpairIndx, maxpairs_);
212  unsigned xsize = X_.size();
213  X_.erase(X_.begin(), X_.begin() + std::min(xsize, minsize));
214  U_.erase(U_.begin(), U_.begin() + std::min(xsize, minsize));
215  }
216  for (unsigned jiter = 0; jiter < convIndx.size(); ++jiter) {
217  VECTOR *ww = new VECTOR(*Zl[0], false);
218  ww->zero();
219  for (unsigned iiter = 0; iiter < nvec - 1; ++iiter) {
220  ww->axpy(evecs[convIndx[jiter]][iiter], *Zl[iiter]);
221  }
222  // Add new information
223  X_.emplace_back(ww);
224  }
225  for (unsigned jiter = 0; jiter < convIndx.size(); ++jiter) {
226  VECTOR *ww = new VECTOR(*Zl[0], false);
227  ww->zero();
228  for (unsigned iiter = 0; iiter < nvec - 1; ++iiter) {
229  ww->axpy(evecs[convIndx[jiter]][iiter], *Zhl[iiter]);
230  }
231  // Add new information
232  U_.emplace_back(ww);
233  }
234 
235  if (RitzPrecond_) {
236  Zlast_.emplace_back(*Zl[nvec - 1]);
237  Zhlast_.emplace_back(*Zhl[nvec - 1]);
238 
239  // Calculate the matrix Y = [U1*omega1, ..., Uk*omegak]
240  Y_.clear();
241  zcount.clear();
242  unsigned kk = 0;
243  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
244  if (usedpairIndx_[kiter] != 0) {
245  zcount.push_back(kiter);
246  VECTOR *ww = new VECTOR(*Zl[0], false);
247  for (unsigned jiter = 0; jiter < usedpairIndx_[kiter]; ++jiter) {
248  ww->axpy(omega_[jiter + kk], *U_[jiter + kk]);
249  }
250  kk += usedpairIndx_[kiter];
251  Y_.emplace_back(ww);
252  }
253  }
254 
255  // Calculate the matrix S = [X1*omega1, ..., Xk*omegak]
256  S_.clear();
257  kk = 0;
258  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
259  if (usedpairIndx_[kiter] != 0) {
260  VECTOR *ww = new VECTOR(*Zl[0], false);
261  for (unsigned jiter = 0; jiter < usedpairIndx_[kiter]; ++jiter) {
262  ww->axpy(omega_[jiter + kk], *X_[jiter + kk]);
263  }
264  kk += usedpairIndx_[kiter];
265  S_.emplace_back(ww);
266  }
267  }
268  }
269 
270  // Release remaining memory
271  Zv.clear();
272  Zl.clear();
273  Zhl.clear();
274  alphas.clear();
275  betas.clear();
276  }
277 }

281 template <typename VECTOR>
282 void SpectralLMP<VECTOR>::multiply(const VECTOR &a, VECTOR &b) const {
283  b = a; // P_0 = I
284 
285  for (unsigned iiter = 0; iiter < eigvals_.size(); ++iiter) {
286  double zeval = std::min(10.0, eigvals_[iiter]);
287  // double zeval = eigvals_[iiter];
288  double zz = 1.0 / zeval - 1.0;
289  b.axpy(zz * dot_product(a, *X_[iiter]), *U_[iiter]);
290  }
291 
292  if (RitzPrecond_ && eigvals_.size() != 0) {
293  // Sk'*a
294  std::vector<double> sta;
295  sta.clear();
296  for (unsigned iiter = 0; iiter < S_.size(); ++iiter) {
297  sta.push_back(dot_product(*S_[iiter], a));
298  }
299 
300  // Yk (sta(k) - Zlast' a)
301  for (unsigned kiter = 0; kiter < S_.size(); ++kiter) {
302  for (unsigned iiter = 0; iiter < eigvals_.size(); ++iiter) {
303  double wxap = sta[kiter] - dot_product(a, Zlast_[zcount[kiter]]);
304  b.axpy(wxap, *Y_[kiter]);
305  }
306  }
307 
308  // -Zhlast sta
309  for (unsigned kiter = 0; kiter < S_.size(); ++kiter) {
310  b.axpy(-sta[kiter], Zhlast_[zcount[kiter]]);
311  }
312  }
313 }
