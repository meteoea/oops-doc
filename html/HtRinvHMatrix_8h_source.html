65 template <typename MODEL>
66 void HtRinvHMatrix<MODEL>::multiply(const CtrlInc_& dx, CtrlInc_& dz) const {
67  // Increment counter
68  iter_++;
69 
70  // Setup TL terms of cost function
71  PostProcessorTL<Increment_> costtl;
72  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
73  costtl.enrollProcessor(j_.jterm(jj).setupTL(dx));
74  }
75 
76  // Run TLM
77  CtrlInc_ mdx(dx);
78  j_.runTL(mdx, costtl);
79 
80  // Get TLM outputs, multiply by covariance inverses, and setup ADJ forcing
81  // terms
82  j_.zeroAD(dz);
83  PostProcessorAD<Increment_> costad;
84 
85  DualVector<MODEL> ww;
86  DualVector<MODEL> zz;
87 
88  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
89  ww.append(costtl.releaseOutputFromTL(jj));
90  zz.append(j_.jterm(jj).multiplyCoInv(*ww.getv(jj)));
91  costad.enrollProcessor(j_.jterm(jj).setupAD(zz.getv(jj), dz));
92  }
93 
94  // Run ADJ
95  j_.runAD(dz, costad);
96 
97  if (test_) {
98  // <G dx, dy>, where dy = Rinv H dx
99  double adj_tst_fwd = dot_product(ww, zz);
100  // <dx, Gt dy> , where dy = Rinv H dx
101  double adj_tst_bwd = dot_product(dx, dz);
102 
103  Log::info() << "Online adjoint test, iteration: " << iter_ << std::endl
104  << util::PrintAdjTest(adj_tst_fwd, adj_tst_bwd, "G")
105  << std::endl;
106  }
107 }
