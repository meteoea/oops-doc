113 std::vector<int> ObsTable::timeSelect(const util::DateTime & t1,
114  const util::DateTime & t2) const {
115  std::vector<int> mask;
116  for (unsigned int jobs = 0; jobs < nobs(); ++jobs)
117  if (times_[jobs] > t1 && times_[jobs] <= t2) mask.push_back(jobs);
118  return mask;

123 void ObsTable::generateDistribution(const eckit::Configuration & config) {
124  oops::Log::trace() << "ObsTable::generateDistribution starting" << std::endl;
125 
126  util::Duration first(config.getString("begin"));
127  util::Duration last(winend_-winbgn_);
128  if (config.has("end")) {
129  last = util::Duration(config.getString("end"));
130  }
131  util::Duration freq(config.getString("obs_frequency"));
132 
133  int nobstimes = 0;
134  util::Duration step(first);
135  while (step <= last) {
136  ++nobstimes;
137  step += freq;
138  }
139 
140  const unsigned int nobs_locations = config.getInt("obs_density");
141  const unsigned int nobs = nobs_locations*nobstimes;
142  double dx = 1.0/static_cast<double>(nobs_locations);
143  oops::Log::trace() << "ObservationL95:generateDistribution nobs=" << nobs << std::endl;
144 
145  times_.resize(nobs);
146  locations_.resize(nobs);
147 
148  unsigned int iobs = 0;
149  util::DateTime now(winbgn_);
150  step = first;
151  while (step <= last) {
152  now = winbgn_ + step;
153  for (unsigned int jobs = 0; jobs < nobs_locations; ++jobs) {
154  double xpos = jobs*dx;
155  times_[iobs] = now;
156  locations_[iobs] = xpos;
157  ++iobs;
158  }
159  step += freq;
160  }
161  ASSERT(iobs == nobs);
162 
163 // Generate obs error
164  const double err = config.getDouble("obs_error");
165  std::vector<double> obserr(nobs);
166  for (unsigned int jj = 0; jj < nobs; ++jj) {
167  obserr[jj] = err;
168  }
169  this->putdb("ObsErr", obserr);
170 
171  oops::Log::trace() << "ObsTable::generateDistribution done" << std::endl;
172 }

214 void ObsTable::otOpen(const std::string & filename) {
215  oops::Log::trace() << "ObsTable::ot_read starting" << std::endl;
216  std::ifstream fin(filename.c_str());
217  if (!fin.is_open()) ABORT("ObsTable::otOpen: Error opening file");
218 
219  int ncol, nobs;
220  fin >> ncol;
221 
222  std::vector<std::string> colnames;
223  for (int jc = 0; jc < ncol; ++jc) {
224  std::string col;
225  fin >> col;
226  colnames.push_back(col);
227  }
228 
229  fin >> nobs;
230  locations_.resize(nobs);
231  std::vector<double> newcol(nobs);
232  for (int jc = 0; jc < ncol; ++jc) {
233  ASSERT(data_.find(colnames[jc]) == data_.end());
234  data_.insert(std::pair<std::string, std::vector<double> >(colnames[jc], newcol));
235  }
236 
237  times_.clear();
238  int jjj;
239  for (int jobs = 0; jobs < nobs; ++jobs) {
240  fin >> jjj;
241  ASSERT(jjj == jobs);
242  std::string sss;
243  fin >> sss;
244  util::DateTime ttt(sss);
245  times_.push_back(ttt);
246  fin >> locations_[jobs];
247  for (std::map<std::string, std::vector<double> >::iterator jo = data_.begin();
248  jo != data_.end(); ++jo)
249  fin >> jo->second[jobs];
250  }
251 
252  fin.close();
253  oops::Log::trace() << "ObsTable::ot_read done" << std::endl;
254 }

258 void ObsTable::otWrite(const std::string & filename,
259  const bool & writeScreened) const {
260  oops::Log::trace() << "ObsTable::otWrite writing " << filename << std::endl;
261  std::ofstream fout(filename.c_str());
262  if (!fout.is_open()) ABORT("ObsTable::otWrite: Error opening file");
263 
264  std::vector<util::DateTime> * times =0;
265  std::vector<double> * locations =0;
266  std::map<std::string, std::vector<double> > * data =0;
267 
268  // set pointers
269  if (writeScreened) {
270  times = &screenedTimes_;
271  locations = &screenedLocations_;
272  data = &screenedData_;
273  } else {
274  times = &times_;
275  locations = &locations_;
276  data = &data_;
277  }
278  ASSERT(times);
279  ASSERT(locations);
280  ASSERT(data);
281 
282  int ncol = data->size();
283  fout << ncol << std::endl;
284 
285  for (auto const & vec: *data)
286  fout << vec.first << std::endl;
287 
288  int nobs = times->size();
289  fout << nobs << std::endl;
290  for (int jobs = 0; jobs < nobs; ++jobs) {
291  fout << jobs;
292  fout << " " << (*times)[jobs];
293  fout << " " << (*locations)[jobs];
294  for (auto const & vec: *data)
295  fout << " " << vec.second[jobs];
296  fout << std::endl;
297  }
298 
299  // reset pointers
300  times =0;
301  locations =0;
302  data =0;
303 
304  fout.close();
305  oops::Log::trace() << "ObsTable::otWrite done" << std::endl;
306 }
