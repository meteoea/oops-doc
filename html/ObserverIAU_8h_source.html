77 template <typename MODEL, typename STATE>
78 void ObserverIAU<MODEL, STATE>::doInitialize(STATE &xx,
79  const util::DateTime &end,
80  const util::Duration &tstep) {
81  const util::Duration timestep(config_.getString("frequency"));
82  const util::Duration step(config_.getString("increment_frequency"));
83  const util::Duration halftstep = step / 2;
84  util::DateTime now(xx.validTime());
85 
86  std::vector<eckit::LocalConfiguration> confs;
87  config_.get("increment", confs);
88 
89  const Variables_ vars(eckit::LocalConfiguration(config_, "variables"));
90  const Geometry_ resol(eckit::LocalConfiguration(config_, "resolution"));
91  Increment_ stp(resol, vars, xx.validTime());
92 
93  // Read increments
94  MapTime_ bgntime, endtime;
95  for (unsigned int jm = 0; jm < confs.size(); ++jm) {
96  Log::info() << "ObserverIAU: Read increment at: " << now << std::endl;
97  std::shared_ptr<Increment_> dx(new Increment_(resol, vars, xx.validTime()));
98 
99  const eckit::LocalConfiguration stpConfig(confs[jm], "starting_point");
100  const eckit::LocalConfiguration anaConfig(confs[jm], "analysis");
101 
102  stp.read(stpConfig);
103  dx->read(anaConfig);
104  (*dx) -= stp;
105  increments_[now] = dx;
106 
107  bgntime.emplace(now, now - halftstep);
108  endtime.emplace(now, bgntime[now] + step);
109  now += step;
110  }
111 
112  // Initialize IAU steps
113  util::DateTime lasttime(now - step);
114  const util::DateTime firsttime(xx.validTime() + timestep);
115 
116  if (timestep.toSeconds() == 0) {
117  steps_.emplace(firsttime, firsttime);
118  } else {
119  if (confs.size() == 1) {
120  lasttime = firsttime + halftstep;
121  }
122  for (util::DateTime tm(firsttime); tm <= lasttime; tm += timestep) {
123  for (util::DateTime t(xx.validTime()); t <= lasttime; t += step) {
124  if ((tm > bgntime[t]) && (tm <= endtime[t])) {
125  steps_.emplace(tm, t);
126  }
127  }
128  Log::info() << "ObserverIAU: tstep:" << tm << " increment:" << steps_[tm]
129  << std::endl;
130  }
131 
132  util::Duration wlen(lasttime - xx.validTime());
133  int nb = (wlen.toSeconds() / timestep.toSeconds());
134  double weight = 1.0 / ((double)nb);
135  Log::info() << "ObserverIAU: weight: " << weight << std::endl;
136  for (util::DateTime t(xx.validTime()); t <= lasttime; t += step) {
137  (*increments_[t]) *= weight;
138  }
139 
140  if (config_.has("inflation")) {
141  Log::info() << "ObserverIAU: apply extra inflation" << std::endl;
142  const eckit::LocalConfiguration inflation(config_, "inflation");
143  for (util::DateTime t(xx.validTime()); t <= lasttime; t += step) {
144  inflation_->multiply((*increments_[t]));
145  }
146  }
147  }
148 }

152 template <typename MODEL, typename STATE>
153 void ObserverIAU<MODEL, STATE>::doProcessing(STATE &xx) {
154  const util::DateTime now(xx.validTime());
155 
156  if (steps_.count(now)) {
157  Log::info() << "ObserverIAU: Process IAU at: " << now << std::endl;
158  std::shared_ptr<Increment_> dx(increments_[steps_[now]]);
159  util::Duration offset(xx.validTime() - dx->validTime());
160  dx->updateTime(offset);
161  xx += (*dx);
162  }
163 }
