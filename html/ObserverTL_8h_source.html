115 template <typename MODEL, typename INCR>
116 void ObserverTL<MODEL, INCR>::doInitializeTL(const INCR &dx,
117  const util::DateTime &end,
118  const util::Duration &tstep) {
120  tstep_ = tstep;
121  const util::DateTime bgn(dx.validTime());
122  if (hslot_ == util::Duration(0)) hslot_ = tstep / 2;
123  if (subwindows_) {
124  if (bgn == end) {
125  bgn_ = bgn - hslot_;
126  end_ = end + hslot_;
127  } else {
128  bgn_ = bgn;
129  end_ = end;
130  }
131  }
132  if (bgn_ < winbgn_) bgn_ = winbgn_;
133  if (end_ > winend_) end_ = winend_;
134 
137  gvalsIter_ it = gvals_.find(bgn_);
138  if (it == gvals_.end()) {
139  std::vector<std::unique_ptr<GeoVaLs_> > gvalsvec;
140  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
141  gvalsvec.emplace_back(
142  new GeoVaLs_(obspaces_[jj], hoptlad_.variables(jj), dx, bgn_, end_));
143  }
144  gvals_.emplace(bgn_, std::move(gvalsvec));
145  it = gvals_.find(bgn_);
146  ASSERT(it != gvals_.end());
147  }
148  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
149  it->second[jj]->zero();
150  }
151 }

153 template <typename MODEL, typename INCR>
154 void ObserverTL<MODEL, INCR>::doProcessingTL(const INCR & dx) {
155  util::DateTime xxtime(bgn_);
156  int ts = hslot_.toSeconds()*2;
157 
158  if (ts == tstep_.toSeconds()) {
160  xxtime = dx.validTime();
161  }
162  else {
164  util::Duration term(dx.validTime()-bgn_);
165  int vt = term.toSeconds();
166  if ( (vt % ts) > tstep_.toSeconds()/2 ) return;
167 
168  // xxtime : validTime of timeslot to use
169  util::Duration slot((int)std::round(((double)vt)/((double)ts))*ts);
170  xxtime += slot;
171  }
172 
174  util::DateTime t1(xxtime - hslot_);
175  util::DateTime t2(xxtime + hslot_);
176  if (t1 < bgn_) t1 = bgn_;
177  if (t2 > end_) t2 = end_;
178 
179  //Log::info() << "ObserverTL doProcessing for : t1=" << t1 << " t2=" << t2 << std::endl;
180 
181  // For each observation space:
182  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
184  Locations_ locs(obspaces_[jj], t1, t2);
185 
188  gvalsIter_ it = gvals_.find(bgn_);
189  ASSERT(it != gvals_.end());
190  dx.interpolateTL(locs, *it->second[jj]);
191  }
192 }

194 template <typename MODEL, typename INCR>
195 void ObserverTL<MODEL, INCR>::doFinalizeTL(const INCR &) {
196  // For each observation space:
197  gvalsIter_ it = gvals_.find(bgn_);
198  ASSERT(it != gvals_.end());
199  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
201  hoptlad_[jj].obsEquivTL(*it->second[jj], ydep_[jj].obsvector(),
202  (*dybias_)[jj]);
204  if (ydep_[jj].hasBias())
205  hoptlad_[jj].obsBiasEquivTL(*it->second[jj], ydep_[jj].biasvector(),
206  (*dybias_)[jj]);
207  }
208 }
