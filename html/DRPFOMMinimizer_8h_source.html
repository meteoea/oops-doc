109 template <typename MODEL>
110 double DRPFOMMinimizer<MODEL>::solve(CtrlInc_ &dx, CtrlInc_ &dxh, CtrlInc_ &rr,
111  const Bmat_ &B, const HtRinvH_ &HtRinvH,
112  const double costJ0Jb,
113  const double costJ0JoJc, const int maxiter,
114  const double tolerance) {
115  // dx increment
116  // dxh B^{-1} dx
117  // rr (sum B^{-1} dx_i^{b} +) G^T R^{-1} d
118 
119  CtrlInc_ zz(dxh);
120  CtrlInc_ pr(dxh);
121  CtrlInc_ vv(rr);
122 
123  std::vector<double> ss;
124  std::vector<double> dd;
125  std::vector<std::vector<double> > Hess;
126  std::vector<std::vector<double> > UpHess;
127 
128  // J0
129  const double costJ0 = costJ0Jb + costJ0JoJc;
130 
131  // lmp_.update(vvecs_, hvecs_, zvecs_, alphas_, betas_);
132  hvecs_.clear();
133  zvecs_.clear();
134  vvecs_.clear();
135  // !!!!! Needs to be generalized for Hessenberg Matrix.
136 
137  // z_{0} = B LMP r_{0}
138  // lmp_.multiply(vv, pr);
139  pr = vv;
140  B.multiply(pr, zz);
141 
142  // beta_{0} = sqrt( z_{0}^T r_{0} )
143  double beta = sqrt(dot_product(zz, vv));
144  const double beta0 = beta;
145 
146  // v_{1} = r_{0} / beta_{0}
147  vv *= 1 / beta;
148  // pr_{1} = LMP r_{0} / beta_{0}
149  pr *= 1 / beta;
150  // z_{1} = z_{0} / beta_{0}
151  zz *= 1 / beta;
152 
153  // hvecs[0] = pr_{1} --> required for solution
154  hvecs_.emplace_back(new CtrlInc_(pr));
155  // zvecs[0] = z_{1} ---> for re-orthogonalization
156  zvecs_.emplace_back(new CtrlInc_(zz));
157  // vvecs[0] = v_{1} ---> for re-orthogonalization
158  vvecs_.emplace_back(new CtrlInc_(vv));
159 
160  // Initialiaze (maxiter + 1) by maxiter matrix H
161  Hess.resize(maxiter);
162  for (int ii = 0; ii <= maxiter - 1; ii++) {
163  Hess[ii].resize(maxiter + 1);
164  for (int jj = 0; jj <= maxiter; jj++) {
165  Hess[ii][jj] = 0;
166  }
167  }
168 
169  double normReduction = 1.0;
170 
171  Log::info() << std::endl;
172  for (int jiter = 0; jiter < maxiter; ++jiter) {
173  Log::info() << "DRPFOM Starting Iteration " << jiter + 1 << std::endl;
174 
175  // v_{i+1} = ( pr_{i} + H^T R^{-1} H z_{i} )
176  HtRinvH.multiply(zz, vv);
177  vv += pr;
178  // Arnoldi Process
179  for (int jj = 0; jj <= jiter; ++jj) {
180  Hess[jiter][jj] = dot_product(*zvecs_[jj], vv);
181  vv.axpy(-Hess[jiter][jj], *vvecs_[jj]);
182  }
183 
184  // z_{i+1} = B LMP v_{i+1}
185  // lmp_.multiply(vv, pr);
186  pr = vv;
187  B.multiply(pr, zz);
188 
189  // beta_{i+1} = sqrt( zz_{i+1}^t, vv_{i+1} )
190  beta = sqrt(dot_product(zz, vv));
191 
192  Hess[jiter][jiter + 1] = beta;
193 
194  // v_{i+1} = v_{i+1} / beta_{i+1}
195  vv *= 1 / beta;
196  // pr_{i+1} = pr_{i+1} / beta_{i+1}
197  pr *= 1 / beta;
198  // z_{i+1} = z_{i+1} / beta_{i+1}
199  zz *= 1 / beta;
200 
201  // hvecs[i+1] =pr_{i+1}
202  hvecs_.emplace_back(new CtrlInc_(pr));
203  // zvecs[i+1] = z_{i+1}
204  zvecs_.emplace_back(new CtrlInc_(zz));
205  // vvecs[i+1] = v_{i+1}
206  vvecs_.emplace_back(new CtrlInc_(vv));
207 
208  if (jiter == 0) {
209  ss.push_back(beta0 / Hess[0][0]);
210  dd.push_back(beta0);
211  } else {
212  // Solve the upper Hessenberg system H_{i} s_{i} = beta0 * e_1
213  dd.push_back(beta0 * dot_product(*zvecs_[0], vv));
214  UpHess = Hess;
215  UpHess.resize(jiter + 1);
216  for (int ii = 0; ii <= jiter; ii++) {
217  UpHess[ii].resize(jiter + 1);
218  for (int jj = 0; jj <= jiter; jj++) {
219  UpHess[ii][jj] = Hess[ii][jj];
220  }
221  }
222  UpHessSolve(UpHess, dd, ss);
223  }
224 
225  betas_.push_back(beta);
226 
227  // Compute the quadratic cost function
228  // J[du_{i}] = J[0] - 0.5 s_{i}^T Z_{i}^T r_{0}
229  // Jb[du_{i}] = 0.5 s_{i}^T V_{i}^T Z_{i} s_{i}
230  double costJ = costJ0;
231  double costJb = costJ0Jb;
232  for (int jj = 0; jj < jiter + 1; ++jj) {
233  costJ -= 0.5 * ss[jj] * dot_product(*zvecs_[jj], rr);
234  costJb += 0.5 * ss[jj] * dot_product(*vvecs_[jj], *zvecs_[jj]) * ss[jj];
235  }
236  double costJoJc = costJ - costJb;
237 
238  // Gradient norm in precond metric --> sqrt(r'z) --> beta * s_{i}
239  double rznorm = beta * std::abs(ss[jiter]);
240  normReduction = rznorm / beta0;
241 
242  Log::info() << "DRPFOM end of iteration " << jiter + 1 << std::endl
243  << " Norm reduction (" << std::setw(2) << jiter + 1
244  << ") = " << util::full_precision(normReduction) << std::endl
245  << " Quadratic cost function: J (" << std::setw(2)
246  << jiter + 1 << ") = " << util::full_precision(costJ)
247  << std::endl
248  << " Quadratic cost function: Jb (" << std::setw(2)
249  << jiter + 1 << ") = " << util::full_precision(costJb)
250  << std::endl
251  << " Quadratic cost function: JoJc(" << std::setw(2)
252  << jiter + 1 << ") = " << util::full_precision(costJoJc)
253  << std::endl
254  << std::endl;
255 
256  if (normReduction < tolerance) {
257  Log::info() << "DRPFOM: Achieved required reduction in residual norm."
258  << std::endl;
259  break;
260  }
261  }
262 
263  // Calculate the solution (dxh = Binv dx)
264  for (unsigned int jj = 0; jj < ss.size(); ++jj) {
265  dx.axpy(ss[jj], *zvecs_[jj]);
266  dxh.axpy(ss[jj], *hvecs_[jj]);
267  }
268 
269  return normReduction;
270 }
