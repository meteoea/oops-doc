110 template <typename MODEL>
111 void EnsembleCovariance4DDistributed<MODEL>::advectedLinearize(
112  const State4D_ &xb, const Geometry_ &resol) {
113  /*
114  * Build localization
115  */
116  geom_.reset(new Geometry_(resol));
117  const eckit::LocalConfiguration confLoc(config_, "localization");
118  loc_.reset(new Localization_(*geom_, confLoc));
119 
120  /*
121  * Build ensemble
122  */
123  std::vector<eckit::LocalConfiguration> confEnsemble;
124  config_.get("ensemble", confEnsemble);
125 
126  time_.clear();
127  nsubwin_ = confEnsemble.size();
128  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
129  EnsemblePtr_ ens_k(
130  new Ensemble_(xb[jsub].validTime(), confEnsemble[jsub], true));
131  ens_k->build(xb[jsub], resol);
132  EnsemblesCollection_::getInstance().put(xb[jsub].validTime(), ens_k);
133  time_.push_back(xb[jsub].validTime());
134  }
135 
136  /*
137  * Calculate perturbations
138  */
139  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
140  EnsemblePtr_ ens =
141  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
142  ens->to_perturbations((bool)true);
143  }
144 
145  Log::trace() << "EnsembleCovariance4DDistributed linearized.";
146 }

157 template <typename MODEL>
158 void EnsembleCovariance4DDistributed<MODEL>::advectedMultiply(
159  const Increment4D_ &dxi, Increment4D_ &dxo) const {
160  Increment_ dx_loc(dxo[0]);
161  Increment_ tmp(dxo[0]);
162  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxi[0].validTime()];
163 
164  dxo.zero();
165  for (unsigned l = 0; l < e->size(); ++l) {
166  unsigned ix = get_xb_prim(l);
167  dx_loc.zero();
168  for (unsigned k2 = 0; k2 < nsubwin_; ++k2) {
169  EnsemblePtr_ e_k2 =
170  EnsemblesCollection_::getInstance()[dxi[k2].validTime()];
171  schur_add((*e_k2)[ix], dxi[k2], dx_loc, tmp);
172  }
173 
174  loc_->multiply(dx_loc);
175 
176  for (unsigned k1 = 0; k1 < nsubwin_; ++k1) {
177  EnsemblePtr_ e_k1 =
178  EnsemblesCollection_::getInstance()[dxi[k1].validTime()];
179  schur_add((*e_k1)[ix], dx_loc, dxo[k1], tmp);
180  }
181  }
182 }

233 template <typename MODEL>
234 unsigned EnsembleCovariance4DDistributed<MODEL>::get_xb_prim(unsigned l) const {
235  util::MPIWrapper &mpiw = util::MPIWrapper::Instance();
236 
237  // Circular permutation that depends on my rank
238  unsigned i_xb_prim =
239  (mpiw.nb_worlds() - mpiw.my_rank() + l) % mpiw.nb_worlds();
240 
241  if (mpiw.my_rank() == i_xb_prim)
242  // Nothing to do, since xb_prim is my perturbation
243  return 0;
244 
245  Log::info() << "EnsembleCovariance4DDistributed.get_xb_prim(): my_rank="
246  << mpiw.my_rank() << " get xb_prim=" << i_xb_prim;
247 
248  util::MPIThreadPool<Increment_> pool;
249  for (unsigned k = 0; k < nsubwin_; k++) {
250  EnsemblePtr_ e_k = EnsemblesCollection_::getInstance()[time_[k]];
251  pool.add(&(e_k->pack()), &(*e_k)[1]);
252  }
253  int tag = l;
254  pool.send_recv_nb(i_xb_prim, tag);
255 
256  // xb_prim is the perturbation I have received and stored in (*e_k)[1])
257  return 1;
258 }
