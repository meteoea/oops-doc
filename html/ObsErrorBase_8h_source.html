128 template <typename MODEL>
129 ObsErrorBase<MODEL> *ObsErrorFactory<MODEL>::create(
130  const ObsSpace_ &obs, const eckit::Configuration &conf) {
131  Log::trace() << "ObsErrorBase<MODEL>::create starting" << std::endl;
132  const std::string id = conf.getString("covariance");
133  typename std::map<std::string, ObsErrorFactory<MODEL> *>::iterator jerr =
134  getMakers().find(id);
135  if (jerr == getMakers().end()) {
136  Log::error() << id << " does not exist in observation error factory."
137  << std::endl;
138  ABORT("Element does not exist in ObsErrorFactory.");
139  }
140  ObsErrorBase<MODEL> *ptr = jerr->second->make(obs, conf);
141  Log::trace() << "ObsErrorBase<MODEL>::create done" << std::endl;
142  return ptr;
143 }
