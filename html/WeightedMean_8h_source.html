103 template <typename MODEL, typename FLDS>
104 void WeightedMean<MODEL, FLDS>::doInitialize(const FLDS &xx,
105  const util::DateTime &end,
106  const util::Duration &tstep) {
107  const util::DateTime bgn(xx.validTime());
108  if (!linit_ && bgn <= end_ && end >= bgn_) {
109  weights_ = wfct_->setWeights(bgn_, end_, tstep);
110  linit_ = true;
111  }
112  endleg_ = end;
113  // moved the accumulator instantiation from constructor to here, to allow use
114  // of copy constructor
115  avg_ = new Accumulator<MODEL, FLDS, FLDS>(xx);
116  avg_->validTime() = time_;
117 }

121 template <typename MODEL, typename FLDS>
122 void WeightedMean<MODEL, FLDS>::doProcessing(const FLDS &xx) {
123  const util::DateTime now(xx.validTime());
124  if (now != endleg_ || now == end_) {
125  ASSERT(weights_.find(now) != weights_.end());
126  const double zz = weights_[now];
127  avg_->accumul(zz, xx);
128  sum_ += zz;
129  Log::debug() << "WeightedMean: time = " << now << ", weight = " << zz
130  << ", sum = " << sum_ << std::endl;
131  }
132 }
