111 template <typename MODEL, typename INCR>
112 void WeightedDiffTL<MODEL, INCR>::doInitializeTL(const INCR &dx,
113  const util::DateTime &end,
114  const util::Duration &tstep) {
115  const util::DateTime bgn(dx.validTime());
116  ASSERT(bgn <= end);
117  if (!linit_ && bgn <= end_ && end >= bgn_) {
118  if (tstep_ == util::Duration(0)) tstep_ = tstep;
119  ASSERT(tstep_ > util::Duration(0));
120  weights_ = wfct_.setWeights(bgn_, end_, tstep_);
121  linit_ = true;
122  ASSERT(weights_.find(vtime_) != weights_.end());
123  weights_[vtime_] -= 1.0;
124  }
125  bgnleg_ = bgn;
126  endleg_ = end;
127  Log::info() << "WeightedDiffTL: initialized"
128  << " bgnleg_ = " << bgnleg_ << ", endleg_ = " << endleg_
129  << ", bgn_ = " << bgn_ << ", end_ = " << end_ << std::endl;
130  Log::debug() << " Weights:" << std::endl;
131  for (auto const &weight : weights_) {
132  Log::debug() << " " << weight.first << " : " << weight.second
133  << std::endl;
134  }
135 }

139 template <typename MODEL, typename INCR>
140 void WeightedDiffTL<MODEL, INCR>::doProcessingTL(const INCR &xx) {
141  const util::DateTime now(xx.validTime());
142  if (((bgnleg_ < end_ && endleg_ > bgn_) || bgnleg_ == endleg_) &&
143  (now != endleg_ || now == end_ || now == bgnleg_)) {
144  ASSERT(weights_.find(now) != weights_.end());
145  const double zz = weights_[now];
146  avg_->axpy(zz, xx, false);
147  sum_ += zz;
148  }
149 }
