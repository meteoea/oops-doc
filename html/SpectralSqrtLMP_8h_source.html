145 template <typename MODEL>
146 void SpectralSqrtLMP<MODEL>::update(
147  const std::vector<std::shared_ptr<CtrlVec_> > &Zhl,
148  const std::vector<double> &alphas, const std::vector<double> &betas) {
149  if (active_) {
150  Log::info() << "SpectralSqrtLMP::update starting" << std::endl;
151  // If useoldpairs = false, use only current information
152  if (!useoldpairs_) {
153  eigvals_.clear();
154  usedpairIndx_.clear();
155  U_.clear();
156  }
157 
158  const size_t nvec = Zhl.size();
159  if (nvec > 0) {
160  ASSERT(alphas.size() == nvec - 1);
161  ASSERT(betas.size() == nvec - 1);
162 
163  // Compute spectrum of tri-diagonal matrix
164  std::vector<double> evals;
165  std::vector<std::vector<double> > evecs;
166  TriDiagSpectrum(alphas, betas, evals, evecs);
167 
168  // Determine the converged eigenvalues
169  std::vector<size_t> convIndx;
170  std::vector<double> neweigvals;
171  for (int ji = nvec - 2; ji >= 0; ji--) {
172  double erritz = evecs[ji][nvec - 2] * betas[nvec - 2];
173  double lambda = evals[ji];
174  // Add new converged eigenvalues
175  if (std::abs(erritz) <= errthreshold_ * lambda && lambda > mineigval_ &&
176  eigvals_.size() < maxpairs_) {
177  Log::info() << "Eigenvalue accepted: " << lambda
178  << ", error= " << std::abs(erritz)
179  << ", threshold= " << errthreshold_ * lambda << std::endl;
180  convIndx.push_back(ji);
181  eigvals_.push_back(lambda);
182  neweigvals.push_back(lambda);
183  } else {
184  if (eigvals_.size() >= maxpairs_)
185  Log::info() << "Warning: Maximum allowed number of pairs exceeded"
186  << std::endl;
187 
188  Log::info() << "Eigenvalue not accepted: " << lambda
189  << ", error= " << std::abs(erritz)
190  << ", threshold= " << errthreshold_ * lambda << std::endl;
191  }
192  }
193  // Keep information on how many pairs are used at each minimization
194  // loop
195  usedpairIndx_.push_back(convIndx.size());
196  Log::info() << " Outer loop : " << update_ << std::endl
197  << " Number of inner iterations : " << Zhl.size() - 1
198  << std::endl
199  << " Maximum number of pairs : " << maxpairs_
200  << std::endl
201  << " Number of new converged pairs : " << convIndx.size()
202  << std::endl
203  << " New converged eigenvalues : "
204  << util::full_precision(neweigvals) << std::endl;
205  size_t count = 0;
206  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
207  count += usedpairIndx_[kiter];
208  Log::info() << " Number of used pairs from outer loop " << kiter
209  << " : " << usedpairIndx_[kiter] << std::endl;
210  }
211  Log::info() << " Total number of used pairs : " << count << std::endl;
212  // Calculate/Update the matrix U
213  for (size_t ji = 0; ji < convIndx.size(); ++ji) {
214  std::shared_ptr<CtrlVec_> uu(new CtrlVec_(J_.jb()));
215  uu->zero();
216  for (size_t ii = 0; ii < nvec - 1; ++ii) {
217  uu->axpy(evecs[convIndx[ji]][ii], *Zhl[ii]);
218  }
219  // Insert new eigenvectors
220  U_.push_back(uu);
221  }
222  }
223  ++update_;
224  Log::info() << "SpectralSqrtLMP::update done" << std::endl;
225  }
226 }

230 template <typename MODEL>
231 void SpectralSqrtLMP<MODEL>::postprocess() {
232  if (active_) {
233  /* The Ritz vectors combined from different minimizations are not
234  * necessarily orthogonal, while the preconditioner requires a set of
235  * orthogonal vectors. We first compute eigenvectors v_i of (R) (R)^T,
236  * where R=Q U and U is a rectangular matrix with columns being Ritz
237  * vectors and Q is a Hausholder matrix - R is an upper triangular matrix
238  * and (R)(R)^T has only a non-zero block in the top left corner of size
239  * LxL, where L is the number of Ritz vectors forming U. The required
240  * preconditioning vectors are given by w_i = Q^T v_i and the required
241  * eigenvalues are given by eigenvalues of the LxL matrix.
242  */
243  Log::info() << "SpectralSqrtLMP::postprocess starting" << std::endl;
244 
245  if (orthogonalize_) {
246  Log::info() << "SpectralSqrtLMP::postprocess orthogonalization starting"
247  << std::endl;
248  size_t nvecs = U_.size();
249  ASSERT(nvecs <= maxpairs_);
250 
251  // Copy and transform preconditioning vectors/eigenvectors
252  std::vector<std::shared_ptr<CtrlVec_> > Q;
253  Q.clear();
254  for (size_t jj = 0; jj < nvecs; ++jj) {
255  std::shared_ptr<CtrlVec_> qqvec(new CtrlVec_(*U_[jj]));
256  U_[jj].reset();
257  double scale = std::sqrt(eigvals_[jj] - 1);
258  (*qqvec) *= scale;
259  Q.push_back(qqvec);
260  }
261 
262  // mp_R matrix - stored in a column major order
263  Matrix<double> mp_R(nvecs, nvecs);
264  for (size_t jj = 0; jj < nvecs; ++jj) {
265  std::vector<double> qqvec(jj + 1);
266  Q[jj]->state()[0].modCtlVec().getSection(qqvec, 1, jj + 1);
267  double zz =
268  dot_product(*Q[jj], *Q[jj]) -
269  std::inner_product(qqvec.begin(), qqvec.end(), qqvec.begin(), 0.0);
270  if (qqvec[jj] < 0) {
271  mp_R(jj, jj) = -std::sqrt(zz + qqvec[jj] * qqvec[jj]);
272  } else {
273  mp_R(jj, jj) = std::sqrt(zz + qqvec[jj] * qqvec[jj]);
274  }
275  qqvec[jj] = qqvec[jj] - mp_R(jj, jj);
276  for (size_t jk = 0; jk < jj; ++jk) {
277  mp_R(jk, jj) = qqvec[jk];
278  }
279  zz = zz + qqvec[jj] * qqvec[jj];
280  std::vector<double> zzvec(jj + 1);
281  zzvec[jj] = qqvec[jj];
282  Q[jj]->state()[0].modCtlVec().setSection(zzvec, 1, jj + 1);
283  *Q[jj] *= std::sqrt(2.0 / zz);
284 
285  // We now have Householder vectors in Q[jj-1] and the non-zero elements
286  // of the transformed vector in mp_R. Now apply the Householder
287  // transfromations to the remaining vectors.
288 
289  for (size_t ji = jj + 1; ji < nvecs; ++ji) {
290  double zz = dot_product(*Q[jj], *Q[ji]);
291  Q[ji]->axpy(-zz, *Q[jj]);
292  }
293  }
294 
295  if (debug_) {
296  Log::info() << "SpectralSqrtLMP::QR check orthonormality of Q:"
297  << std::endl;
298  for (size_t jj = 0; jj < nvecs; ++jj) {
299  for (size_t jk = 0; jk < nvecs; ++jk) {
300  double zz = dot_product(*Q[jj], *Q[jk]);
301  Log::info() << " <Q[" << jj << "," << jk
302  << "]> = " << util::full_precision(zz) << std::endl;
303  }
304  Log::info() << std::endl;
305  }
306  }
307 
308  // Multiply the upper triangle by its transpose and find eigenvectors and
309  // eigenvalues
310  for (size_t jj = 0; jj < nvecs; ++jj) {
311  for (size_t ji = jj + 1; ji < nvecs; ++ji) {
312  mp_R(ji, jj) = 0.0;
313  }
314  }
315 
316  // RR^T matrix - stored in a column major order
317  Matrix<double> mp_RRT = mp_R * mp_R.transpose();
318 
319  // Find eigenpairs of RR^T = L U L^T
320  ColVector<double> cv_U;
321  Matrix<double> mp_L;
322  SymMatSpectrum(mp_RRT, cv_U, mp_L);
323 
324  // Put eigenvalues in a standard vector
325  std::vector<double> evals(cv_U.size());
326  for (size_t jj = 0; jj < evals.size(); ++jj) evals[jj] = cv_U(jj);
327 
328  // Transform eigenvalues
329  for (auto &val : evals) {
330  val = val + 1.0;
331  }
332 
333  Log::info() << "Eigenvalues: " << util::full_precision(evals)
334  << std::endl;
335 
336  // Order eigenvalues
337  std::vector<int> indices(evals.size());
338  std::size_t n(0);
339  std::generate(std::begin(indices), std::end(indices),
340  [&] { return n++; });
341 
342  std::sort(std::begin(indices), std::end(indices),
343  [&](int ii1, int ii2) { return evals[ii1] > evals[ii2]; });
344 
345  Log::info() << "Sorted eigenvalues (" << indices.size()
346  << "):" << std::endl;
347  for (const auto &ind : indices)
348  Log::info() << "Index: " << ind << ", value: " << evals[ind]
349  << std::endl;
350 
351  // Accept or reject eigenvectors
352  indices.erase(std::remove_if(indices.begin(), indices.end(),
353  [&](int n) { return evals[n] <= 1.0; }),
354  indices.end());
355 
356  Log::info() << "Sorted accepted eigenvalues (" << indices.size()
357  << "):" << std::endl;
358  for (const auto &ind : indices)
359  Log::info() << "Index: " << ind << ", value: " << evals[ind]
360  << std::endl;
361 
362  if (indices.size() == 0)
363  Log::info()
364  << "Warning: cannot form preconditioner - no positive eigenvalues; "
365  << "minimisation will not be preconditioned" << std::endl;
366 
367  // Transform back u_i = Q^T v_i
368  U_.clear();
369  eigvals_.clear();
370  for (auto const &ind : indices) {
371  // Limit eigenvalues
372  evals[ind] = std::min(10.0, evals[ind]);
373  eigvals_.push_back(evals[ind]);
374  std::shared_ptr<CtrlVec_> uu(new CtrlVec_(J_.jb()));
375  std::vector<double> evec(nvecs);
376  for (size_t ji = 0; ji < nvecs; ++ji) evec[ji] = mp_L(ji, ind);
377  uu->zero();
378  uu->state()[0].modCtlVec().setSection(evec, 1, nvecs);
379  for (int ji = nvecs - 1; ji >= 0; --ji) {
380  double zz = dot_product(*Q[ji], *uu);
381  uu->axpy(-zz, *Q[ji]);
382  }
383  U_.push_back(uu);
384  }
385  // Release Q
386  Q.clear();
387 
388  if (debug_) {
389  Log::info() << "SpectralSqrtLMP::QR check orthonormality of U:"
390  << std::endl;
391  for (size_t jj = 0; jj < nvecs; ++jj) {
392  for (size_t jk = 0; jk < nvecs; ++jk) {
393  double zz = dot_product(*U_[jj], *U_[jk]);
394  Log::info() << " <U[" << jj << "," << jk
395  << "]> = " << util::full_precision(zz) << std::endl;
396  }
397  Log::info() << std::endl;
398  }
399  }
400 
401  Log::info() << "SpectralSqrtLMP::postprocess orthogonalization done"
402  << std::endl;
403  } else {
404  // Limit eigenvalues
405  for (auto &val : eigvals_) val = std::min(10.0, val);
406  }
407  Log::info() << "SpectralSqrtLMP::postprocess done" << std::endl;
408  }
409 }

427 template <typename MODEL>
428 void SpectralSqrtLMP<MODEL>::inverseMultiply(const CtrlVec_ &a,
429  CtrlVec_ &b) const {
430  b = a; // P_0 = I
431 
432  for (size_t ii = 0; ii < U_.size(); ++ii) {
433  double zeval = eigvals_[ii];
434  ASSERT(zeval);
435  double zz = 1.0 / zeval - 1.0;
436  b.axpy(zz * dot_product(a, *U_[ii]), *U_[ii]);
437  }
438 }

546  auxconf.set("update", update_);
547  Log::info() << " " << classname() << "::checkpoint update "

552  }
