167 template <typename MODEL>
168 double CostFct4DVarMFTemplate<MODEL>::evaluate(const CtrlVar_ &fguess,
169  const eckit::Configuration &config,
170  PostProcessor<State_> post) const {
171  Log::trace() << "CostFct4DVarMFTemplate::evaluate start" << std::endl;
172 
174  PostProcessor<State_> pp(post);
175  std::shared_ptr<WeightedMean_> pdfi;
176  if (config.has("dfi")) {
177  const eckit::LocalConfiguration dfiConf(config, "dfi");
178  const util::Duration dfispan(dfiConf.getString("filter_span"));
179  const util::DateTime dfitime(fguess.state()[0].validTime() + dfispan / 2);
180  pdfi.reset(new WeightedMean_(dfitime, dfispan, geom_, dfiConf));
181  pp.enrollProcessor(pdfi);
182  }
183 
185  double zzz = this->CostFunction<MODEL>::evaluate(fguess, config, pp);
186 
197  if (config.has("dfi")) {
198  const eckit::LocalConfiguration dfiConf(config, "dfi");
199  std::unique_ptr<State_> xdfi(pdfi->releaseMean());
200  (*xdfi) += (*biasDFI_);
201  Log::info() << "CostFunction: Write filtered analysis in the middle of the window.";
202  xdfi->write(eckit::LocalConfiguration(dfiConf, "output"));
203  }
204 
205  Log::trace() << "CostFct4DVarMFTemplate::evaluate done" << std::endl;
206  return zzz;
207 }

211 template<typename MODEL>
212 double CostFct4DVarMFTemplate<MODEL>::linearize(const CtrlVar_ & fguess,
213  const eckit::Configuration & innerConf,
214  PostProcessor<State_> post) {
253  const eckit::LocalConfiguration resConf(innerConf, "resolution");
255  const eckit::LocalConfiguration tlmConf(innerConf, "linearmodel");
256  std::unique_ptr<eckit::LocalConfiguration> dfiConf;
257  const Geometry_ lowres(resConf);
258  std::shared_ptr<WeightedDiff_> pdfi;
259  std::unique_ptr<WeightingFct> wfct;
260 
261  PostProcessor<State_> pp(post);
262  pp.enrollProcessor(this->getJterm(0).initialize(fguess, innerConf));
263 
266  if (innerConf.has("screening.dfi")) {
267  dfiConf.reset(new eckit::LocalConfiguration(innerConf, "screening.dfi"));
268  const util::Duration dfispan(dfiConf->getString("filter_span"));
269  const util::DateTime dfitime(fguess.state()[0].validTime() + dfispan / 2);
270  wfct.reset(new DolphChebyshev(*dfiConf));
271  pdfi.reset(new WeightedDiff_(dfitime, dfispan, geom_, *dfiConf, util::Duration(0), *wfct));
272  pp.enrollProcessor(pdfi);
273  if (dfiConf->has("output")) {
274  const eckit::LocalConfiguration outConfig(*dfiConf, "output");
275  pp.enrollProcessor(new StateWriter<State_>(fguess.state()[0].validTime(), outConfig));
276  }
277  }
278 
281  CtrlVar_ fghr(fguess);
282  this->runNL(fghr, pp);
283  costJoJc_ = this->getJterm(0).finalize(innerConf);
284 
291  if (innerConf.has("screening.dfi")) {
292  biasDFI_.reset(pdfi->releaseDiff());
293  (*biasDFI_) *= -1.0;
294  }
295 
297  std::unique_ptr<CtrlVar_> fglr;
298  PostProcessor<State_> no_post;
299  const eckit::LocalConfiguration modelConfig(innerConf, "low_resolution_model");
300  trajectoryModel_.reset(new Model_(lowres, modelConfig));
301  fglr.reset(new CtrlVar_(lowres, *trajectoryModel_, fguess));
302  malloc_trim(0);
303 
304  // 6. Setup trajectory calculation at low resolution
305  PostProcessor<State_> pplr(post);
306  std::shared_ptr<JqTerm_> jq(this->getJb().initializeTraj(*fglr, lowres, innerConf));
307  pplr.enrollProcessor(jq);
308 
309  for (unsigned iterm=0; iterm < this->nterms(); ++iterm) {
310  pplr.enrollProcessor(this->getJterm(iterm).initializeTraj(*fglr, lowres, innerConf));
311  }
312  std::shared_ptr<TrajectorySaver_ > trajectorySaver(
313  new TrajectorySaver_(fglr->state()[0], fglr->modVar(), lowres,
314  this->getTLMs(), tlmConf));
315  pplr.enrollProcessor(trajectorySaver);
316 
317  std::shared_ptr<Advector_ > advector = Advector_::get();
318  if (advector) pplr.enrollProcessor(advector);
319 
327  if (innerConf.has("low_resolution_model")) {
328  util::LogbookWriter<std::string> log("MF4DVarTrajectory",
329  "onlyHorizInterpol");
330  this->runNL(*fglr, *trajectoryModel_, pplr);
331  }
332 
334  costJb_ = this->getJb().finalizeTraj(jq.get(), innerConf);
335  /* Do not change Jo.gradFG */
336  double costJoJc = 0.0;
337  for (unsigned iterm=1; iterm < this->nterms(); ++iterm) {
338  costJoJc += this->getJterm(iterm).finalizeTraj(innerConf);
339  }
340 
342  double costJ = costJb_ + costJoJc_;
343  Log::test() << "CostFunction: Nonlinear J = " << costJ << std::endl;
344  return costJ;
345 }

439 template <typename MODEL>
440 void CostFct4DVarMFTemplate<MODEL>::addIncr(CtrlVar_ &xfghr, const CtrlInc_ &dxlr,
441  PostProcessor<Increment_> &) const {
442  ASSERT(xfghr.state().checkStatesNumber(1));
443 
445  State_ fghr(geom_, *trajectoryModel_, *fglr_);
446 
448  (*fglr_) += dxlr.state()[0];
449  State_ xahr(geom_, *trajectoryModel_, *fglr_);
450 
452  Increment_ dxhr(geom_, dxlr.state()[0]);
453  dxhr.diff(xahr, fghr);
454 
456  xfghr.state()[0] += dxhr;
457 
458  // Quick fix: write low_resolution analysis for FSOi
459  if (write_analyse_) {
460  eckit::LocalConfiguration outputConfig;
461  std::ostringstream naml;
462  naml << "naml_write_anaminbr" << (++fgnum_);
463  outputConfig.set("namelist", naml.str());
464  fglr_->write(outputConfig);
465  }
466 }
