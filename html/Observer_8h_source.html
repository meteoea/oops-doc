120 template <typename MODEL, typename STATE>
121 void Observer<MODEL, STATE>::doInitialize(const STATE &xx,
122  const util::DateTime &end,
123  const util::Duration &tstep) {
125  tstep_ = tstep;
126  const util::DateTime bgn(xx.validTime());
127  if (hslot_ == util::Duration(0)) hslot_ = tstep / 2;
128  if (subwindows_) {
129  if (bgn == end) {
130  bgn_ = bgn - hslot_;
131  end_ = end + hslot_;
132  } else {
133  bgn_ = bgn;
134  end_ = end;
135  }
136  }
137  if (bgn_ < winbgn_) bgn_ = winbgn_;
138  if (end_ > winend_) end_ = winend_;
139 
141  gvals_.clear();
142  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
143  std::shared_ptr<GeoVaLs_> tmp(
144  new GeoVaLs_(obspaces_[jj], hop_.variables(jj), xx, bgn_, end_));
145  gvals_.push_back(tmp);
146  }
147 }

149 template <typename MODEL, typename STATE>
150 void Observer<MODEL, STATE>::doProcessing(const STATE & xx) {
151  util::DateTime xxtime(bgn_);
152  int ts = hslot_.toSeconds()*2;
153 
154  if (ts == tstep_.toSeconds()) {
156  xxtime = xx.validTime();
157  }
158  else {
160  util::Duration term(xx.validTime()-bgn_);
161  int vt = term.toSeconds();
162 
163  Log::info() << "Observer doProcessing check at : " << xx.validTime() << " : " << vt % ts << " ?>? " << tstep_.toSeconds()/2 << std::endl;
164  if ( (vt % ts) > tstep_.toSeconds()/2 ) return;
165 
167  util::Duration slot((int)std::round(((double)vt)/((double)ts))*ts);
168  xxtime += slot;
169  }
170 
172  util::DateTime t1(xxtime-hslot_);
173  util::DateTime t2(xxtime+hslot_);
174 
175  if (t1 < bgn_) t1 = bgn_;
176  if (t2 > end_) t2 = end_;
177 
178  Log::info() << "Observer doProcessing at: " << xx.validTime() << " for : t1=" << t1 << " t2=" << t2 << std::endl;
179 
180  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
182  Locations_ locs(obspaces_[jj], t1, t2);
183 
186  xx.interpolate(locs, *gvals_.at(jj));
187  }
188 }

190 template <typename MODEL, typename STATE>
191 void Observer<MODEL, STATE>::doFinalize(const STATE &) {
192  // For each observation space:
193  for (std::size_t jj = 0; jj < obspaces_.size(); ++jj) {
195  if (htlad_) (*htlad_)[jj].setTrajectory(*gvals_.at(jj), ybias_[jj]);
196 
198  hop_[jj].obsEquiv(*gvals_.at(jj), (*yobs_)[jj].obsvector(), ybias_[jj]);
199 
201  if ((*yobs_)[jj].hasBias())
202  hop_[jj].obsBiasEquiv(*gvals_.at(jj), (*yobs_)[jj].biasvector(),
203  ybias_[jj]);
204  }
205 }
