73 template <typename MODEL, typename INCR>
74 void ObserverEnVarTL<MODEL, INCR>::doProcessingTL(const INCR &dx) {
75  INCR *dxo = dx_[dx.validTime()];
76  // Use axpy instead of copy-cstor, since it is able to interoperate
77  // fields and fields_base
78  if (to_low_resol_.get()) {
79  INCR dxout(dx,false);
80  dxout.zero();
81  dxout.axpy(1.0,dx);
82  to_low_resol_->inverseMultiply(dxout);
83  dxo->zero();
84  dxo->axpy(1.0,dxout);
85  }
86  else {
87  dxo->zero();
88  dxo->axpy(1.0,dx);
89  }
90 }
