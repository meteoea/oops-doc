69  int execute(const eckit::Configuration& fullConfig) const {
71  const eckit::LocalConfiguration resolConfig(fullConfig, "resolution");
72  const Geometry_ resol(resolConfig);
73 
75  const eckit::LocalConfiguration modelConfig(fullConfig, "model");
76  const Model_ model(resol, modelConfig);
77  Log::trace() << "Variational: model has been set up" << std::endl;
78 
90  const eckit::LocalConfiguration cfConf(fullConfig, "cost_function");
91  std::unique_ptr<CostFunction<MODEL> > J(
92  CostFactory<MODEL>::create(cfConf, resol, model));
93  Log::trace() << "Variational: cost function has been set up" << std::endl;
94 
96  IncrementalAssimilation<MODEL> incrementalAssimilation(*J);
97  incrementalAssimilation.assimilate(fullConfig);
98  Log::trace() << "Variational: incremantal assimilation done" << std::endl;
99 
101  if (incrementalAssimilation.skipFinalNonlinearTrajectory()) return 0;
102 
104  std::shared_ptr<ControlVariable<MODEL> > xx(
105  incrementalAssimilation.getAnalysis());
106 
109  PostProcessor<State_> post;
110  const util::DateTime winbgn(cfConf.getString("window_begin"));
111  if (fullConfig.has("output")) {
112  const eckit::LocalConfiguration outConfig(fullConfig, "output");
113  post.enrollProcessor(new StateWriter<State_>(winbgn, outConfig));
114  }
115 
116  const eckit::LocalConfiguration finalConfig(fullConfig, "final");
117  if (finalConfig.has("prints")) {
118  const eckit::LocalConfiguration prtConfig(finalConfig, "prints");
119  post.enrollProcessor(new StateInfo<State_>("final", prtConfig));
120  }
121 
122  J->evaluate(*xx, finalConfig, post);
123 
124  return 0;
125  }
126  // -----------------------------------------------------------------------------
