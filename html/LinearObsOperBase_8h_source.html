139 template <typename MODEL>
140 LinearObsOperBase<MODEL> *LinearObsOperFactory<MODEL>::create(
141  const ObsSpace_ &odb, const eckit::Configuration &conf) {
142  const std::string id = conf.getString("ObsType");
143  typename std::map<std::string, LinearObsOperFactory<MODEL> *>::iterator jloc =
144  getMakers().find(id);
145  if (jloc == getMakers().end()) {
146  Log::error() << id << " does not exist in LinearObsOperFactory."
147  << std::endl;
148  ABORT("Element does not exist in LinearObsOperFactory.");
149  }
150  LinearObsOperBase<MODEL> *ptr = jloc->second->make(odb, conf);
151  return ptr;
152 }
