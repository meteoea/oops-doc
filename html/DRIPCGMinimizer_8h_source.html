103 template <typename MODEL>
104 double DRIPCGMinimizer<MODEL>::solve(CtrlInc_ &xx, CtrlInc_ &xh, CtrlInc_ &rr,
105  const Bmat_ &B, const HtRinvH_ &HtRinvH,
106  const double costJ0Jb,
107  const double costJ0JoJc, const int maxiter,
108  const double tolerance) {
109  CtrlInc_ ap(xh);
110  CtrlInc_ pp(xh);
111  CtrlInc_ ph(xh);
112  CtrlInc_ ss(xh);
113  CtrlInc_ sh(xh);
114  CtrlInc_ dr(xh);
115  CtrlInc_ ww(xh);
116 
117  std::vector<CtrlInc_> vvecs; // for re-orthogonalization
118  std::vector<CtrlInc_> zvecs; // for re-orthogonalization
119  std::vector<double> scals; // for re-orthogonalization
120 
121  lmp_.multiply(rr, sh);
122  B.multiply(sh, ss);
123 
124  double dotRr0 = dot_product(rr, rr);
125  double dotSr0 = dot_product(rr, ss);
126  double normReduction = 1.0;
127  double rdots = dotSr0;
128  double rdots_old = dotSr0;
129 
130  vvecs.push_back(rr);
131  zvecs.push_back(ss);
132  scals.push_back(1.0 / dotSr0);
133 
134  Log::info() << std::endl;
135  for (int jiter = 0; jiter < maxiter; ++jiter) {
136  Log::info() << " DRIPCG Starting Iteration " << jiter + 1 << std::endl;
137 
138  if (jiter == 0) {
139  pp = ss;
140  ph = sh;
141  } else {
142  dr -= rr; // dr=oldr-r
143  double beta = -dot_product(ss, dr) / rdots_old;
144 
145  pp *= beta;
146  pp += ss; // p = s + beta*p
147 
148  ph *= beta;
149  ph += sh; // ph = sh + beta*ph
150  }
151 
152  HtRinvH.multiply(pp, ap);
153  ap += ph;
154 
155  dr = rr;
156 
157  double rho = dot_product(pp, ap);
158  double alpha = rdots / rho;
159 
160  xx.axpy(alpha, pp); // xx = xx + alpha*pp
161  xh.axpy(alpha, ph); // xh = xh + alpha*ph
162  rr.axpy(-alpha, ap); // rr = rr - alpha*ap
163 
164  // Re-orthogonalization
165  for (int jj = 0; jj < jiter; ++jj) {
166  double proj = scals[jj] * dot_product(rr, zvecs[jj]);
167  rr.axpy(-proj, vvecs[jj]);
168  }
169 
170  lmp_.multiply(rr, sh);
171  B.multiply(sh, ss);
172 
173  rdots_old = rdots;
174  rdots = dot_product(rr, ss);
175  normReduction = sqrt(dot_product(rr, rr) / dotRr0);
176 
177  Log::info() << "DRIPCG end of iteration " << jiter + 1
178  << ". Norm reduction= " << util::full_precision(normReduction)
179  << std::endl
180  << std::endl;
181 
182  // Save the pairs for preconditioning
183  lmp_.push(pp, ph, ap, rho);
184 
185  if (normReduction < tolerance) {
186  Log::info() << "DRIPCG: Achieved required reduction in residual norm."
187  << std::endl;
188  break;
189  }
190 
191  vvecs.push_back(rr);
192  zvecs.push_back(ss);
193  scals.push_back(1.0 / rdots);
194  }
195 
196  // Generate the (second-level) Limited Memory Preconditioner
197  lmp_.update(B);
198 
199  return normReduction;
200 }
