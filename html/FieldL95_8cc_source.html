112 void FieldL95::interp(const LocsL95 & locs, GomL95 & gom) const {
113  const double dres = static_cast<double>(resol_);
114  for (int jobs = 0; jobs < locs.nobs(); ++jobs) {
115  int ii = round(locs[jobs] * dres);
116  ASSERT(ii >= 0 && ii <= resol_);
117  if (ii == resol_) ii = 0;
118  gom[gom.current()+jobs] = x_[ii];
119  }
120  gom.current() += locs.nobs();
121 }

123 void FieldL95::interpAD(const LocsL95 & locs, const GomL95 & gom) {
124  const double dres = static_cast<double>(resol_);
125  if (gom.current() == 0) gom.current() = gom.nobs();
126  gom.current() -= locs.nobs();
127  for (int jobs = 0; jobs < locs.nobs(); ++jobs) {
128  int ii = round(locs[jobs] * dres);
129  ASSERT(ii >= 0 && ii <= resol_);
130  if (ii == resol_) ii = 0;
131  x_[ii] += gom[gom.current()+jobs];
132  }
133 }
