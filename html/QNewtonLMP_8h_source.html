26 /* The QUASI-NEWTON Limited Memory preconditioner matrix \f$ C \f$
27  * for the system matrix \f$ AB = (I + Ht Rinv H B) \f$
28  *
29  * It is defined as
30  * \f$ C_(k+1) = (I - rho_k ph_k q_k^T) C_k (I - rho_k q_k p_k^T) + rho_k
31  * ph_k p_k^T\f$
32  *
33  * For details, we refer to S. Gratton, A. Sartenaer and J. Tshimanga,
34  * SIAM J. Optim.,21(3),912â€“935,2011 and S. Gurol, PhD Manuscript, 2013.
35  *
36  * The solvers represent matrices as objects that implement a "multiply"
37  * method. This class defines objects that apply the preconditioner matrix \f$
38  * P \f$.
39  */
40

80 QNewtonLMP<VECTOR, BMATRIX>::QNewtonLMP(const eckit::Configuration &conf)
81  : maxpairs_(0),

84  maxouter_(0),
85  update_(1) {

89 
90  const eckit::LocalConfiguration precond(conf, "preconditioner");
91  if (precond.has("maxpairs")) maxpairs_ = precond.getInt("maxpairs");
92  if (maxpairs_ > 0) {
93  std::string old;
94  if (precond.has("useoldpairs")) old = precond.getString("useoldpairs");
95  useoldpairs_ = (old == "on" || old == "true");
96  if (useoldpairs_ && precond.has("maxnewpairs")) {
97  maxnewpairs_ = precond.getInt("maxnewpairs");
98  maxnewpairs_ = std::min(maxpairs_, maxnewpairs_);
99  } else {
100  maxnewpairs_ = maxpairs_;
101  }
102  }
103  }

108 template <typename VECTOR, typename BMATRIX>
109 void QNewtonLMP<VECTOR, BMATRIX>::push(const VECTOR &p, const VECTOR &ph,
110  const VECTOR &ap, const double &rho) {
111  ASSERT(savedP_.size() <= maxnewpairs_);
112  if (maxnewpairs_ > 0 && update_ < maxouter_) {
113  if (savedP_.size() == maxnewpairs_) {
114  savedP_.erase(savedP_.begin());
115  savedPh_.erase(savedPh_.begin());
116  savedAP_.erase(savedAP_.begin());
117  savedrhos_.erase(savedrhos_.begin());
118  }
119  savedP_.push_back(p);
120  savedPh_.push_back(ph);
121  savedAP_.push_back(ap);
122  savedrhos_.push_back(1.0 / rho);
123  }
124 }

128 template <typename VECTOR, typename BMATRIX>
129 void QNewtonLMP<VECTOR, BMATRIX>::update(const BMATRIX &Bmat) {
130  Log::debug() << "QNewtonLMP size saved Ph = " << savedPh_.size()
131  << std::endl;
132  Log::debug() << "QNewtonLMP size saved P = " << savedP_.size()
133  << std::endl;
134  Log::debug() << "QNewtonLMP size saved AP = " << savedAP_.size()
135  << std::endl;
136  Log::debug() << "QNewtonLMP size saved rhos = " << savedrhos_.size()
137  << std::endl;
138 
139  const unsigned nvec = savedPh_.size();
140  ASSERT(savedP_.size() == nvec);
141  ASSERT(savedAP_.size() == nvec);
142  ASSERT(savedrhos_.size() == nvec);
143 
144  if (!useoldpairs_ || nvec >= maxpairs_) {
145  Ph_.clear();
146  P_.clear();
147  AP_.clear();
148  BAP_.clear();
149  rhos_.clear();
150  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
151  usedpairIndx_[kiter] = 0;
152  }
153  }
154 
155  if (nvec > 0 && update_ < maxouter_) {
156  Log::info() << "QNewtonLMP: update " << update_
157  << ", max = " << maxouter_ - 1 << std::endl;
158  unsigned newpairs = std::min(nvec, maxpairs_);
159  unsigned oldpairs = P_.size();
160  unsigned rmpairs = 0;
161  // First remove pairs we no longer need.
162  if (oldpairs + newpairs > maxpairs_) {
163  rmpairs = oldpairs + newpairs - maxpairs_;
164  for (unsigned jv = 0; jv < rmpairs; ++jv) {
165  Ph_.erase(Ph_.begin());
166  P_.erase(P_.begin());
167  AP_.erase(AP_.begin());
168  BAP_.erase(BAP_.begin());
169  rhos_.erase(rhos_.begin());
170  }
171  oldpairs -= rmpairs;
172  // Keep information on how many pairs are used at each minimization
173  // loop
174  unsigned removed = rmpairs;
175  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
176  unsigned rmiter = std::min(usedpairIndx_[kiter], removed);
177  usedpairIndx_[kiter] -= rmiter;
178  removed -= rmiter;
179  }
180  ASSERT(removed == 0);
181  }
182  ASSERT(P_.size() == oldpairs);
183  ASSERT(oldpairs + newpairs <= maxpairs_);
184 
185  // Add the new pairs.
186  for (unsigned jv = nvec - newpairs; jv < nvec; ++jv) {
187  P_.push_back(savedP_[jv]);
188  Ph_.push_back(savedPh_[jv]);
189  AP_.push_back(savedAP_[jv]);
190  rhos_.push_back(savedrhos_[jv]);
191  // Save B*ap
192  VECTOR ww(savedAP_[jv]);
193  Bmat.multiply(savedAP_[jv], ww);
194  BAP_.push_back(ww);
195  }
196  ASSERT(P_.size() == oldpairs + newpairs);
197 
198  Log::info() << "Number of inner iterations : " << nvec << std::endl;
199  Log::info() << "Number of maximum pairs : " << maxpairs_
200  << std::endl;
201  Log::info() << "Number of used total pairs : " << oldpairs + newpairs
202  << std::endl;
203  Log::info() << "Number of new pairs : " << newpairs
204  << std::endl;
205  Log::info() << "Number of removed old pairs : " << rmpairs
206  << std::endl;
207  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
208  Log::info() << "Number of used pairs from outer loop " << kiter + 1
209  << " : " << usedpairIndx_[kiter];
210  }
211  }
212 
213  ++update_;
214  savedP_.clear();
215  savedPh_.clear();
216  savedAP_.clear();
217  savedrhos_.clear();
218 }

222 template <typename VECTOR, typename BMATRIX>
223 void QNewtonLMP<VECTOR, BMATRIX>::multiply(const VECTOR &a, VECTOR &b) const {
224  b = a;
225  const unsigned nvec = P_.size();
226  if (nvec != 0) {
227  std::vector<double> etas(nvec, 0);
228  // etas.clear();
229  for (int iiter = nvec - 1; iiter >= 0; iiter--) {
230  double eta = dot_product(b, P_[iiter]);
231  eta *= rhos_[iiter];
232  etas[iiter] = eta;
233  b.axpy(-eta, AP_[iiter]);
234  }
235  b *= dot_product(AP_[nvec - 1], Ph_[nvec - 1]) /
236  dot_product(AP_[nvec - 1], AP_[nvec - 1]);
237  for (unsigned iiter = 0; iiter < nvec; ++iiter) {
238  double sigma = dot_product(b, BAP_[iiter]);
239  sigma *= rhos_[iiter];
240  sigma -= etas[iiter];
241  b.axpy(-sigma, Ph_[iiter]);
242  }
243  }
244 }

248 template <typename VECTOR, typename BMATRIX>
249 void QNewtonLMP<VECTOR, BMATRIX>::tmultiply(const VECTOR &a, VECTOR &b) const {
250  b = a;
251  const unsigned nvec = P_.size();
252  if (nvec != 0) {
253  std::vector<double> etas;
254  etas.clear();
255  for (int iiter = nvec - 1; iiter >= 0; iiter--) {
256  double eta = dot_product(b, Ph_[iiter]);
257  eta *= rhos_[iiter];
258  etas.push_back(eta);
259  b.axpy(-eta, BAP_[iiter]);
260  }
261  b *= dot_product(AP_[nvec - 1], AP_[nvec - 1]) /
262  dot_product(AP_[nvec - 1], Ph_[nvec - 1]);
263  for (unsigned iiter = 0; iiter < nvec; ++iiter) {
264  double sigma = dot_product(b, AP_[iiter]);
265  sigma *= rhos_[iiter];
266  sigma -= etas[iiter];
267  b.axpy(-sigma, P_[iiter]);
268  }
269  }
270 }
