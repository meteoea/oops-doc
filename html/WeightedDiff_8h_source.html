108 template <typename MODEL, typename INCR, typename FLDS>
109 void WeightedDiff<MODEL, INCR, FLDS>::doInitialize(
110  const FLDS &xx, const util::DateTime &end, const util::Duration &tstep) {
111  const util::DateTime bgn(xx.validTime());
112  if (!linit_ && bgn <= end_ && end >= bgn_) {
113  if (tstep_ == util::Duration(0)) tstep_ = tstep;
114  ASSERT(tstep_ > util::Duration(0));
115  weights_ = wfct_.setWeights(bgn_, end_, tstep_);
116  linit_ = true;
117  ASSERT(weights_.find(vtime_) != weights_.end());
118  weights_[vtime_] -= 1.0;
119  }
120  bgnleg_ = bgn;
121  endleg_ = end;
122  Log::info() << "WeightedDiff: initialized"
123  << " bgnleg_ = " << bgnleg_ << ", endleg_ = " << endleg_
124  << ", bgn_ = " << bgn_ << ", end_ = " << end_ << std::endl;
125  Log::info() << " Weights:" << std::endl;
126  for (auto const &weight : weights_) {
127  Log::info() << " " << weight.first << " : " << weight.second
128  << std::endl;
129  }
130 }

134 template <typename MODEL, typename INCR, typename FLDS>
135 void WeightedDiff<MODEL, INCR, FLDS>::doProcessing(const FLDS &xx) {
136  const util::DateTime now(xx.validTime());
137  if (((bgnleg_ < end_ && endleg_ > bgn_) || bgnleg_ == endleg_) &&
138  (now != endleg_ || now == end_ || now == bgnleg_)) {
139  ASSERT(weights_.find(now) != weights_.end());
140  const double zz = weights_[now];
141  avg_->accumul(zz, xx);
142  sum_ += zz;
143  Log::debug() << "WeightedDiff: time = " << now << ", weight = " << zz
144  << ", sum = " << sum_ << std::endl;
145  }
146 }
