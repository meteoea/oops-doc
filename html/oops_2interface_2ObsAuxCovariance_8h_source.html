122 template <typename MODEL>
123 ObsAuxCovariance<MODEL>::ObsAuxCovariance(const ObservationSpace_ &os,
124  const eckit::Configuration &conf)
125  : size_(0), covs_(), conf_(new eckit::LocalConfiguration(conf)) {
126  Log::trace() << "ObsAuxCovariance<MODEL>::ObsAuxCovariance starting"
127  << std::endl;
128  util::Timer timer(classname(), "ObsAuxCovariance");
129 
130  std::vector<eckit::LocalConfiguration> covconfs;
131  conf.get("covarianceObsAux", covconfs);
132  size_ = covconfs.size();
133 
134  for (auto const &config : covconfs) {
135  std::string key(config.getString("type"));
136  std::unique_ptr<ObsAuxCovarianceBase_> auxcov(
137  ObsAuxCovarianceFactory<MODEL>::create(os, config));
138  covs_.emplace(key, std::move(auxcov));
139  }
140  Log::trace() << "ObsAuxCovariance<MODEL>::ObsAuxCovariance done" << std::endl;
141 }

188 template <typename MODEL>
189 void ObsAuxCovariance<MODEL>::update(const ObsAuxControl_ &xx,
190  const eckit::Configuration &conf) {
191  Log::trace() << "ObsAuxCovariance<MODEL>::update starting" << std::endl;
192  util::Timer timer(classname(), "update");
193 
194  conf_.reset(new eckit::LocalConfiguration(conf));
195 
196  std::vector<eckit::LocalConfiguration> confs;
197  conf.get("covarianceObsAux", confs);
198 
199  for (iter_ it = covs_.begin(); it != covs_.end(); ++it) {
200  for (auto const &config : confs) {
201  if (config.getString("type") == it->first && config.getBool("update")) {
202  it->second->update(xx.obsauxcontrol(it->first), config);
203  }
204  }
205  }
206  Log::trace() << "ObsAuxCovariance<MODEL>::update done" << std::endl;
207 }
