93 template <typename MODEL>
94 ObservationSpaces<MODEL>::ObservationSpaces(const eckit::Configuration &conf,
95  const Geometry_ &geom,
96  const util::DateTime &bgn,
97  const util::DateTime &end,
98  const bool lscreened)
99  : wbgn_(bgn), wend_(end), lscreened_(lscreened), spaces_(0) {
100  std::vector<eckit::LocalConfiguration> obsconf;
101  conf.get("ObsTypes", obsconf);
102  for (std::size_t jj = 0; jj < obsconf.size(); ++jj) {
103  Log::debug() << "ObservationSpaces::ObservationSpaces : conf "
104  << obsconf[jj] << std::endl;
105  std::shared_ptr<ObsSpace_> tmp(
106  new ObsSpace_(obsconf[jj], geom, bgn, end, lscreened));
107  spaces_.push_back(tmp);
108  // Generate locations etc... if required
109  if (obsconf[jj].has("Generate")) {
110  const eckit::LocalConfiguration gconf(obsconf[jj], "Generate");
111  spaces_[jj]->generateDistribution(gconf);
112  }
113  }
114 }
