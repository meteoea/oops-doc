68 template <typename MODEL>
69 ControlIncrement<MODEL> *LBMinimizer<MODEL>::doMinimize(
70  const eckit::Configuration &config) {
71  int ninner = config.getInt("ninner");
72  double gnreduc = config.getDouble("gradient_norm_reduction");
73 
74  if (gradJb_ == 0) {
75  gradJb_.reset(new CtrlInc_(J_.jb()));
76  } else {
77  gradJb_.reset(new CtrlInc_(J_.jb().resolution(), *gradJb_));
78  }
79 
80  Log::info() << std::endl;
81  Log::info() << classname() << ": max iter = " << ninner
82  << ", requested norm reduction = " << gnreduc << std::endl;
83 
84  // Define the matrices
85  const Bmat_ B(J_);
86  const LBHessianMatrix_ LBHessianMatrix(J_);
87 
88  // Compute RHS (sum dx^{b}_{i} + ) B H^T R^{-1} d
89  CtrlInc_ rhs(J_.jb());
90  CtrlInc_ brhs(J_.jb());
91  J_.computeGradientFG(rhs);
92  J_.jb().multiplyB(rhs, brhs);
93  J_.jb().addGradientFG(brhs, *gradJb_);
94 
95  brhs *= -1.0;
96  Log::info() << classname() << " rhs" << brhs << std::endl;
97 
98  // Define minimisation starting point
99  CtrlInc_ *dx = new CtrlInc_(J_.jb());
100 
101  // Solve the linear system
102  this->solve(*dx, brhs, LBHessianMatrix, ninner, gnreduc);
103 
104  Log::info() << classname() << " output increment:" << *dx << std::endl;
105 
106  // Update gradient Jb
107  *gradJb_ += *dx;
108 
109  return dx;
110 }
