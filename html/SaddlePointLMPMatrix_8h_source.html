103 template <typename MODEL>
104 void SaddlePointLMPMatrix<MODEL>::setup(const std::vector<SPVector_> &xyVEC,
105  const std::vector<SPVector_> &pqVEC) {
106  xyVEC_.clear();
107  pqVEC_.clear();
108  RpVEC_.clear();
109  RqVEC_.clear();
110  for (unsigned ii = 0; ii < xyVEC.size(); ++ii) {
111  xyVEC_.push_back(xyVEC[ii]);
112  pqVEC_.push_back(pqVEC[ii]);
113  }
114 
115  nvec_ = xyVEC_.size();
116 
117  if (nvec_ != 0) {
118  spvecinit_.reset(new SPVector_(xyVEC_[0]));
119  (*spvecinit_).zero();
120 
121  SPVector_ ww(xyVEC_[0]);
122  SPVector_ yy(xyVEC_[0]);
123 
124  // RpRqVEC = pqVEC - P0 * xyVEC
125  for (unsigned jv = 0; jv < nvec_; ++jv) {
126  this->Pinitmultiply(xyVEC_[jv], ww);
127  yy = pqVEC_[jv];
128  yy -= ww;
129  RpVEC_.push_back(yy.lambda());
130  RqVEC_.push_back(yy.dx());
131  }
132 
133  // Z = RpVEC'*xVEC
134  ZMat_.resize(nvec_, nvec_);
135  double calpha = 10;
136  for (unsigned jj = 0; jj < nvec_; ++jj) {
137  for (unsigned ii = 0; ii < nvec_; ++ii) {
138  ZMat_(ii, jj) = calpha * dot_product(RpVEC_[ii], xyVEC_[jj].lambda());
139  }
140  }
141 
142  // Get F Matrix explicitly
143  // Set Identity matrix
144  Eigen::MatrixXd Ik;
145  Ik = Eigen::MatrixXd::Identity(2 * nvec_, 2 * nvec_);
146 
147  // Apply Fmultiply to identity matrix
148  FMat_.resize(2 * nvec_, 2 * nvec_);
149  Eigen::VectorXd output(2 * nvec_);
150  Eigen::VectorXd input(2 * nvec_);
151  for (int ii = 0; ii < 2 * nvec_; ii++) {
152  input = Ik.col(ii);
153  Fmultiply(input, output);
154  FMat_.col(ii) = output;
155  }
156  }
157 }

160 template <typename MODEL>
161 void SaddlePointLMPMatrix<MODEL>::Pinitmultiply(const SPVector_ &x,
162  SPVector_ &z) const {
163  // P0 = [D 0 L]
164  // [0 R 0]
165  // [Lt 0 0]
166  //
167  // where M = I.
168 
169  CtrlInc_ ww(x.lambda().dx());
170 
171  // ADJ block
172  j_.runAD(ww, true);
173  z.dx() = x.lambda().dx();
174  z.dx() -= ww;
175 
176  // TLM block
177  ww = x.dx();
178  j_.runTL(ww, true);
179  z.lambda().zero();
180  z.lambda().dx() = x.dx();
181  z.lambda().dx() -= ww;
182 
183  // Diagonal block
184  DualVector<MODEL> diag;
185  diag.dx(new CtrlInc_(j_.jb()));
186  j_.jb().multiplyB(x.lambda().dx(), diag.dx());
187  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
188  diag.append(j_.jterm(jj).multiplyCovar(*x.lambda().getv(jj)));
189  }
190 
191  z.lambda() += diag;
192 
193  /*
194  // *********************************
195  // P0 = [D 0 I]
196  // [0 R 0]
197  // [I 0 0]
198  z.dx() = x.lambda().dx();
199  DualVector<MODEL> diag;
200  diag.dx(new CtrlInc_(j_.jb()));
201  j_.jb().multiplyB(x.lambda().dx(), diag.dx());
202  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
203  diag.append(j_.jterm(jj).multiplyCovar(*x.lambda().getv(jj)));
204  }
205  z.lambda().zero();
206  z.lambda().dx() = x.dx();
207  z.lambda() += diag;
208  // *********************************
209  */
210 }

212 template <typename MODEL>
213 void SaddlePointLMPMatrix<MODEL>::multiply(const SPVector_ &x,
214  SPVector_ &z) const {
215  // z = P0inv*x - P0inv*R*Finv*G*P0inv*x
216 
217  SaddlePointPrecondMatrix<MODEL> P0inv(j_);
218  SPVector_ svw(x);
219  SPVector_ svt(x);
220 
221  P0inv.multiply(x, z);
222  if (nvec_ != 0) {
223  Eigen::VectorXd rhs(2 * nvec_);
224  Eigen::VectorXd finvx(2 * nvec_);
225  this->Gmultiply(z, rhs); // rhs = G*z
226  finvx = FMat_.colPivHouseholderQr().solve(rhs); // finvx = Finv*rhs
227  this->Rmultiply(finvx, svw); // svw = R * finvx
228  P0inv.multiply(svw, svt); // svt = P0inv * svw
229  z -= svt; // z = z - svt
230  }
231 }

234 template <typename MODEL>
235 void SaddlePointLMPMatrix<MODEL>::Gmultiply(const SPVector_ &x,
236  Eigen::VectorXd &z) const {
237  Eigen::VectorXd v1(nvec_);
238  Eigen::VectorXd v2(nvec_);
239  Eigen::VectorXd ax(nvec_);
240  Eigen::VectorXd axt(nvec_);
241  Eigen::MatrixXd ZMatt(nvec_, nvec_);
242  ZMatt = ZMat_.transpose();
243 
244  // Rp'*x and Rq'*y
245  for (unsigned jj = 0; jj < nvec_; ++jj) {
246  v1(jj) = dot_product(RpVEC_[jj], x.lambda());
247  v2(jj) = dot_product(RqVEC_[jj], x.dx());
248  }
249 
250  ax = ZMat_.colPivHouseholderQr().solve(v1); // Zinv*Rp'*x
251  axt = ZMatt.colPivHouseholderQr().solve(v2); // Ztinv*Rq'*y
252  z << ax, axt; // z = [ax ; axt]
253 }

256 template <typename MODEL>
257 void SaddlePointLMPMatrix<MODEL>::Rmultiply(Eigen::VectorXd &x,
258  SPVector_ &rx) const {
259  LagVector_ ww(rx.lambda());
260  ww.zero();
261  CtrlInc_ zz(rx.dx());
262  zz.zero();
263 
264  for (unsigned ii = 0; ii < nvec_; ++ii) {
265  ww.axpy(x(ii + nvec_), RpVEC_[ii]);
266  zz.axpy(x(ii), RqVEC_[ii]);
267  }
268  rx.lambda() = ww;
269  rx.dx() = zz;
270 }

274 void SaddlePointLMPMatrix<MODEL>::Fmultiply(Eigen::VectorXd &x,
275  Eigen::VectorXd &fx) const {
276  SaddlePointPrecondMatrix<MODEL> P0inv(j_);
277  SPVector_ tmp(*spvecinit_);
278  SPVector_ tmp2(*spvecinit_);
279  Eigen::VectorXd ww(2 * nvec_);
280 
281  fx = x;
282  this->Rmultiply(x, tmp);
283  P0inv.multiply(tmp, tmp2);
284  this->Gmultiply(tmp2, ww);
285  fx += ww;
