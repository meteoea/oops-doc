147 template <typename MODEL>
148 ObsAuxCovarianceBase<MODEL> *ObsAuxCovarianceFactory<MODEL>::create(
149  const ObservationSpace<MODEL> &os, const eckit::Configuration &conf) {
150  const std::string id = conf.getString("type");
151  typename std::map<std::string, ObsAuxCovarianceFactory<MODEL> *>::iterator
152  jloc = getMakers().find(id);
153  if (jloc == getMakers().end()) {
154  Log::error() << id << " does not exist in ObsAuxCovarianceFactory."
155  << std::endl;
156  ABORT("Element does not exist in ObsAuxCovarianceFactory.");
157  }
158  ObsAuxCovarianceBase<MODEL> *ptr = jloc->second->make(os, conf);
159  return ptr;
160 }
