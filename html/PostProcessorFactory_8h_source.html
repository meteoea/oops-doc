83 template <typename FLDS>
84 PostBase<FLDS> *PostProcessorFactory<FLDS>::create(
85  const eckit::Configuration &config) {
86  std::string id = config.getString("post_processor_type");
87  Log::trace() << "Post processor Type=" << id << std::endl;
88  typename std::map<std::string, PostProcessorFactory<FLDS> *>::iterator j =
89  getMakers().find(id);
90  if (j == getMakers().end()) {
91  Log::error() << id << " does not exist in post processor factory."
92  << std::endl;
93  ABORT("Element does not exist in PostProcessorFactory.");
94  }
95  return (*j).second->make(config);
96 }
