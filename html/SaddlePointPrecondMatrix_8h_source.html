67 template <typename MODEL>
68 void SaddlePointPrecondMatrix<MODEL>::multiply(const SPVector_ &x,
69  SPVector_ &z) const {
70  z.lambda().clear();
71  z.lambda().dx(new CtrlInc_(j_.jb()));
72  z.dx(new CtrlInc_(j_.jb()));
73 
74  Lhatinvt(x.dx(), z.lambda().dx(), norder_, idmodel_);
75 
76  CtrlInc_ l(z.lambda().dx());
77  j_.jb().multiplyB(z.lambda().dx(), l);
78  l *= -1.0;
79  l += x.lambda().dx();
80 
81  Lhatinv(l, z.dx(), norder_, idmodel_);
82 
83  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
84  z.lambda().append(j_.jterm(jj).multiplyCoInv(*x.lambda().getv(jj)));
85  }
86 }

92  const int norder,
93  const bool idmodel) const {
94  // Approximate L=I + (I-L) + (I-L)^2 + ... + (I-L)^norder
95  CtrlInc_ ww(xx);
96 
97  zz = xx;
