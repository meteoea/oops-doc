124 template<typename MODEL>
125 void StaticWithTLModelCovariance4D<MODEL>::advectedLinearize(
126  const State4D_ &xb, const Geometry_ &resol, const eckit::Configuration & config) {
127  const util::Duration length(config.getString("window_length"));
128  PostProcessor<State_> post;
129 
130  const eckit::LocalConfiguration low_resol_config(config, "low_resolution");
131  low_resol_.reset(new Geometry_(low_resol_config));
132  const eckit::LocalConfiguration modelConf(config, "model");
133  model_.reset(new Model_(*low_resol_, modelConf));
134  const eckit::LocalConfiguration tlmConf(config, "linearmodel");
135  ModelAuxCtrl_ moderr(*low_resol_, *model_, config);
136  modaux_.reset(new ModelAuxIncrement_(*low_resol_, config));
137 
138  State_ x0(xb[0]);
139  if (config.has("matrix")) {
140  to_low_resol_.reset( MatrixFactory_::create(config));
141  to_low_resol_->multiply(x0);
142  }
143  static_->linearize(x0, *low_resol_);
144  tlm_.clear();
145  post.enrollProcessor(
146  new TrajectorySaver_(x0, moderr, *low_resol_, tlm_, tlmConf));
147  model_->forecast(x0, moderr, length, post);
148 
149  Log::trace() << "StaticWithTLModelCovariance4D linearized.";
150 }

159 template<typename MODEL>
160 void StaticWithTLModelCovariance4D<MODEL>::advectedMultiply(
161  const Increment4D_ &dxi, Increment4D_ &dxo) const {
162  util::Duration length = dxi[dxi.last()].validTime() - dxi[0].validTime();
163  PostProcessor<Increment_> no_post;
164 
165  dxo.zero();
166  std::shared_ptr<ObserverTL_> obstl(new ObserverTL_(dxo, to_low_resol_));
167  PostProcessorTL_ costtl;
168  costtl.enrollProcessor(obstl);
169 
170  std::shared_ptr<ObserverAD_> obsad(new ObserverAD_(dxi, to_low_resol_));
171  PostProcessorAD_ costad;
172  costad.enrollProcessor(obsad);
173 
174  Increment_ dx(*low_resol_, variables_, dxi[dxi.first()].validTime(),
175  dxi[dxi.last()].validTime());
176  dx.zero();
177 
178  tlm_[0]->forecastAD(dx, *modaux_, length, no_post, costad);
179  static_->multiply(dx, dx);
180  tlm_[0]->forecastTL(dx, *modaux_, length, no_post, costtl);
181 }
