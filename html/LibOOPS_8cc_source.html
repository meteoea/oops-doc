49 void LibOOPS::initialise() {
50  rank_ = static_cast<int>(eckit::mpi::comm().rank());
51  const int it = getEnv("OOPS_TRACE", 0);
52  if (it > 0 && rank_ == 0) trace_ = true;
53  if (it < 0) {
54  trace_ = true;
55  pretrace_ += "[" + std::to_string(rank_) + "]";
56  }
57  const int id = getEnv("OOPS_DEBUG", 0);
58  if (id > 0 && rank_ == 0) debug_ = true;
59  if (id < 0) {
60  debug_ = true;
61  predebug_ += "[" + std::to_string(rank_) + "]";
62  }
63 }
