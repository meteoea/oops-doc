46  int execute(const eckit::Configuration& fullConfig) const {
47  // Setup resolution
48  const eckit::Configuration resolConfig(fullConfig, "resolution");
49  const Geometry_ resol(resolConfig);
50 
51  // Setup Model
52  const eckit::Configuration modelConfig(fullConfig, "model");
53  const Model_ model(resol, modelConfig);
54 
55  unsigned nm = config.getElementSize("member");
56  for (unsigned jj = 0; jj < nm; ++jj) {
57  // Setup initial state
58  const eckit::Configuration initialConfig(fullConfig, "member", jj);
59  State_ xx(resol, initialConfig);
60  Log::test() << "Initial state: " << xx.norm() << std::endl;
61 
62  // Setup augmented state
63  const ModelAux_ moderr(initialConfig);
64 
65  // Setup times
66  const util::Duration fclength(fullConfig.getString("forecast_length"));
67  const util::DateTime bgndate(xx.validTime());
68  const util::DateTime enddate(bgndate + fclength);
69  Log::info() << "Running forecast " << jj << " from " << bgndate << " to "
70  << enddate << std::endl;
71 
72  // Setup forecast outputs
73  PostProcessor<State_> post;
74 
75  const eckit::Configuration prtConfig(fullConfig, "prints", true);
76  post.enrollProcessor(new StateInfo<State_>("fc", prtConfig));
77 
78  const eckit::Configuration outConfig(fullConfig, "output");
79  post.enrollProcessor(new StateWriter<State_>(bgndate, outConfig));
80 
81  // Run forecast
82  model.forecast(xx, moderr, fclength, post);
83 
84  Log::test() << "Final state " << jj << " : " << xx.norm() << std::endl;
85  }
86  return 0;
87  }
88  // -----------------------------------------------------------------------------
89  private:
90  std::string appname() const {
91  return "oops::EnsForecast<" + MODEL::name() + ">";
92  }
