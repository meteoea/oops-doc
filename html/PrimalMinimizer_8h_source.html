63 template <typename MODEL>
64 ControlIncrement<MODEL> *PrimalMinimizer<MODEL>::doMinimize(
65  const eckit::Configuration &config) {
66  int ninner = config.getInt("ninner");
67  double gnreduc = config.getDouble("gradient_norm_reduction");
68 
69  bool lauxMinDiags = false;
70  if (config.has("onlineDiagnostics")) {
71  const eckit::LocalConfiguration onlineDiag(config, "onlineDiagnostics");
72  lauxMinDiags = onlineDiag.getBool("auxMinDiags");
73  }
74 
75  Log::info() << classname() << ": max iter = " << ninner
76  << ", requested norm reduction = " << gnreduc << std::endl;
77 
78  // Define the matrices
79  Hessian_ hessian(J_, lauxMinDiags);
80  Bmat_ B(J_);
81 
82  // Compute RHS
83  CtrlInc_ rhs(J_.jb());
84  J_.computeGradientFG(rhs);
85  J_.jb().addGradientFG(rhs);
86  Log::info() << classname() << " rhs" << rhs << std::endl;
87 
88  // Define minimisation starting point
89  CtrlInc_ *dx = new CtrlInc_(J_.jb());
90 
91  // Solve the linear system
92  double reduc = this->solve(*dx, rhs, hessian, B, ninner, gnreduc);
93 
94  Log::test() << classname() << ": reduction in residual norm = " << reduc
95  << std::endl;
96  Log::info() << classname() << " output" << *dx << std::endl;
97 
98  return dx;
99 }
