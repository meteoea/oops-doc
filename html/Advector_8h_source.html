112 template <typename MODEL>
113 void Advector<MODEL>::set_wind(const State4D_ &wind,
114  const util::DateTime &date_start) {
115  util::Duration step(wind[1]->validTime() - wind[0]->validTime());
116  double deltat = (double)step.toSeconds();
117  unsigned kstart = 0;
118  unsigned kstop = 0;
119 
120  // Initialize with vector (0,0)
121  set_vector(config_);
122 
123  for (unsigned k = 0; k < wind.size(); k++) {
124  if (wind[k]->validTime() == time_) kstop = k;
125  if (wind[k]->validTime() == date_start) kstart = k;
126  }
127 
128  // Advect the grid step by step
129  Advector<MODEL> s_star(*this);
130  for (unsigned k = kstop; k > kstart; k -= 1) {
131  for (unsigned iter = 0; iter < niter_; iter++) {
132  Accumulator<MODEL, State_, State_> uv(*wind[k - 1]);
133  uv.accumul(wind_coeff_, *wind[k - 1]);
134  s_star.advect(uv, vars_, uv);
135 
136  Accumulator<MODEL, State_, State_> delta_xy(uv);
137  delta_xy.accumul(deltat, uv);
138 
139  s_star = (*this);
140  s_star -= delta_xy;
141  }
142  (*this) = s_star;
143  }
144 }
