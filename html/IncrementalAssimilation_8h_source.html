161 template <typename MODEL>
162 void IncrementalAssimilation<MODEL>::assimilate(
163  const eckit::Configuration &config) {
165  xx_.reset(new CtrlVariable_(J_.jb().getBackground()));
166  Log::trace() << "IncrementalAssimilation: first guess has been set up"
167  << std::endl;
168 
170  std::vector<eckit::LocalConfiguration> iterconfs;
171  config.get("variational.iteration", iterconfs);
172  const size_t nouterconfSize = iterconfs.size();
173 
175  eckit::LocalConfiguration minConf(config, "minimizer");
176  minim_.reset(MinFactory<MODEL>::create(minConf, J_));
177 
180  util::Restart restarter(
181  {"FirstGuess", "Minimizer", "IncrementalAssimilation"});
182  this->connectToRestart(restarter);
183  restarter.restart({"FirstGuess", "IncrementalAssimilation"});
184 
186  if (not restarter.isRestarted() &&
187  config.has("variational.WarmStart.FirstGuess")) {
188  xx_->read(config.getSubConfiguration("variational.WarmStart.FirstGuess"));
189  }
190 
192  if (not restarter.isRestarted() &&
193  config.has("variational.WarmStart.RecentreFirstGuess")) {
194  eckit::LocalConfiguration recentreFirstGuessConf(
195  config, "variational.WarmStart.RecentreFirstGuess");
196  this->recentreFirstGuess(recentreFirstGuessConf);
197  }
198 
200  std::shared_ptr<StateSaver<State_> > stateSaver;
201  std::shared_ptr<StateWriter<State_> > stateWriter;
202  std::unique_ptr<eckit::LocalConfiguration> late4Dconf;
203 
205  std::unique_ptr<CtrlInc_> dx;
206 
208  this->setOuterLoopBounds(nouterconfSize);
209 
210  Log::info() << "Running incremental assimilation with " << nouterconfSize
211  << " outer iterations." << std::endl;
212 
214  for (size_t jj = jstart_; jj < jend_; ++jj) {
216  this->setOuterLoopIndex(jj, iterconfs[jj]);
217 
218  // Get configuration for current outer iteration
219  Log::info() << "IncrementalAssimilation: Configuration for outer iteration "
220  << jouter_ << ":\n"
221  << iterconfs[jj];
222 
224  util::LogbookWriter<size_t> log("OuterLoop", jouter_);
225 
227  this->setSkipFinalNonlinearTrajectory(false);
228  if (iterconfs[jj].has("skipFinalTraj")) {
229  const bool skipTraj = iterconfs[jj].getBool("skipFinalTraj");
230  this->setSkipFinalNonlinearTrajectory(skipTraj);
231  }
232 
234  if (iterconfs[jj].has("late4d")) {
235  this->setLate4D(true);
236  late4Dconf.reset(new eckit::LocalConfiguration(iterconfs[jj], "late4d"));
237  }
238 
240  PostProcessor<State_> post;
242  if (iterconfs[jj].has("prints")) {
243  const eckit::LocalConfiguration prtConfig(iterconfs[jj], "prints");
244  post.enrollProcessor(new StateInfo<State_>("traj", prtConfig));
245  }
247  if (this->isLate4D()) {
248  const eckit::LocalConfiguration late4DstateConf(*late4Dconf, "state");
249  if (late4DstateConf.has("writeToDisk")) {
250  const eckit::LocalConfiguration late4DstateWriteConf(late4DstateConf,
251  "writeToDisk");
252  stateWriter.reset(new StateWriter<State_>(late4DstateWriteConf));
253  post.enrollProcessor(stateWriter);
254  } else {
255  stateSaver.reset(new StateSaver<State_>(late4DstateConf));
256  post.enrollProcessor(stateSaver);
257  }
258  }
259 
261  J_.linearize(*xx_, iterconfs[jj], post);
262 
264  {
265  restarter.checkpoint(iterconfs[jj], {"CostFunction"});
266  util::LogbookWriter<size_t> log("CostFunctionCheckpoint", jouter_);
267  }
268 
270  if (jouter_ == 0 && config.has("variational.WarmStart.Minimizer")) {
271  minim_->warmStart(
272  config.getSubConfiguration("variational.WarmStart.Minimizer"));
273  }
274 
276  if (jj == jstart_) restarter.restart({"Minimizer"});
277 
279  dx.reset(minim_->minimize(iterconfs[jj]));
280  if (jj == jend_) minim_.reset();
281 
283  J_.addIncrement(*xx_, *dx);
284 
286  if (iterconfs[jj].has("increment"))
287  dx->write(iterconfs[jj].getSubConfiguration("increment"));
288 
290  if (!isLate4D()) J_.resetLinearization();
291 
293  restarter.checkpoint(
294  iterconfs[jj],
295  {"FirstGuess", "Minimizer", "IncrementalAssimilation"});
296  }
297 
299  if (this->isLate4D()) {
301  this->late4Dstart(*late4Dconf, stateSaver, *dx);
303  stateSaver.reset();
304  }
305 }

435 template <typename MODEL>
436 void IncrementalAssimilation<MODEL>::late4Dstart(
437  const eckit::LocalConfiguration &conf,
438  const std::shared_ptr<StateSaver<State_> > stateSaver, const CtrlInc_ &dx) {
439  Log::info() << "IncrementalAssimilation::late4Dstart starting" << std::endl;
440  const eckit::LocalConfiguration late4DstateConf(conf, "state");
441 
446  PostProcessorTL<Increment_> cost;
447  PostProcessor<Increment_> post;
448 
449  std::shared_ptr<StateSaver<Increment_> > incrementSaver;
450  incrementSaver.reset(new StateSaver<Increment_>(late4DstateConf));
451  post.enrollProcessor(incrementSaver);
452 
453  CtrlInc_ dxtmp(dx);
454  J_.runTL(dxtmp, cost, post);
455 
457  const std::vector<std::shared_ptr<Increment_> > &increments(
458  incrementSaver->releaseFields());
459 
462  bool lSanityCheckPassed = true;
463  std::vector<eckit::LocalConfiguration> confs;
464  if (late4DstateConf.has("incrementSanityChecks"))
465  late4DstateConf.get("incrementSanityChecks", confs);
466 
467  for (auto const &config : confs) {
468  const std::vector<std::string> varname = config.getStringVector("variables");
469  const double zmin = config.getDouble("min");
470  const double zmax = config.getDouble("max");
471  Variables_ var(config);
472  for (auto const &incr : increments) {
473  const double zincmin = incr->min(var);
474  const double zincmax = incr->max(var);
475  Log::info() << "IncrementalAssimilation::late4Dstart " << std::endl
476  << " Sanity Test: date: " << incr->validTime()
477  << ", variable: " << varname[0]
478  << ", bounds: (" << zmin << ", " << zmax << ")"
479  << ", min = " << zincmin
480  << ", max = " << zincmax << std::endl;
481  if (zincmin >= zmin && zincmax <= zmax) {
482  Log::info() << " Sanity Test: passed" << std::endl;
483  } else {
484  Log::info() << " Sanity Test: NOT passed" << std::endl;
485  lSanityCheckPassed = false;
486  // Don't skip final trajectory
487  this->setSkipFinalNonlinearTrajectory(false);
488  }
489  }
490  }
491 
493  if (lSanityCheckPassed) {
494  this->setSkipFinalNonlinearTrajectory(true);
495  } else {
496  this->setSkipFinalNonlinearTrajectory(false);
497  }
498  if (conf.has("skipFinalTraj")) {
499  this->setSkipFinalNonlinearTrajectory(conf.getBool("skipFinalTraj"));
500  }
501 
503  xx_->obsVar().write(conf);
504  xx_->modVar().write(conf);
505 
507  J_.resetLinearization();
508 
510  if (conf.getBool("write4Vfields") || not lSanityCheckPassed) {
511  xx_->state()[0].write(late4DstateConf);
512  util::LogbookWriter<bool> log("4VfieldsWrittenOut", true);
513  }
514 
516  util::LogbookWriter<bool> log("Late4D_check", lSanityCheckPassed);
517 
519  std::vector<std::shared_ptr<State_> > states;
520  if (stateSaver) states = stateSaver->releaseFields();
521 
524  if (lSanityCheckPassed) {
525  Log::info() << "IncrementalAssimilation::late4Dstart " << std::endl
526  << " All increment sanity tests PASSED " << std::endl
527  << " Skipping final non-linear trajectory" << std::endl
528  << conf << std::endl;
529 
531  if (late4DstateConf.has("writeToDisk")) {
532  eckit::LocalConfiguration writeToDiskConf(
533  late4DstateConf.getSubConfiguration("writeToDisk"));
534  ASSERT(states.size() == 0);
535  ASSERT(writeToDiskConf.has("restart_file"));
536  ASSERT(writeToDiskConf.has("steps"));
537  util::DateTime refTime(xx_->state()[0].validTime());
538  const std::string restartFileBase =
539  writeToDiskConf.getString("restart_file");
540  const std::string steps = writeToDiskConf.getString("steps");
541  eckit::Tokenizer tok("\t\n\v\f\r !\"#$%&'()*+,-./:;<=>?@[\\]^_`{|}~");
542  std::vector<std::string> tokens;
543  tok(steps, tokens);
544  for (const std::string &token : tokens) {
545  util::DateTime validTime(refTime);
546  const util::Duration step(token);
547  validTime += step;
548  std::string restartFile = restartFileBase + validTime.toString();
549  Log::info() << "IncrementalAssimilation::late4Dstart " << std::endl
550  << " -> read in FG state: " << restartFile << std::endl;
551  eckit::LocalConfiguration config(writeToDiskConf);
552  config.set("restart_file", restartFile);
553  config.set("step", step.toString());
554  std::shared_ptr<State_> state(new State_(xx_->state()[0]));
555  state->read(config);
556  states.push_back(state);
557  }
558  }
559  size_t nAnStates = states.size();
560  ASSERT(increments.size() == nAnStates);
561 
563  if (skipFinalNonlinearTrajectory()) xx_.reset();
564 
566  for (size_t jj = 0; jj < nAnStates; ++jj) {
568  *states[jj] += *increments[jj];
569 
571  states[jj]->write(late4DstateConf);
572 
574  increments[jj]->write(late4DstateConf);
575 
576  Log::info() << "IncrementalAssimilation::late4Dstart " << std::endl
577  << " State An valid at: " << states[jj]->validTime()
578  << " written out." << std::endl;
579 
581  states[jj].reset();
582  }
583  }
584 
585  Log::info() << "IncrementalAssimilation::late4Dstart finished" << std::endl;
586 }
