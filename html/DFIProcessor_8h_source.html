107 template<typename MODEL>
108 void DFIProcessor<MODEL>::doInitialize(const State_ &xx,
109  const util::DateTime &end, const util::Duration &tstep) {
110  const util::DateTime bgn(xx.validTime());
111  if (!linit_ && bgn <= end_ && end >= bgn_) {
112  weights_ = wfct_->setWeights(bgn_, end_, tstep);
113  linit_ = true;
114  }
115  endleg_ = end;
116  avg_ = new Accumulator<MODEL, State_, State_>(xx);
117  avg_->validTime() = time_;
118 }

120 // -----------------------------------------------------------------------------
121 template<typename MODEL>
122 void DFIProcessor<MODEL>::doProcessing(const State_ &xx) {
123  const util::DateTime now(xx.validTime());
124  if (now != endleg_ || now == end_) {
125  ASSERT(weights_.find(now) != weights_.end());
126 
127  const double zz = weights_[now];
128  const double zgfl = (now == time_) ? 1.0 : 0.0;
129  accumulate(zz, zgfl, xx);
130 
131  sum_ += zz;
132  Log::debug() << "DFIProcessor: time = " << now << ", weight = " << zz
133  << ", sum = " << sum_ << std::endl;
134  }
135 }

138 template<typename MODEL>
139 void DFIProcessor<MODEL>::accumulate(double zz, double zgfl, const State_ &rhs) {
140  eckit::LocalConfiguration cfg;
141  cfg.set("matrix", "variable_dependent_coeff");
142  cfg.set("analysed_variables_coefficient", zz);
143  cfg.set("other_variables_coefficient", zgfl);
144  Log::debug() << "DFI coefficient: " << cfg << std::endl;
145  std::unique_ptr < GenericMatrix_ > coeff(MatrixFactory_::create(cfg));
146  coeff->accumul(*avg_, rhs);
147 }
