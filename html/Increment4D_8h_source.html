143 Increment4D<MODEL>::Increment4D(const JbState_ &jb)
144  : incr4d_(), first_(0), last_(jb.nstates() - 1) {
145  for (int jsub = 0; jsub <= last_; ++jsub) {
146  std::unique_ptr<Increment_> incr(jb.newStateIncrement(jsub));
147  incr4d_.emplace(jsub, std::move(incr));
148  }

300 template <typename MODEL>
301 void Increment4D<MODEL>::shift_forward() {
302  using rit =
303  typename std::map<int, std::unique_ptr<Increment_>>::reverse_iterator;
304  for (rit jsub = incr4d_.rbegin(); jsub != incr4d_.rend(); ++jsub) {
305  const int isub = jsub->first;
306  if (isub > first_) this->get(isub) = this->get(isub - 1);
307  }
308  incr4d_.erase(first_);
309  Log::info() << "Increment4D::shift_forward erased " << first_ << std::endl;
310  first_ += 1;
311 }
