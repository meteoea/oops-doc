100 void GomL95::print(std::ostream & os) const {
101  double zmin = locval_[0];
102  double zmax = locval_[0];
103  double zavg = 0.0;
104  for (int jj = 0; jj < size_; ++jj) {
105  if (locval_[jj] < zmin) zmin = locval_[jj];
106  if (locval_[jj] > zmax) zmax = locval_[jj];
107  zavg += locval_[jj];
108  }
109  zavg /= size_;
110  os << size_ << "values, Min=" << zmin << ", Max=" << zmax << ", Average=" << zavg;
111 }
