56 template <typename MODEL>
57 void SaddlePointMatrix<MODEL>::multiply(const SPVector_ &x,
58  SPVector_ &z) const {
59  CtrlInc_ ww(j_.jb());
60 
61  // The three blocks below could be done in parallel
62 
63  // ADJ block
64  PostProcessorAD<Increment_> costad;
65  j_.zeroAD(ww);
66  z.dx(new CtrlInc_(j_.jb()));
67  JqTermAD_ *jqad = j_.jb().initializeAD(z.dx(), x.lambda().dx());
68  costad.enrollProcessor(jqad);
69  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
70  costad.enrollProcessor(j_.jterm(jj).setupAD(x.lambda().getv(jj), ww));
71  }
72  j_.runAD(ww, costad);
73  z.dx() += ww;
74 
75  // TLM block
76  PostProcessorTL<Increment_> costtl;
77  JqTermTL_ *jqtl = j_.jb().initializeTL();
78  costtl.enrollProcessor(jqtl);
79  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
80  costtl.enrollProcessor(j_.jterm(jj).setupTL(x.dx()));
81  }
82  CtrlInc_ mdx(x.dx());
83  j_.runTL(mdx, costtl);
84  z.lambda().clear();
85  z.lambda().dx(new CtrlInc_(j_.jb()));
86  j_.jb().finalizeTL(jqtl, x.dx(), z.lambda().dx());
87  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
88  z.lambda().append(costtl.releaseOutputFromTL(jj + 1));
89  }
90 
91  // Diagonal block
92  DualVector<MODEL> diag;
93  diag.dx(new CtrlInc_(j_.jb()));
94  j_.jb().multiplyB(x.lambda().dx(), diag.dx());
95  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
96  diag.append(j_.jterm(jj).multiplyCovar(*x.lambda().getv(jj)));
97  }
98 
99  // The three blocks above could be done in parallel
100 
101  z.lambda() += diag;
102 }
