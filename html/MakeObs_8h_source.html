42 template <typename MODEL>
43 class MakeObs : public Application {
44  using Departures_ = Departures<MODEL>;
45  using Geometry_ = Geometry<MODEL>;
46  using Model_ = Model<MODEL>;
47  using ModelAux_ = ModelAuxControl<MODEL>;
48  using ObsAuxCtrls_ = ObsAuxControls<MODEL>;
49  using Observations_ = Observations<MODEL>;
50  using ObsSpaces_ = ObservationSpaces<MODEL>;
51  using ObsOperators_ = ObsOperators<MODEL>;
52  using State_ = State<MODEL>;
53 
54  public:
55  // -----------------------------------------------------------------------------
56  MakeObs() { instantiateObsErrorFactory<MODEL>(); }
57  // -----------------------------------------------------------------------------
58  virtual ~MakeObs() {}
59  // -----------------------------------------------------------------------------
60  int execute(const eckit::Configuration& fullConfig) const {
61  // Setup observation window
62  const eckit::LocalConfiguration windowConf(fullConfig,
63  "Assimilation Window");
64  const util::DateTime bgn(windowConf.getString("Begin"));
65  const util::DateTime end(windowConf.getString("End"));
66  const util::Duration fclen(end - bgn);
67  Log::info() << "Observation window is:" << windowConf << std::endl;
68 
69  // Setup resolution
70  const eckit::LocalConfiguration geomConfig(fullConfig, "Geometry");
71  const Geometry_ geom(geomConfig);
72 
73  // Setup Model
74  const eckit::LocalConfiguration modelConfig(fullConfig, "Model");
75  const Model_ model(geom, modelConfig);
76 
77  // Setup initial "true" state
78  const eckit::LocalConfiguration initialConfig(fullConfig,
79  "Initial Condition");
80  Log::info() << "Initial configuration is:" << initialConfig << std::endl;
81  State_ xx(geom, model, initialConfig);
82  Log::test() << "Initial state: " << xx.norm() << std::endl;
83 
84  // Setup augmented state
85  ModelAux_ moderr(geom, model, initialConfig);
86 
87  // Setup forecast outputs
88  PostProcessor<State_> post;
89 
90  eckit::LocalConfiguration prtConf;
91  fullConfig.get("prints", prtConf);
92  post.enrollProcessor(new StateInfo<State_>("fc", prtConf));
93 
94  // Setup observations
95  const eckit::LocalConfiguration obsconf(fullConfig, "Observations");
96  Log::info() << "Observation configuration is:" << obsconf << std::endl;
97  ObsSpaces_ obspace(obsconf, geom, bgn, end);
98  ObsOperators_ hop(obspace);
99 
100  // Setup observation bias
101  ObsAuxCtrls_ ybias(obspace, obsconf);
102 
103  std::shared_ptr<Observer<MODEL, State_> > pobs(
104  new Observer<MODEL, State_>(obspace, hop, ybias));
105  post.enrollProcessor(pobs);
106 
107  // Run forecast and generate observations
108  model.forecast(xx, moderr, fclen, post);
109  Log::test() << "Final state: " << xx.norm() << std::endl;
110  Log::info() << "MakeObs: Finished observation generation." << std::endl;
111 
112  std::unique_ptr<Observations_> yobs(pobs->release());
113  Log::info() << "Generated observation: " << *yobs << std::endl;
114 
115  // Perturb observations
116  if (obsconf.has("obspert")) {
117  Departures_ ypert(obspace);
118  ObsErrors<MODEL> matR(obspace);
119  matR.randomize(ypert);
120  double opert = obsconf.getDouble("obspert");
121  ypert *= opert;
122  *yobs += ypert;
123  Log::info() << "Perturbed observation: " << *yobs << std::endl;
124  }
125 
126  // Save observations
127  for (std::size_t jj = 0; jj < yobs->size(); ++jj) {
128  Log::test() << "Generated observation: " << (*yobs)[jj] << std::endl;
129  }
130  yobs->save(obsconf);
131 
132  return 0;
133  }
134  // -----------------------------------------------------------------------------
135  private:
136  std::string appname() const { return "oops::MakeObs<" + MODEL::name() + ">"; }
137  // -----------------------------------------------------------------------------
138 };
