93 template <typename MODEL>
94 ControlIncrement<MODEL> *SaddlePointMinimizer<MODEL>::doMinimize(
95  const eckit::Configuration &config) {
96  int ninner = config.getInt("ninner");
97  int gnreduc = config.getDouble("gradient_norm_reduction");
98 
99  // if (gradJb_ == 0) gradJb_.reset(new CtrlInc_(J_.jb()));
100 
101  Log::info() << "SaddlePointMinimizer: max iter = " << ninner
102  << ", requested norm reduction = " << gnreduc << std::endl;
103 
104  // Define saddle-point control vectors
105  Multipliers_ *pdxx = new Multipliers_();
106  pdxx->dx(new CtrlInc_(J_.jb()));
107  for (unsigned jj = 0; jj < J_.nterms(); ++jj) {
108  pdxx->append(J_.jterm(jj).newDualVector());
109  }
110  CtrlInc_ *tmp1 = new CtrlInc_(J_.jb());
111  SaddlePointVector<MODEL> spdx(tmp1, pdxx);
112 
113  // Compute RHS
114  Multipliers_ *pdfg = new Multipliers_();
115  CtrlInc_ *tmp3 = new CtrlInc_(J_.jb().getFirstGuessIncrement());
116  *tmp3 *= -1.0;
117  pdfg->dx(tmp3);
118  for (unsigned jj = 0; jj < J_.nterms(); ++jj) {
119  std::unique_ptr<GeneralizedDepartures> ww(J_.jterm(jj).newGradientFG());
120  pdfg->append(J_.jterm(jj).multiplyCovar(*ww));
121  }
122  CtrlInc_ *tmp2 = new CtrlInc_(J_.jb());
123  SaddlePointVector<MODEL> rhs(tmp2, pdfg);
124 
125  // Define the matrices
126  SaddlePointMatrix<MODEL> A(J_);
127 
128  // Inexact constraint preconditioner
129  if (precond_type_ == "identity") {
130  precond_order_ = 0;
131  } else if (precond_type_ == "Midentity") {
132  precond_order_ = 1; // should be nsub
133  idmodel_ = true;
134  } else if (precond_type_ == "NeumannApprox") {
135  idmodel_ = false;
136  } else {
137  Log::error() << "SaddlePointMinimizer: wrong choice of preconditioner: "
138  << precond_type_ << std::endl;
139  }
140  SaddlePointPrecondMatrix<MODEL> Pinv(J_, precond_order_, idmodel_);
141 
142  // Initialize the limited memory preconditioner
143  // Eigen if (!Pinv_.get()) {
144  // Eigen Pinv_.reset(new LMP_(J_));
145  // Eigen }
146 
147  // Update the preconditioner
148  // Eigen Pinv_->setup(xyVEC_, pqVEC_);
149 
150  // Solve the linear system
151  double reduc = 1.0;
152  if (solver_ == "GMRESR") {
153  reduc = GMRESR(spdx, rhs, A, Pinv, ninner, gnreduc);
154  } else if (solver_ == "FullGMRES") {
155  reduc = FullGMRES(spdx, rhs, A, Pinv, ninner, gnreduc, pqVEC_, xyVEC_);
156  } else {
157  Log::error() << "SaddlePointMinimizer: wrong choice of solver: " << solver_
158  << std::endl;
159  }
160 
161  CtrlInc_ *dx = new CtrlInc_(spdx.dx());
162 
163  Log::test() << "SaddlePointMinimizer: reduction in residual norm = " << reduc
164  << std::endl;
165  Log::info() << "SaddlePointMinimizer output" << *dx << std::endl;
166 
167  // *gradJb_ = dx.lambda().dx();
168  // *gradJb_ *= -1.0;
169 
170  return dx;
171 }
