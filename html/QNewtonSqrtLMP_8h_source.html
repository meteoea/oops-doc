26 /* The QUASI-NEWTON Limited Memory preconditioner matrix \f$ C \f$
27  * for the system matrix \f$ AB = (I + Ht Rinv H B) \f$
28  *
29  * It is defined as
30  * \f$ C_(k+1) = (I - rho_k rr_k q_k^T) C_k (I - rho_k q_k p_k^T) + rho_k
31  * rr_k p_k^T\f$
32  *
33  * For details, we refer to S. Gratton, A. Sartenaer and J. Tshimanga,
34  * SIAM J. Optim.,21(3),912â€“935,2011 and S. Gurol, PhD Manuscript, 2013.
35  *
36  * The solvers represent matrices as objects that implement a "multiply"
37  * method. This class defines objects that apply the preconditioner matrix \f$
38  * P \f$.
39  */

98  if (precond.has("maxpairs")) maxpairs_ = precond.getInt("maxpairs");
99  if (maxpairs_ > 0) {

102  useoldpairs_ = (old == "on" || old == "true");
103  if (useoldpairs_ && precond.has("maxnewpairs")) {
104  maxnewpairs_ = precond.getInt("maxnewpairs");
105  maxnewpairs_ = std::min(maxpairs_, maxnewpairs_);

115 template <typename MODEL>
116 void QNewtonSqrtLMP<MODEL>::push(const CtrlVec_ &p, const CtrlVec_ &ph,
117  const CtrlVec_ &ap, const double &rho) {
118  if (active_) {
119  ASSERT(savedP_.size() <= maxnewpairs_);
120  if (maxnewpairs_ > 0 && update_ < maxouter_) {
121  if (savedP_.size() == maxnewpairs_) {
122  savedP_.erase(savedP_.begin());
123  savedR_.erase(savedR_.begin());
124  savedAP_.erase(savedAP_.begin());
125  savedrhos_.erase(savedrhos_.begin());
126  }
127  savedP_.push_back(p);
128  savedR_.push_back(ph);
129  savedAP_.push_back(ap);
130  savedrhos_.push_back(1.0 / rho);
131  }
132  }
133 }

137 template <typename MODEL>
138 void QNewtonSqrtLMP<MODEL>::update() {
139  if (active_) {
140  Log::debug() << "QNewtonLMP size saved Ph = " << savedR_.size()
141  << std::endl;
142  Log::debug() << "QNewtonLMP size saved P = " << savedP_.size()
143  << std::endl;
144  Log::debug() << "QNewtonLMP size saved AP = " << savedAP_.size()
145  << std::endl;
146  Log::debug() << "QNewtonLMP size saved rhos = " << savedrhos_.size()
147  << std::endl;
148 
149  const unsigned nvec = savedR_.size();
150  ASSERT(savedP_.size() == nvec);
151  ASSERT(savedAP_.size() == nvec);
152  ASSERT(savedrhos_.size() == nvec);
153 
154  if (!useoldpairs_ || nvec >= maxpairs_) {
155  R_.clear();
156  P_.clear();
157  AP_.clear();
158  rhos_.clear();
159  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
160  usedpairIndx_[kiter] = 0;
161  }
162  }
163 
164  if (nvec > 0 && update_ < maxouter_) {
165  Log::info() << "QNewtonLMP: update " << update_
166  << ", max = " << maxouter_ - 1 << std::endl;
167  unsigned newpairs = std::min(nvec, maxpairs_);
168  unsigned oldpairs = P_.size();
169  unsigned rmpairs = 0;
170  // First remove pairs we no longer need.
171  if (oldpairs + newpairs > maxpairs_) {
172  rmpairs = oldpairs + newpairs - maxpairs_;
173  for (unsigned jv = 0; jv < rmpairs; ++jv) {
174  R_.erase(R_.begin());
175  P_.erase(P_.begin());
176  AP_.erase(AP_.begin());
177  rhos_.erase(rhos_.begin());
178  }
179  oldpairs -= rmpairs;
180  // Keep information on how many pairs are used at each minimization
181  // loop
182  unsigned removed = rmpairs;
183  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
184  unsigned rmiter = std::min(usedpairIndx_[kiter], removed);
185  usedpairIndx_[kiter] -= rmiter;
186  removed -= rmiter;
187  }
188  ASSERT(removed == 0);
189  }
190  ASSERT(P_.size() == oldpairs);
191  ASSERT(oldpairs + newpairs <= maxpairs_);
192 
193  // Add the new pairs.
194  for (unsigned jv = nvec - newpairs; jv < nvec; ++jv) {
195  P_.push_back(std::shared_ptr<CtrlVec_>(new CtrlVec_(savedP_[jv])));
196  R_.push_back(std::shared_ptr<CtrlVec_>(new CtrlVec_(savedR_[jv])));
197  AP_.push_back(std::shared_ptr<CtrlVec_>(new CtrlVec_(savedAP_[jv])));
198  rhos_.push_back(savedrhos_[jv]);
199  }
200  ASSERT(P_.size() == oldpairs + newpairs);
201 
202  Log::info() << "Number of inner iterations : " << nvec << std::endl;
203  Log::info() << "Number of maximum pairs : " << maxpairs_
204  << std::endl;
205  Log::info() << "Number of used total pairs : "
206  << oldpairs + newpairs << std::endl;
207  Log::info() << "Number of new pairs : " << newpairs
208  << std::endl;
209  Log::info() << "Number of removed old pairs : " << rmpairs
210  << std::endl;
211  for (unsigned kiter = 0; kiter < usedpairIndx_.size(); ++kiter) {
212  Log::info() << "Number of used pairs from outer loop " << kiter + 1
213  << " : " << usedpairIndx_[kiter];
214  }
215  }
216 
217  ++update_;
218  savedP_.clear();
219  savedR_.clear();
220  savedAP_.clear();
221  savedrhos_.clear();
222  }
223 }

240 template <typename MODEL>
241 void QNewtonSqrtLMP<MODEL>::inverseMultiplySqrt(const CtrlVec_ &a,
242  CtrlVec_ &b) const {
243  b = a;
244  const unsigned nvec = P_.size();
245  if (nvec != 0) {
246  for (unsigned iiter = 0; iiter < nvec; ++iiter) {
247  double factor = sqrt(rhos_[iiter]) * dot_product(*R_[iiter], b) /
248  sqrt(dot_product(*R_[iiter], *R_[iiter]));
249  factor += rhos_[iiter] * dot_product(*AP_[iiter], b);
250  b.axpy(-factor, *P_[iiter]);
251  }
252  }
253 }

257 template <typename MODEL>
258 void QNewtonSqrtLMP<MODEL>::inverseMultiply(const CtrlVec_ &a,
259  CtrlVec_ &b) const {
260  b = a;
261  const unsigned nvec = P_.size();
262  if (nvec != 0) {
263  std::vector<double> etas(nvec, 0);
264  // etas.clear();
265  for (int iiter = nvec - 1; iiter >= 0; iiter--) {
266  double eta = dot_product(b, *P_[iiter]);
267  eta *= rhos_[iiter];
268  etas[iiter] = eta;
269  b.axpy(-eta, *AP_[iiter]);
270  }
271  b *= dot_product(*AP_[nvec - 1], *P_[nvec - 1]) /
272  dot_product(*AP_[nvec - 1], *AP_[nvec - 1]);
273  for (unsigned iiter = 0; iiter < nvec; ++iiter) {
274  double sigma = dot_product(b, *AP_[iiter]);
275  sigma *= rhos_[iiter];
276  sigma -= etas[iiter];
277  b.axpy(-sigma, *P_[iiter]);
278  }
279  }
280 }
