113 template <typename MODEL>
114 double DRPCGMinimizer<MODEL>::solve(CtrlInc_ &dx, CtrlInc_ &dxh, CtrlInc_ &rr,
115  const Bmat_ &B, const HtRinvH_ &HtRinvH,
116  const double costJ0Jb,
117  const double costJ0JoJc, const int maxiter,
118  const double tolerance) {
119  // dx increment
120 
121  CtrlInc_ qq(dxh, false);
122  CtrlInc_ pp(dxh, false);
123  CtrlInc_ hh(dxh, false);
124  CtrlInc_ zz(dxh, false);
125  CtrlInc_ pr(dxh, false);
126  CtrlInc_ r0(dxh, false);
127  CtrlInc_ ww(dxh, false);
128  CtrlInc_ dx0(dx);
129  CtrlInc_ dxh0(dxh);
130 
131  // vectors for re-orthogonalization
132  std::vector<CtrlInc_> vvecs;
133  std::vector<CtrlInc_> zvecs;
134  std::vector<double> scals;
135 
136  // r_{0} = b - A x_{0}
137  HtRinvH.multiply(dx, zz);
138  zz += dxh;
139  rr -= zz;
140 
141  // save the initial residual
142  r0 = rr;
143 
144  // J0
145  const double costJ0 = costJ0Jb + costJ0JoJc;
146 
147  // zero out initial guess
148  dx.zero();
149  dxh.zero();
150 
151  // r_{0}^T r_{0}
152  Log::info() << "normr0 " << dot_product(rr, rr) << std::endl;
153 
154  // z_{0} = B LMP r_{0}
155  lmp_.multiply(rr, pr);
156  B.multiply(pr, zz);
157  // p_{0} = z_{0}
158  pp = zz;
159  // h_{0} = LMP r_{0}
160  hh = pr;
161 
162  // r_{0}^T z_{0}
163  double dotRr0 = dot_product(rr, zz);
164  double normReduction = 1.0;
165  double rdots = dotRr0;
166  double rdots_old = 0.0;
167 
168  vvecs.push_back(rr);
169  zvecs.push_back(zz);
170  scals.push_back(1.0 / dotRr0);
171 
172  Log::info() << std::endl;
173  for (int jiter = 0; jiter < maxiter; ++jiter) {
174  Log::info() << " DRPCG Starting Iteration " << jiter + 1 << std::endl;
175 
176  if (jiter > 0) {
177  // beta_{i} = r_{i+1}^T z_{i+1} / r_{i}^T z_{i}
178  double beta = rdots / rdots_old;
179  Log::debug() << "DRPCG beta = " << beta << std::endl;
180 
181  // p_{i+1} = z_{i+1} + beta*p_{i}
182  pp *= beta;
183  pp += zz;
184 
185  // h_{i+1} = LMP r_{i+1} + beta*h_{i}
186  hh *= beta;
187  hh += pr;
188  }
189 
190  // q_{i} = h_{i} + H^T R^{-1} H p_{i}
191  HtRinvH.multiply(pp, qq);
192  qq += hh;
193 
194  // alpha_{i} = r_{i}^T z_{i} / q_{i}^T p_{i}
195  double rho = dot_product(pp, qq);
196  double alpha = rdots / rho;
197 
198  // dx_{i+1} = dx_{i} + alpha * p_{i}
199  dx.axpy(alpha, pp);
200  // dxh_{i+1} = dxh_{i} + alpha * h_{i} ! for diagnosing Jb
201  dxh.axpy(alpha, hh);
202  // r_{i+1} = r_{i} - alpha * q_{i}
203  rr.axpy(-alpha, qq);
204 
205  // Compute the quadratic cost function
206  // J[dx_{i}] = J[0] - 0.5 dx_{i}^T r_{0}
207  double costJ = costJ0 - 0.5 * dot_product(dx, r0);
208  // Jb[dx_{i}] = 0.5 dx_{i}^T f_{i}
209  double costJb = costJ0Jb + 0.5 * dot_product(dx, dxh);
210  // Jo[dx_{i}] + Jc[dx_{i}] = J[dx_{i}] - Jb[dx_{i}]
211  double costJoJc = costJ - costJb;
212 
213  // Re-orthogonalization
214  for (int jj = 0; jj < jiter; ++jj) {
215  double proj = scals[jj] * dot_product(rr, zvecs[jj]);
216  rr.axpy(-proj, vvecs[jj]);
217  }
218 
219  // z_{i+1} = B LMP r_{i+1}
220  lmp_.multiply(rr, pr);
221  B.multiply(pr, zz);
222 
223  // r_{i}^T z_{i}
224  rdots_old = rdots;
225  // r_{i+1}^T z_{i+1}
226  rdots = dot_product(rr, zz);
227 
228  Log::info() << "rdots " << rdots << " iteration " << jiter
229  << std::endl;
230 
231  // r_{i+1}^T z_{i+1} / r_{0}^T z_{0}
232  normReduction = sqrt(rdots / dotRr0);
233 
234  Log::info() << "DRPCG end of iteration " << jiter + 1 << std::endl
235  << " Norm reduction (" << std::setw(2) << jiter + 1
236  << ") = " << util::full_precision(normReduction) << std::endl
237  << " Quadratic cost function: J (" << std::setw(2)
238  << jiter + 1 << ") = " << util::full_precision(costJ)
239  << std::endl
240  << " Quadratic cost function: Jb (" << std::setw(2)
241  << jiter + 1 << ") = " << util::full_precision(costJb)
242  << std::endl
243  << " Quadratic cost function: JoJc(" << std::setw(2)
244  << jiter + 1 << ") = " << util::full_precision(costJoJc)
245  << std::endl
246  << std::endl;
247 
248  // Save the pairs for preconditioning
249  lmp_.push(pp, hh, qq, rho);
250 
251  if (normReduction < tolerance) {
252  Log::info() << "DRPCG: Achieved required reduction in residual norm."
253  << std::endl;
254  break;
255  }
256 
257  vvecs.push_back(rr);
258  zvecs.push_back(zz);
259  scals.push_back(1.0 / rdots);
260  }
261 
262  // Generate the (second-level) Limited Memory Preconditioner
263  lmp_.update(B);
264 
265  // Add increment to initial guess
266  dx += dx0;
267  dxh += dxh0;
268 
269  return normReduction;
270 }
