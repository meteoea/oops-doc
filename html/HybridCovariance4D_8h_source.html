114 template<typename MODEL>
115 void HybridCovariance4D<MODEL>::advectedLinearize(const State4D_ &xb,
116  const Geometry_ &resol, const eckit::Configuration &config) {
117  const eckit::LocalConfiguration jbConf(config, "Jb");
118  const eckit::LocalConfiguration covConf(jbConf, "Covariance");
119  const eckit::LocalConfiguration ensConf(covConf, "ensemble_covariance");
120  const eckit::LocalConfiguration statConf(covConf, "static_covariance");
121 
122  if (covConf.has("override_ensemble_weight")) {
123  ensemble_weight_.reset(
124  MatrixFactory_::create(
125  eckit::LocalConfiguration(covConf, "override_ensemble_weight")));
126  }
127  if (covConf.has("override_static_weight")) {
128  static_weight_.reset(
129  MatrixFactory_::create(
130  eckit::LocalConfiguration(covConf, "override_static_weight")));
131  }
132 
133  static_->linearize(xb, resol, statConf);
134  ensemble_->linearize(xb, resol, ensConf);
135 
136  Log::trace() << "HybridCovariance4D linearized.";
137 }

141 template <typename MODEL>
142 void HybridCovariance4D<MODEL>::advectedMultiply(const Increment4D_ &dxi,
143  Increment4D_ &dxo) const {
144  Increment4D_ tmp(dxo);
145  Increment4D_ dxistat(dxi);
146  static_weight_->multiply(dxistat);
147  static_->multiply(dxistat, dxo);
148  static_weight_->multiply(dxo);
149 
150  Increment4D_ dxiens(dxi);
151  ensemble_weight_->multiply(dxiens);
152  ensemble_->multiply(dxiens, tmp);
153  ensemble_weight_->multiply(tmp);
154 
155  dxo += tmp;
156 }

169 template<typename MODEL>
170 void HybridCovariance4D<MODEL>::advectedMultiplySqrtTrans(
171  const Increment4D_ &dxi, IncrCtlVec_ &dvo) const {
172  { // Separate blocks improve memory handling
173  Increment4D_ dxistat(dxi);
174  static_weight_->multiply(dxistat);
175  static_->multiplySqrtTrans(dxistat, dvo);
176  }
177  {
178  Increment4D_ dxiens(dxi);
179  ensemble_weight_->multiply(dxiens);
180  ensemble_->multiplySqrtTrans(dxiens, dvo);
181  }
182 }
