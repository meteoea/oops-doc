159 template <typename MODEL>
160 void TrajectorySaverFP<MODEL>::recordTrajectory() {
161  util::LogbookWriter<std::string> log("MF4DVarTrajectory", "onlyHorizInterpol");
162 
164  for (size_t jsub = 0; jsub < subwin_; ++jsub) {
166  subtlm_.emplace_back(new LinearModel_(resol_, tlConf_));
167  ASSERT(subtlm_[jsub] != nullptr);
168 
170  xlr_.forceWith(*forcing_[jsub][0], varsForced_);
171  for(std::shared_ptr<PostBase_> jp: processors_)
172  jp->initialize(xlr_, end_[jsub], step_[jsub]);
173  subtlm_[0]->setTrajectory(xlr_, lrBias_);
175  forcing_[jsub][0].reset();
176 
178  for (size_t jj = 1; jj < forcing_[jsub].size(); ++jj) {
180  xlr_.forceWith(*forcing_[jsub][jj], varsForced_);
181 
183  for(std::shared_ptr<PostBase_> jp: processors_)
184  jp->process(xlr_);
185 
187  subtlm_[jsub]->setTrajectory(xlr_, lrBias_);
189  forcing_[jsub][jj].reset();
190  }
192  for(std::shared_ptr<PostBase_> jp: processors_)
193  jp->finalize(xlr_);
194  }
195  tlm_.swap(subtlm_);
196 }
