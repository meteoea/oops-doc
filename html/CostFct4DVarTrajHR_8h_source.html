115 template <typename MODEL>
116 double CostFct4DVarTrajHR<MODEL>::evaluate(const CtrlVar_ &fguess,
117  const eckit::Configuration &config,
118  PostProcessor<State_> post) const {
119  Log::trace() << "CostFct4DVarTrajHR::evaluate start" << std::endl;
120 
122  PostProcessor<State_> pp(post);
123  std::shared_ptr<WeightedMean_> pdfi;
124  if (config.has("dfi")) {
125  const eckit::LocalConfiguration dfiConf(config, "dfi");
126  const util::Duration dfispan(dfiConf.getString("filter_span"));
127  const util::DateTime dfitime(fguess.state()[0].validTime() + dfispan / 2);
128  pdfi.reset(new WeightedMean_(dfitime, dfispan, geom_, dfiConf));
129  pp.enrollProcessor(pdfi);
130  }
131 
133  double zzz = this->CostFunction<MODEL>::evaluate(fguess, config, pp);
134 
145  if (config.has("dfi")) {
146  const eckit::LocalConfiguration dfiConf(config, "dfi");
147  std::unique_ptr<State_> xdfi(pdfi->releaseMean());
148  (*xdfi) += (*biasDFI_);
149  Log::info() << "CostFunction: Write filtered analysis in the middle of the window.";
150  xdfi->write(eckit::LocalConfiguration(dfiConf, "output"));
151  }
152 
153  Log::trace() << "CostFct4DVarTrajHR::evaluate done" << std::endl;
154  return zzz;
155 }

159 template<typename MODEL>
160 double CostFct4DVarTrajHR<MODEL>::linearize(const CtrlVar_ & fguess,
161  const eckit::Configuration & innerConf,
162  PostProcessor<State_> post) {
209  PostProcessor<State_> no_post;
211  const eckit::LocalConfiguration resConf(innerConf, "resolution");
212  const eckit::LocalConfiguration tlmConf(innerConf, "linearmodel");
213  std::unique_ptr<eckit::LocalConfiguration> dfiConf;
214  const Geometry_ lowres(resConf);
215  std::shared_ptr<WeightedDiff_> pdfi;
216  std::unique_ptr<WeightingFct> wfct;
217 
219  const eckit::LocalConfiguration lowresModelConfig(innerConf, "low_resolution_model");
220  trajectoryModel_.reset(new Model_(lowres, lowresModelConfig));
221  CtrlVar_ fglr(lowres, *trajectoryModel_, fguess);
222  fglr_.reset(new State_(fglr.state()[0]));
223  malloc_trim(0);
224 
226  PostProcessor<State_> pp(post);
227  pp.enrollProcessor(this->getJterm(0).initialize(fguess, innerConf));
228 
229  std::shared_ptr< TrajectorySaver_ >
230  trajectorySaver(new TrajectorySaver_(fglr.state()[0], fglr.modVar(),
231  lowres, this->getTLMs(), tlmConf));
232  pp.enrollProcessor(trajectorySaver);
233 
234  if (innerConf.has("screening.dfi")) {
235  dfiConf.reset(new eckit::LocalConfiguration(innerConf, "screening.dfi"));
236  const util::Duration dfispan(dfiConf->getString("filter_span"));
237  const util::DateTime dfitime(fguess.state()[0].validTime() + dfispan / 2);
238  wfct.reset(new DolphChebyshev(*dfiConf));
239  pdfi.reset(new WeightedDiff_(dfitime, dfispan, geom_, *dfiConf, util::Duration(0), *wfct));
240  pp.enrollProcessor(pdfi);
241  if (dfiConf->has("output")) {
242  const eckit::LocalConfiguration outConfig(*dfiConf, "output");
243  pp.enrollProcessor(new StateWriter<State_>(fguess.state()[0].validTime(), outConfig));
244  }
245  }
246 
251  CtrlVar_ fghr(fguess);
252  this->runNL(fghr, pp);
253 
254  costJoJc_ = this->getJterm(0).finalize(innerConf);
255 
262  if (innerConf.has("screening.dfi")) {
263  biasDFI_.reset(pdfi->releaseDiff());
264  (*biasDFI_) *= -1.0;
265  }
266 
267  //TODO:
268  // 5. Reset the observation space component of the first guess
269  //ObsAuxControls_ &obsvarfg = const_cast<ObsAuxControls_ &>(fglr.obsVar());
270  //obsvarfg.reset(*obsSpaces_, "fg", innerConf);
271 
278  std::shared_ptr<JqTerm_> jq(this->getJb().initializeTraj(fglr, lowres, innerConf));
279  trajectorySaver->enrollProcessor(jq);
280  for (unsigned iterm=0; iterm < this->nterms(); ++iterm) {
281  trajectorySaver->enrollProcessor(this->getJterm(iterm).initializeTraj(fglr, lowres, innerConf));
282  }
283  std::shared_ptr<Advector_ > advector = Advector_::get();
284  if (advector) trajectorySaver->enrollProcessor(advector);
285 
286  trajectorySaver->recordTrajectory();
287 
289  costJb_ = this->getJb().finalizeTraj(jq.get(), innerConf);
290  double costJoJcTraj = 0.0;
291  for (unsigned iterm=0; iterm < this->nterms(); ++iterm) {
292  costJoJcTraj += this->getJterm(iterm).finalizeTraj(innerConf);
293  }
294 
296  double costJ = costJb_ + costJoJc_;
297  Log::test() << "CostFunction: Nonlinear J = " << costJ << std::endl;
298  return costJ;
299 }

303 template<typename MODEL>
304 void CostFct4DVarTrajHR<MODEL>::addIncr(CtrlVar_ & xfghr, const CtrlInc_ & dxlr,
305  PostProcessor<Increment_> &) const {
306  ASSERT(xfghr.state().checkStatesNumber(1));
307 
309  State_ fghr(geom_, *trajectoryModel_, *fglr_);
310 
312  (*fglr_) += dxlr.state()[0];
313  State_ xahr(geom_, *trajectoryModel_, *fglr_);
314 
316  Increment_ dxhr(geom_, dxlr.state()[0]);
317  dxhr.diff(xahr, fghr);
318 
320  xfghr.state()[0] += dxhr;
321 }
