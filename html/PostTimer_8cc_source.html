73 void PostTimer::initialize(const util::DateTime &bgn, const util::DateTime &end,
74  const util::Duration &) {
75  util::DateTime start(bgn);
76  if (start_ != 0) {
77  start = *start_;
78  }
79  bgn_ = start;
80  util::DateTime finish(end);
81  if (finish_ != 0) {
82  finish = *finish_;
83  }
84  end_ = finish;
85 
86  // User specified configuration
87  Log::debug() << "PProc configuration is:" << conf_ << std::endl;
88  // User specified frequency and start
89  if (conf_.has("frequency"))
90  freq_ = util::Duration(conf_.getString("frequency"));
91  if (conf_.has("first")) {
92  const util::Duration first(conf_.getString("first"));
93  bgn_ += first;
94  }
95  // User specified steps
96  if (conf_.has("steps")) {
97  const std::string steps = conf_.getString("steps");
98  // this separator list is to maintain compatibility with previous
99  // boost::tokenizer default values
100  eckit::Tokenizer tok("\t\n\v\f\r !\"#$%&'()*+,-./:;<=>?@[\\]^_`{|}~");
101  std::vector<std::string> tokens;
102  tok(steps, tokens);
103  for (const std::string &token : tokens) {
104  const util::Duration step(token);
105  const util::DateTime tt = start + step;
106  if (bgn <= tt && tt <= end) pptimes_.push_back(tt);
107  }
108  }
109 }

111 bool PostTimer::itIsTime(const util::DateTime &now) {
112  bool doit = false;
113  if (now >= bgn_ && now <= end_) {
114  doit = (freq_.toSeconds() == 0 && pptimes_.empty());
115  if (!doit && freq_.toSeconds() > 0) {
116  const util::Duration dt = now - bgn_;
117  doit = (dt >= util::Duration(0) && dt % freq_ == 0);
118  }
119  if (!doit && !pptimes_.empty()) {
120  std::vector<util::DateTime>::iterator it;
121  it = find(pptimes_.begin(), pptimes_.end(), now);
122  doit = (it != pptimes_.end());
123  }
124  }
125  Log::trace() << "In PostTimer:itIsTime, time = " << now << ", doit = " << doit
126  << std::endl;
127  return doit;
128 }
