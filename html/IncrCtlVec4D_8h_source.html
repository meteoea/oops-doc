142  : incrCtlVec4d_(), first_(0), last_(jb.nchistates() - 1) {
143  for (int jsub = 0; jsub <= last_; ++jsub) {
144  std::unique_ptr<IncrCtlVec_> incrCtlVec(jb.newIncrCtlVec(jsub));
