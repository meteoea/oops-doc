116 template <typename MODEL>
117 double DRPLanczosMinimizer<MODEL>::solve(
118  CtrlInc_ &dx, CtrlInc_ &dxh, CtrlInc_ &rr, const Bmat_ &B,
119  const HtRinvH_ &HtRinvH, const double costJ0Jb, const double costJ0JoJc,
120  const int maxiter, const double tolerance) {
121  // dx increment
122  // dxh B^{-1} dx
123  // rr (sum B^{-1} dx_i^{b} +) G^T H^{-1} d
124 
125  CtrlInc_ zz(dxh);
126  CtrlInc_ pr(dxh);
127  CtrlInc_ vv(rr);
128  CtrlInc_ mm(rr);
129 
130  std::vector<double> ss;
131  std::vector<double> dd;
132 
133  // J0
134  const double costJ0 = costJ0Jb + costJ0JoJc;
135 
136  lmp_.update(vvecs_, hvecs_, zvecs_, alphas_, betas_);
137 
138  // z_{0} = B LMP r_{0}
139  lmp_.multiply(vv, pr);
140  B.multiplyB(pr, zz);
141 
142  // beta_{0} = sqrt( z_{0}^T r_{0} )
143  double beta = sqrt(dot_product(zz, vv));
144  const double beta0 = beta;
145 
146  // v_{1} = r_{0} / beta_{0}
147  vv *= 1 / beta;
148  // pr_{1} = LMP r_{0} / beta_{0}
149  pr *= 1 / beta;
150  // z_{1} = z_{0} / beta_{0}
151  zz *= 1 / beta;
152 
153  // hvecs[0] = pr_{1} --> required for solution
154  hvecs_.emplace_back(new CtrlInc_(pr, config_));
155  // zvecs[0] = z_{1} ---> for re-orthogonalization
156  zvecs_.emplace_back(new CtrlInc_(zz, config_));
157  // vvecs[0] = v_{1} ---> for re-orthogonalization
158  vvecs_.emplace_back(new CtrlInc_(vv, config_));
159 
160  double normReduction = 1.0;
161 
162  Log::info() << std::endl;
163  for (int jiter = 0; jiter < maxiter; ++jiter) {
164  Log::info() << "DRPLanczos Starting Iteration " << jiter + 1 << std::endl;
165 
166  // v_{i+1} = ( pr_{i} + H^T R^{-1} H z_{i} ) - beta * v_{i-1}
167  HtRinvH.multiply(zz, vv);
168  B.multiplyId(pr, mm);
169  vv += mm;
170  if (jiter > 0) vv.axpy(-beta, *vvecs_[jiter - 1]);
171 
172  // alpha_{i} = v_{i+1}^T z_{i}
173  double alpha = dot_product(zz, vv);
174 
175  // v_{i+1} = v_{i+1} - alpha_{i} v_{i}
176  vv.axpy(-alpha, *vvecs_[jiter]); // vv = vv - alpha * v_j
177 
178  // Re-orthogonalization
179  for (int jj = 0; jj < jiter; ++jj) {
180  double proj = dot_product(vv, *zvecs_[jj]);
181  vv.axpy(-proj, *vvecs_[jj]);
182  }
183 
184  // z_{i+1} = B LMP v_{i+1}
185  lmp_.multiply(vv, pr);
186  B.multiplyB(pr, zz);
187 
188  // beta_{i+1} = sqrt( zz_{i+1}^t, vv_{i+1} )
189  beta = sqrt(dot_product(zz, vv));
190 
191  // v_{i+1} = v_{i+1} / beta_{i+1}
192  vv *= 1 / beta;
193  // pr_{i+1} = pr_{i+1} / beta_{i+1}
194  pr *= 1 / beta;
195  // z_{i+1} = z_{i+1} / beta_{i+1}
196  zz *= 1 / beta;
197 
198  // hvecs[i+1] =pr_{i+1}
199  hvecs_.emplace_back(new CtrlInc_(pr, config_));
200  // zvecs[i+1] = z_{i+1}
201  zvecs_.emplace_back(new CtrlInc_(zz, config_));
202  // vvecs[i+1] = v_{i+1}
203  vvecs_.emplace_back(new CtrlInc_(vv, config_));
204 
205  alphas_.push_back(alpha);
206 
207  if (jiter == 0) {
208  ss.push_back(beta0 / alpha);
209  dd.push_back(beta0);
210  } else {
211  // Solve the tridiagonal system T_{i} s_{i} = beta0 * e_1
212  dd.push_back(beta0 * dot_product(*zvecs_[0], vv));
213  TriDiagSolve(alphas_, betas_, dd, ss);
214  }
215 
216  betas_.push_back(beta);
217 
218  // Compute the quadratic cost function
219  // J[du_{i}] = J[0] - 0.5 s_{i}^T Z_{i}^T r_{0}
220  // Jb[du_{i}] = 0.5 s_{i}^T V_{i}^T Z_{i} s_{i}
221  double costJ = costJ0;
222  double costJb = costJ0Jb;
223  dx.zero();
224  for (int jj = 0; jj < jiter + 1; ++jj) {
225  dx.axpy(ss[jj], *zvecs_[jj]);
226  // costJ -= 0.5 * ss[jj] * dot_product(*zvecs_[jj], rr);
227  costJb += 0.5 * ss[jj] * dot_product(*vvecs_[jj], *zvecs_[jj]) * ss[jj];
228  }
229  costJ -= 0.5 * dot_product(dx, rr);
230  double costJoJc = costJ - costJb;
231 
232  // Gradient norm in precond metric --> sqrt(r'z) --> beta * s_{i}
233  double rznorm = beta * std::abs(ss[jiter]);
234  normReduction = rznorm / beta0;
235 
236  Log::info() << "DRPLanczos end of iteration " << jiter + 1 << std::endl
237  << " Norm reduction (" << std::setw(2) << jiter + 1
238  << ") = " << util::full_precision(normReduction) << std::endl
239  << " Quadratic cost function: J (" << std::setw(2)
240  << jiter + 1 << ") = " << util::full_precision(costJ)
241  << std::endl
242  << " Quadratic cost function: Jb (" << std::setw(2)
243  << jiter + 1 << ") = " << util::full_precision(costJb)
244  << std::endl
245  << " Quadratic cost function: JoJc(" << std::setw(2)
246  << jiter + 1 << ") = " << util::full_precision(costJoJc)
247  << std::endl
248  << std::endl;
249 
250  if (normReduction < tolerance) {
251  Log::info() << "DRPLanczos: Achieved required reduction in residual norm."
252  << std::endl;
253  break;
254  }
255  }
256 
257  // Calculate the solution (dxh = Binv dx)
258  dx.zero();
259  for (unsigned int jj = 0; jj < ss.size(); ++jj) {
260  dx.axpy(ss[jj], *zvecs_[jj]);
261  dxh.axpy(ss[jj], *hvecs_[jj]);
262  }
263 
264  return normReduction;
265 }
