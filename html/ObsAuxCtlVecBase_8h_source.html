138 template <typename MODEL>
139 ObsAuxCtlVecBase<MODEL> *ObsAuxCtlVecFactory<MODEL>::create(
140  const ObsAuxCovarianceBase<MODEL> &cov) {
141  const std::string id = cov.config().getString("type");
142  typename std::map<std::string, ObsAuxCtlVecFactory<MODEL> *>::iterator jloc =
143  getMakers().find(id);
144  if (jloc == getMakers().end()) {
145  Log::error() << id << " does not exist in ObsAuxCtlVecFactory."
146  << std::endl;
147  ABORT("Element does not exist in ObsAuxCtlVecFactory.");
148  }
149  ObsAuxCtlVecBase<MODEL> *ptr = jloc->second->make(cov);
150  return ptr;
151 }
