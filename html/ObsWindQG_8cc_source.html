50 void ObsWindQG::obsEquiv(const GomQG & gom, ObsVecQG & ovec,
51  const ObsAuxControlPtrMap_ & bias) const {
52  using icst_ = typename ObsAuxControlPtrMap_::const_iterator;
53  icst_ it = bias.find("ObsBiasQG");
54  if (it == bias.end()) {
55  std::vector<double> dummyBias(2, 0.0);
56  qg_wind_equiv_f90(gom.toFortran(), ovec.toFortran(), dummyBias[0]);
57  } else {
58  const ObsBias * pbias = dynamic_cast<const ObsBias*>(it->second.get());
59  ASSERT(pbias != nullptr);
60  qg_wind_equiv_f90(gom.toFortran(), ovec.toFortran(), pbias->wind());
