89 template <typename MODEL>
90 double RPLanczosMinimizer<MODEL>::solve(Dual_ &vv, double &vvp, Dual_ &rr,
91  const HBHt_ &HBHt, const Rinv_ &Rinv,
92  const int &maxiter,
93  const double &tolerance, Dual_ &dy,
94  const double &sigma) {
95  IdentityMatrix<Dual_> precond;
96 
97  Dual_ zz(vv);
98  Dual_ ww(vv);
99  Dual_ tt(vv);
100  Dual_ vold(vv);
101  Dual_ v(vv);
102 
103  // augmented part of the vectors
104  double rrp = 1.0;
105  double zzp;
106  double wwp;
107  double ttp = 0.0;
108  double voldp = 0.0;
109  double vp = 1.0;
110 
111  std::vector<Dual_> vVEC; // required for re-orthogonalization
112  std::vector<Dual_> tVEC; // required for re-orthogonalization
113  std::vector<Dual_> zVEC; // required for solution
114  std::vector<double> vpVEC;
115  std::vector<double> tpVEC;
116  std::vector<double> zpVEC;
117  std::vector<double> alphas;
118  std::vector<double> betas;
119  std::vector<double> y;
120  std::vector<double> d;
121 
122  // zzaug = Gaug rraug
123  precond.multiply(rr, zz);
124  zzp = rrp;
125 
126  // ttaug = HBHtaug zzaug
127  HBHt.multiply(zz, tt);
128  tt.axpy(zzp, dy);
129  ttp = dot_product(dy, zz) + sigma * zzp;
130 
131  double normReduction = 1.0;
132  double beta0 = sqrt(dot_product(rr, tt) + rrp * ttp);
133  double beta = 0.0;
134 
135  vold.zero();
136  v = rr;
137  vp = rrp;
138  v *= 1 / beta0;
139  vp *= 1 / beta0;
140  tt *= 1 / beta0;
141  zz *= 1 / beta0;
142  ttp *= 1 / beta0;
143  zzp *= 1 / beta0;
144 
145  vVEC.clear();
146  zVEC.clear();
147  tVEC.clear();
148  vpVEC.clear();
149  zpVEC.clear();
150  tpVEC.clear();
151 
152  betas.clear();
153  alphas.clear();
154 
155  vVEC.push_back(v);
156  zVEC.push_back(zz);
157  tVEC.push_back(tt);
158  vpVEC.push_back(vp);
159  zpVEC.push_back(zzp);
160  tpVEC.push_back(ttp);
161 
162  int jiter;
163  Log::info() << std::endl;
164  for (jiter = 0; jiter < maxiter; ++jiter) {
165  Log::info() << "RPLanczos Starting Iteration " << jiter + 1 << std::endl;
166 
167  // ww = (RinvHBHt + I) zz - beta * vold
168  Rinv.multiply(tt, ww);
169  ww += zz;
170  wwp = zzp;
171  ww.axpy(-beta, vold);
172  wwp -= beta * voldp;
173 
174  double alpha = dot_product(tt, ww) + ttp * wwp;
175 
176  ww.axpy(-alpha, v); // w = w - alpha * v
177  wwp -= alpha * vp;
178 
179  // Re-orthogonalization
180  for (int iiter = 0; iiter < jiter; ++iiter) {
181  double proj = dot_product(ww, tVEC[iiter]) + wwp * tpVEC[iiter];
182  ww.axpy(-proj, vVEC[iiter]);
183  wwp -= proj * vpVEC[iiter];
184  }
185 
186  // wwaug = Gaug zzaug
187  precond.multiply(ww, zz);
188  zzp = wwp;
189 
190  // ttaug = HBHtaug zzaug
191  HBHt.multiply(zz, tt);
192  tt.axpy(zzp, dy);
193  ttp = dot_product(dy, zz) + sigma * zzp;
194 
195  beta = sqrt(dot_product(tt, ww) + ttp * wwp);
196 
197  vold = v;
198  voldp = vp;
199  v = ww;
200  vp = wwp;
201  v *= 1 / beta;
202  vp *= 1 / beta;
203  tt *= 1 / beta;
204  zz *= 1 / beta;
205  ttp *= 1 / beta;
206  zzp *= 1 / beta;
207 
208  vVEC.push_back(v);
209  zVEC.push_back(zz);
210  tVEC.push_back(tt);
211  vpVEC.push_back(vp);
212  zpVEC.push_back(zzp);
213  tpVEC.push_back(ttp);
214 
215  alphas.push_back(alpha);
216 
217  y.clear();
218  if (jiter == 0) {
219  y.push_back(beta0 / alpha);
220  } else {
221  // Solve the tridiagonal system T_jiter y_jiter = beta0 * e_1
222  d.clear();
223  for (int iiter = 0; iiter <= jiter; ++iiter) {
224  d.push_back(beta0 * (dot_product(tVEC[0], vVEC[iiter]) +
225  tpVEC[0] * vpVEC[iiter]));
226  }
227  TriDiagSolve(alphas, betas, d, y);
228  }
229 
230  // Gradient norm in precond metric --> sqrt(r't) --> beta * y(jiter)
231  double rznorm = beta * std::abs(y[jiter]);
232 
233  normReduction = rznorm / beta0;
234 
235  betas.push_back(beta);
236 
237  Log::info() << "RPLanczos end of iteration " << jiter + 1
238  << ". Norm reduction= " << util::full_precision(normReduction)
239  << std::endl
240  << std::endl;
241 
242  if (normReduction < tolerance) {
243  Log::info() << "RPLanczos: Achieved required reduction in residual norm."
244  << std::endl;
245  break;
246  }
247  }
248 
249  // Calculate the solution (xh = Binv x)
250  for (int iiter = 0; iiter < jiter; ++iiter) {
251  vv.axpy(y[iiter], zVEC[iiter]);
252  vvp += y[iiter] * zpVEC[iiter];
253  }
254 
255  Log::info() << "RPLanczos: end" << std::endl;
256 
257  return normReduction;
258 }
