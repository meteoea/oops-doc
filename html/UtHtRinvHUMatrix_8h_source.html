89 template <typename MODEL>
90 void UtHtRinvHUMatrix<MODEL>::multiply(const CtrlVec_ &dvin,
91  CtrlVec_ &dvout) const {
92  // Increment counter
93  iter_++;
94 
95  // Multiply by the square root of the covariance
96  j_.jb().multiplyBsqrt(dvin, dx_);
97 
98  // Setup TL terms of cost function
99  PostProcessorTL<Increment_> costtl;
100  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
101  costtl.enrollProcessor(j_.jterm(jj).setupTL(dx_));
102  }
103 
104  // Run TLM
105  mdx_ = dx_;
106  j_.runTL(mdx_, costtl);
107 
108  // Get TLM outputs, multiply by covariance inverses, and setup ADJ forcing
109  // terms
110  j_.zeroAD(dz_);
111  PostProcessorAD<Increment_> costad;
112 
113  ww_.reset(new DualVector<MODEL>());
114  zz_.reset(new DualVector<MODEL>());
115 
116  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
117  ww_->append(costtl.releaseOutputFromTL(jj));
118  if (diags_ && j_.jterm(jj).classname() == "Jo") {
119  // Required for diagnostic quadratic JO-table
120  // save H dx[i]= H U dv[i], where
121  // dx = sum_i=1^N dx[i] = U sum_i=1^N dv[i] ss[i]
122  DualVector<MODEL> wwhudvii;
123  wwhudvii.append(costtl.releaseOutputFromTL(jj));
124  wwhudvvec_.push_back(wwhudvii);
125  }
126  zz_->append(j_.jterm(jj).multiplyCoInv(*ww_->getv(jj)));
127  costad.enrollProcessor(j_.jterm(jj).setupAD(zz_->getv(jj), dz_));
128  }
129 
130  // Run ADJ
131  j_.runAD(dz_, costad);
132 
133  // Multiply by the transpose of the square root of the covariance
134  j_.jb().multiplyBsqrtTrans(dz_, dvout);
135 }

139 template <typename MODEL>
140 void UtHtRinvHUMatrix<MODEL>::computeDiagnostics(
141  std::vector<double> &ss, const bool &convergenceReached) const {
142  if (diags_ && (iter_ <= 2 || convergenceReached)) {
143  Log::info() << "Online diagnostics:" << std::endl
144  << "------------------" << std::endl;
145  // Quadratic JO-table
146 
147  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
148  if (j_.jterm(jj).classname() == "Jo") {
149  const CostJo_ &jo = dynamic_cast<const CostJo_ &>(j_.jterm(jj));
150 
151  // We would like to compute H dx = H U dv
152  // Note that dv = sum_i=1^N dv[ii] ss[ii]
153  DualVector<MODEL> wwhudv(wwhudvvec_[0]);
154  DualVector<MODEL> wwhudvii(wwhudvvec_[0]);
155  DualVector<MODEL> dd;
156  DualVector<MODEL> zz;
157  wwhudv.zero();
158  wwhudvii.zero();
159  for (unsigned ii = 0; ii < iter_; ++ii) {
160  // compute H U dv[i] ss[i]
161  wwhudvii = wwhudvvec_[ii];
162  wwhudvii *= ss[ii];
163  // compute H dx = H U dv = sum_i=1^N H U dv[ii] ss[ii]
164  wwhudv += wwhudvii;
165  }
166  // Release d = y - H_NL(x)
167  dd.append(j_.jterm(jj).newDepartures());
168  // Compute dp = y - H_NL(x) - H(dx) = d - H(dx)
169  dd -= wwhudv;
170  // Compute R^-1 (dp)
171  zz.append(j_.jterm(jj).multiplyCoInv(*dd.getv(jj)));
172  // print JO table
173  jo.printJo(dynamic_cast<const Departures_ &>(*dd.getv(jj)),
174  dynamic_cast<const Departures_ &>(*zz.getv(jj)),
175  CostJo_::OS::Quadratic);
176  if (config_.has("onlineDiagnostics.residue")) {
177  dynamic_cast<const Departures_ &>(*dd.getv(jj)).save(
178  eckit::LocalConfiguration(config_, "onlineDiagnostics.residue"));
179  }
180  Log::info() << "----------" << std::endl;
181  }
182  }
183 
184  // Online adjoint test
185 
186  // <G dx, dy>, where dy = Rinv H dx
187  double adj_tst_fwd = dot_product(*ww_, *zz_);
188  // <dx, Gt dy> , where dy = Rinv H dx
189  double adj_tst_bwd = dot_product(dx_, dz_);
190 
191  Log::info() << "Online adjoint test, iteration: " << iter_ << std::endl
192  << util::PrintAdjTest(adj_tst_fwd, adj_tst_bwd, "G")
193  << "----------" << std::endl;
194  }
195 }
