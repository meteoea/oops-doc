90 template <typename INCR>
91 void WeightedDiffAD<INCR>::doFirstAD(INCR &dx, const util::DateTime &bgn,
92  const util::Duration &tstep) {
93  const util::DateTime end(dx.validTime());
94  ASSERT(bgn <= end);
95  if (!linit_ && bgn <= end_ && end >= bgn_) {
96  if (tstep_ == util::Duration(0)) tstep_ = tstep;
97  ASSERT(tstep_ > util::Duration(0));
98  weights_ = wfct_.setWeights(bgn_, end_, tstep_);
99  linit_ = true;
100  ASSERT(weights_.find(vtime_) != weights_.end());
101  weights_[vtime_] -= 1.0;
102  }
103  bgnleg_ = bgn;
104  endleg_ = end;
105  Log::info() << "WeightedDiffAD: initialized"
106  << " bgnleg_ = " << bgnleg_ << ", endleg_ = " << endleg_
107  << ", bgn_ = " << bgn_ << ", end_ = " << end_ << std::endl;
108  Log::debug() << " Weights:" << std::endl;
109  for (auto const &weight : weights_) {
110  Log::debug() << " " << weight.first << " : " << weight.second
111  << std::endl;
112  }
113 }

117 template <typename INCR>
118 void WeightedDiffAD<INCR>::doProcessingAD(INCR &dx) {
119  const util::DateTime now(dx.validTime());
120  if (((bgnleg_ < end_ && endleg_ > bgn_) || bgnleg_ == endleg_) &&
121  (now != endleg_ || now == end_ || now == bgnleg_)) {
122  ASSERT(weights_.find(now) != weights_.end());
123  const double zz = weights_[now];
124  dx.axpy(zz, *forcing_, false);
125  sum_ += zz;
126  }
127 }
