291 template <typename MODEL>
292 void Minimizer<MODEL>::tlmTaylorTest() const {
301  Log::info() << std::endl
302  << "TLM Taylor Test: " << outerIteration_ << std::endl;
303 
305  const M_ M(J_);
306 
308  CtrlInc_ dx(J_.jb().getFirstGuessIncrement(), false);
309  J_.jb().randomize(dx);
310 
313  PostProcessor<State_> pp;
314  ControlVariable<MODEL> mxx(J_.jb().getBackground());
315  J_.runNL(mxx, pp);
316 
319  CtrlInc_ mdx(dx);
320  M.multiply(mdx);
321 
323  for (unsigned int i = 0; i < 14; ++i) {
325  CtrlInc_ pmdx(mdx);
326  pmdx *= 1. / pow(10.0, i);
327  double denom = sqrt(dot_product(pmdx, pmdx));
328 
330  ControlVariable<MODEL> mpertxx(J_.jb().getBackground());
331  CtrlInc_ pdx(dx);
332  pdx *= 1. / pow(10.0, i);
333  J_.addIncrement(mpertxx, pdx);
334  J_.runNL(mpertxx, pp);
335 
337  CtrlInc_ diff_nl(mdx, false);
338  diff_nl.state()[0].diff(mpertxx.state()[0], mxx.state()[0]);
339  double nom = sqrt(dot_product(diff_nl, diff_nl));
340 
342  Log::info() << std::endl
343  << "TLM Taylor test: p = " << std::setw(8) << 1. / pow(10.0, i)
344  << ", ||M(x) - M(x+p dx)|| / ||p M'(dx)|| = "
345  << util::full_precision(1. + std::abs(1. - nom / denom))
346  << std::endl
347  << std::endl;
348  }
349  Log::info() << std::endl;
350 }

355 void Minimizer<MODEL>::adjModelTest() const {
363  const M_ M(J_);

366 
368  CtrlInc_ dx1(J_.jb().getFirstGuessIncrement(), false);
369  CtrlInc_ dx2(J_.jb().getFirstGuessIncrement(), false);
370 
371  J_.jb().randomize(dx1);
372  J_.jb().randomize(dx2);
373 
375  CtrlInc_ mdx1(dx1);
376  M.multiply(mdx1);
377 
379  CtrlInc_ mtdx2(dx2);
380  mtdx2.state()[0].updateTime(mdx1.state()[0].validTime() -
381  dx1.state()[0].validTime());
382  Mt.multiply(mtdx2);
383 
385  double adj_tst_fwd = dot_product(mdx1, dx2);
386 
388  double adj_tst_bwd = dot_product(dx1, mtdx2);
389 
391  Log::info() << "Model Adjoint Test: " << outerIteration_ << std::endl
392  << util::PrintAdjTest(adj_tst_fwd, adj_tst_bwd, "M") << std::endl
393  << std::endl;
