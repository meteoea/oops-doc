139 template <typename MODEL>
140 CostJcDFI<MODEL>::CostJcDFI(const eckit::Configuration &conf,
141  const Geometry_ &resol, const util::DateTime &vt,
142  const util::Duration &span,
143  const util::Duration &tstep)
144  : conf_(conf),
145  resol_(resol),
146  tlres_(),
147  ltraj_(false),
148  vt_(vt),
149  span_(span),
150  tstep_(tstep),
151  tlstep_(),

153  xdep_(),
154  gradFG_(),
155  wfct_(),
156  wfcttraj_(),
157  filter_(),
158  filtertraj_() {
159  alpha_ = conf.getDouble("alpha");
160  if (conf.has("ftime")) vt_ = util::DateTime(conf.getString("ftime"));
161  if (conf.has("span")) span_ = util::Duration(conf.getString("span"));
162  if (conf.has("covariance"))
163  jcovconf_.reset(new eckit::LocalConfiguration(conf, "covariance"));
164  // wfct_.reset(WeightFactory::create(config)); YT
165  wfct_.reset(new DolphChebyshev(conf_));
166  wfcttraj_.reset(new DolphChebyshev(conf_));
167  Log::debug() << "CostJcDFI created vt = " << vt_ << ", span = " << span_
168  << std::endl;
169  Log::trace() << "CostJcDFI created" << std::endl;
170 }

175 std::shared_ptr<PostBase<State<MODEL> > > CostJcDFI<MODEL>::initialize(
176  const CtrlVar_ &x, const eckit::Configuration &) const {
177  filter_.reset(new WeightedDiff<MODEL, Increment_, State_>(
178  vt_, span_, resol_, conf_, tstep_, *wfct_));

187 template <typename MODEL>
188 double CostJcDFI<MODEL>::finalize(const eckit::Configuration &) const {
189  double zz = 0.5 * alpha_;
190  std::unique_ptr<Increment_> dx(filter_->releaseDiff());
191  std::unique_ptr<Increment_> dxnorm(new Increment_(*dx));
192  if (jcovconf_) {
193  jccov_->inverseMultiply(*dx, *dxnorm);
194  }
195  zz *= dot_product(*dx, *dxnorm);
196  filter_.reset();
197  Log::test() << "CostJcDFI: Nonlinear Jc = " << zz << std::endl;
198  return zz;
199 }

203 template <typename MODEL>
204 std::shared_ptr<PostBase<State<MODEL> > > CostJcDFI<MODEL>::initializeTraj(
205  const CtrlVar_ &x, const Geometry_ &tlres,
206  const eckit::Configuration &innerConf) {
207  ltraj_ = true;
208  tlres_.reset(new Geometry_(tlres));
209  tlstep_ = util::Duration(innerConf.getString("linearmodel.tstep"));
210  filtertraj_.reset(new WeightedDiff<MODEL, Increment_, State_>(vt_, span_, *tlres_,
211  conf_, tlstep_, *wfcttraj_));
212  if (jcovconf_) {
213  if (!jccov_.get()) jccov_.reset(new JcCovar_(x.state()[0], x.state()[0].geometry(), *jcovconf_));
214  jccov_->linearize(x.state()[0], tlres);
215  }
216  return filtertraj_;
217 }

221 template <typename MODEL>
222 double CostJcDFI<MODEL>::finalizeTraj(const eckit::Configuration &) {
223  ASSERT(ltraj_ == true);
224  xdep_.reset(filtertraj_->releaseDiff());
225  bool lresetGradFG = false;
226  if (conf_.has("resetGradFG")) lresetGradFG = conf_.getBool("resetGradFG");
227  double zz = 0.;
228  if (lresetGradFG) {
229  gradFG_.reset(new Increment_(*xdep_));
230  xdep_->zero();
231  gradFG_->zero();
232  } else {
233  *xdep_ *= -1.0;
234  gradFG_.reset(this->multiplyCoInv(*xdep_));
235  zz = 0.5 * dot_product(*xdep_, *gradFG_);
236  }
237  ltraj_ = false;
238  filtertraj_.reset();
239  Log::info() << "CostJcDFI: Nonlinear Jc (traj) = " << zz << std::endl;
240  return zz;
241 }

277 template <typename MODEL>
278 Increment<MODEL> *CostJcDFI<MODEL>::multiplyCovar(
279  const GeneralizedDepartures &dv1) const {
280  const Increment_ &dx1 = dynamic_cast<const Increment_ &>(dv1);
281  Increment_ *dx2 = new Increment_(dx1);
282  if (jcovconf_) jccov_->multiply(dx1, *dx2);
283  const double za = 1.0 / alpha_;
284  *dx2 *= za;
285  return dx2;
286 }

290 template <typename MODEL>
291 Increment<MODEL> *CostJcDFI<MODEL>::multiplyCoInv(
292  const GeneralizedDepartures &dv1) const {
293  const Increment_ &dx1 = dynamic_cast<const Increment_ &>(dv1);
294  Increment_ *dx2 = new Increment_(dx1);
295  if (jcovconf_) jccov_->inverseMultiply(dx1, *dx2);
296  *dx2 *= alpha_;
297  double zz = dot_product(dx1, *dx2);
298  Log::info() << "CostJcDFI: multiplyCoInv norm = " << zz << std::endl;
299  return dx2;
300 }
