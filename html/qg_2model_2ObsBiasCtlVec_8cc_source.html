96 void ObsBiasCtlVec::read(const eckit::Configuration & conf) {
97  std::string filename = conf.getString("filename");
98  Log::info() << " ObsBiasCtlVec::read " << filename << std::endl;
99  eckit::PathName path = eckit::LocalPathName::cwd() + "/" + filename;
100  std::ifstream ifs(path.localPath());
101  if (!ifs)
102  throw eckit::CantOpenFile(path.localPath(), Here());
103  for (size_t ii = 0; ii < ObsBias::ntypes; ++ii) {
104  if (!ifs.bad()) {
105  std::string sqrtbias;
106  ifs >> sqrtbias;
107  if (ifs.good()) {
108  sqrtbias_[ii] = std::stod(sqrtbias);
109  } else if (ifs.eof()) {
110  throw eckit::ReadError("Failed to read entire ObsBias", Here());
111  } else if (ifs.fail()) {
112  throw eckit::ReadError("Failed to read ObsBias", Here());
113  }
114  }
115  }
116  Log::info() << " DEBUG ObsBiasCtlVec::read " << util::full_precision(sqrtbias_) << std::endl;
117 }

186 double ObsBiasCtlVec::norm() const {
187  double zz = 0.0;
188  int ii = 0;
189  for (unsigned int jj = 0; jj < ObsBias::ntypes; ++jj) {
190  if (active_[jj]) {
191  zz += sqrtbias_[jj] * sqrtbias_[jj];
192  ++ii;
193  }
194  }
195  if (ii>0) zz = std::sqrt(zz/ii);
196  return zz;
197 }
