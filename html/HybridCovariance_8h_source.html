116 template <typename MODEL>
117 void HybridCovariance<MODEL>::multiply(const Increment_ &dxi,
118  Increment_ &dxo) const {
119  static_->multiply(dxi, dxo);
120  dxo *= (1.0 - ensWeight_);
121  Increment_ tmp(dxo);
122  ensemble_->multiply(dxi, tmp);
123  dxo.axpy(ensWeight_, tmp);
124 }

126 template <typename MODEL>
127 void HybridCovariance<MODEL>::inverseMultiply(const Increment_ &dxi,
128  Increment_ &dxo) const {
129  IdentityMatrix<Increment_> Id;
130  dxo.zero();
131  GMRESR(dxo, dxi, *this, Id, 10, 1.0e-3);
132 }

134 template <typename MODEL>
135 void HybridCovariance<MODEL>::multiplySqrt(const IncrCtlVec_ &dv,
136  Increment_ &dx) const {
137  static_->multiplySqrt(dv, dx);
138  dx *= std::sqrt(1.0 - ensWeight_);
139  Increment_ dxe(dx);
140  dxe.zero();
141  ensemble_->multiplySqrt(dv, dxe);
142  dx.axpy(std::sqrt(ensWeight_), dxe);
143 }

145 template <typename MODEL>

147  IncrCtlVec_ &dv) const {
148  Increment_ dxm(dx);
149  dxm *= std::sqrt(1.0 - ensWeight_);
150  static_->multiplySqrtTrans(dxm, dv);
151  Increment_ dxe(dx);
152  dxe *= std::sqrt(ensWeight_);
153  ensemble_->multiplySqrtTrans(dxe, dv);
154 }
