58 template <typename MODEL>
59 void HtRinvHdxFG<MODEL>::compute(CtrlInc_& dz) const {
60  // Setup TL terms of cost function
61  PostProcessorTL<Increment_> costtl;
62  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
63  costtl.enrollProcessor(j_.jterm(jj).setupTL(dx_));
64  }
65 
66  // Run TLM
67  CtrlInc_ mdx(dx_);
68  j_.runTL(mdx, costtl);
69 
70  // Get TLM outputs, multiply by covariance inverses, and setup ADJ forcing
71  // terms
72  j_.zeroAD(dz);
73  PostProcessorAD<Increment_> costad;
74 
75  DualVector<MODEL> ww;
76  DualVector<MODEL> zz;
77 
78  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
79  ww.append(costtl.releaseOutputFromTL(jj));
80  zz.append(j_.jterm(jj).multiplyCoInv(*ww.getv(jj)));
81  costad.enrollProcessor(j_.jterm(jj).setupAD(zz.getv(jj), dz));
82  }
83 
84  // Run ADJ
85  j_.runAD(dz, costad);
86 }
