52  using Geometry_ = Geometry<MODEL>;

60 
61  public:
62  ModelSpaceCovariance4DBase(const Geometry_ &,
63  const eckit::Configuration & config): isAdvected_(config.has("advection")) {
64  if (isAdvected_) {
65  eckit::LocalConfiguration cfgAdv(config, "advection");
66  adv_.reset(new Advector_(cfgAdv));
67  Advector_::setup(adv_);
68  }
69  }
70 
71  virtual ~ModelSpaceCovariance4DBase() {
72  if (isAdvected_) {
73  std::shared_ptr<Advector_> noadv;
74  Advector_::setup(noadv);
75  }
76  }
77 
78  void linearize(const State4D_ & xb, const Geometry_ & resol,const eckit::Configuration & config) {
79  advectedLinearize(xb, resol, config);
80  }
81 
82  virtual void multiply(const Increment4D_ & dxi, Increment4D_ & dxo) const {
83  if (isAdvected_) {
84  Increment4D_ dxadv(dxi);
85  adv_->mult_ad(dxi, dxadv);
86  advectedMultiply(dxadv, dxo);
87  adv_->mult(dxo, dxo);
88  }
89  else {
90  advectedMultiply(dxi, dxo);
91  }
92  }
93 
94  virtual void multiplySqrt(const IncrCtlVec_ & dvi, Increment4D_ & dxo) const {
95  advectedMultiplySqrt(dvi, dxo);
96  if (isAdvected_) adv_->mult(dxo, dxo);
97  }
98 
99  virtual void multiplySqrtTrans(const Increment4D_ & dxi, IncrCtlVec_ & dvo) const {
100  if (isAdvected_) {
101  Increment4D_ dxadv(dxi);
102  adv_->mult_ad(dxi, dxadv);
103  advectedMultiplySqrtTrans(dxadv, dvo);
104  }
105  else {
106  advectedMultiplySqrtTrans(dxi, dvo);
107  }
108  }
109 
110  void inverseMultiply(const Increment4D_ & dxi, Increment4D_ & dxo) const {
111  Increment4D_ dxadv(dxi);
112  if (isAdvected_) adv_->mult_ad(dxi, dxadv);
113  advectedInverseMultiply(dxadv, dxo);
114  if (isAdvected_) adv_->mult(dxo, dxo);
115  }

121  virtual void advectedMultiplySqrt(const IncrCtlVec_ &,

192 template <typename MODEL>
193 Covariance4DFactory<MODEL>::Covariance4DFactory(const std::string &name) {
194  if (makers_.find(name) != makers_.end()) {
195  Log::error() << name << " already registered in covariance factory.";
196  ABORT("Element already registered in Covariance4DFactory.");
197  }
198  makers_[name] = this;
199 }

203 template <typename MODEL>
204 ModelSpaceCovariance4DBase<MODEL> *Covariance4DFactory<MODEL>::create(
205  const eckit::Configuration &conf, const Geometry_ &resol,
206  const Variables_ &vars, const State4D_ &xb) {
207  const std::string id = conf.getString("covariance");
208  Log::trace() << "ModelSpaceCovariance4DBase type = " << id;
209  typename std::map<std::string, Covariance4DFactory<MODEL> *>::iterator j =
210  makers_.find(id);
211  if (j == makers_.end()) {
212  Log::error() << id << " does not exist in covariance factory.";
213  Log::error() << "Covariance factory contains " << makers_.size()
214  << " elements:";
215  for (typename std::map<std::string,
216  Covariance4DFactory<MODEL> *>::const_iterator jj =
217  makers_.begin();
218  jj != makers_.end(); ++jj) {
219  Log::error() << "A" << jj->first << "B";
220  }
221  ABORT("Element does not exist in Covariance4DFactory.");
222  }
223  return (*j).second->make(conf, resol, vars, xb);
224 }
