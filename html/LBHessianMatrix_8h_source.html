49  void multiply(const CtrlInc_& dx, CtrlInc_& dz) const {
50  // Setup TL terms of cost function
51  PostProcessorTL<Increment_> costtl;
52  JqTermTL_* jqtl = j_.jb().initializeTL();
53  costtl.enrollProcessor(jqtl);
54  unsigned iq = 0;
55  if (jqtl) iq = 1;
56  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
57  costtl.enrollProcessor(j_.jterm(jj).setupTL(dx));
58  }
59 
60  // Run TLM
61  CtrlInc_ mdx(dx);
62  j_.runTL(mdx, costtl);
63 
64  // Finalize Jb+Jq
65 
66  // Get TLM outputs, multiply by covariance inverses and setup ADJ forcing
67  // terms
68  PostProcessorAD<Increment_> costad;
69  dz.zero();
70  CtrlInc_ dw(j_.jb());
71 
72  // Jb
73  CtrlInc_ tmp(j_.jb());
74  j_.jb().finalizeTL(jqtl, dx, dw);
75  tmp = dw;
76  JqTermAD_* jqad = j_.jb().initializeAD(dz, tmp);
77  costad.enrollProcessor(jqad);
78 
79  j_.zeroAD(dw);
80  // Jo + Jc
81  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
82  std::unique_ptr<GeneralizedDepartures> ww(
83  costtl.releaseOutputFromTL(iq + jj));
84  std::shared_ptr<GeneralizedDepartures> zz(
85  j_.jterm(jj).multiplyCoInv(*ww));
86  costad.enrollProcessor(j_.jterm(jj).setupAD(zz, dw));
87  }
88 
89  // Run ADJ
90  j_.runAD(dw, costad);
91 
92  // Multiply by B
93  CtrlInc_ zz(j_.jb());
94  j_.jb().multiplyB(dw, zz);
95 
96  dz += zz;
97  j_.jb().finalizeAD(jqad);
98  }
99 
100  private:
101  CostFct_ const& j_;
102 };
