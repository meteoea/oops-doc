79 void Run::execute(const Application& app) {
80  int status = 1;
81  Log::info() << "Run: Starting " << app << std::endl;
82  try {
83  Log::info() << "Full configuration is:" << *config_ << std::endl;
84  status = app.execute(*config_);
85  } catch (const eckit::Exception& e) {
86  status = 1;
87  Log::error() << e.what() << " caught in " << Here() << std::endl;
88  Log::error() << "Exception: " << app << " terminating..." << std::endl;
89  eckit::Exception::exceptionStack(eckit::Log::error(), true);
90  } catch (const std::exception& e) {
91  status = 1;
92  Log::error() << "Exception: " << e.what() << std::endl;
93  Log::error() << "Exception: " << app << " terminating..." << std::endl;
94  } catch (...) {
95  status = 1;
96  Log::error() << "Unknown exception: " << app << " terminating..."
97  << std::endl;
98  }
99  Log::info() << "Run: Finishing " << app << " with status = " << status
100  << std::endl;
101  if (status) ::exit(status);
102 }

104 eckit::LogTarget* Run::createInfoLogTarget() const {
105  const char* envlogfile = getenv("OOPS_LOGFILE");
106  if (envlogfile) {
107  static std::ofstream logfile;
108  char oopslogfile[100];
109  int rank = static_cast<int>(eckit::mpi::comm().rank());
110  if (rank==0) {
111  sprintf(oopslogfile, "%s", envlogfile);

113  sprintf(oopslogfile, "/dev/null");
114  }
115  logfile.open(oopslogfile, std::ofstream::out | std::ofstream::app);
116  return new eckit::OStreamTarget(logfile);
117  }
118  else {
119  return new eckit::OStreamTarget(std::cout);
