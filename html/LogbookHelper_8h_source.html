106 template <typename ValueT>
107 void LogbookHelper::record(const std::string &s, const ValueT &v,
108  const bool callbacks) {
109  ASSERT(getInstance().logbook_);
110  getInstance().logbook_->set<ValueT>(s, v);
111  // Call callback functions
112  if (callbacks)
113  for (auto const &cb : getInstance().getCallbacks()) {
114  cb(getConstLogbook());
115  }
116 }

120 template <typename ValueT>
121 void LogbookHelper::startRecording(const std::string &name, const ValueT &val) {
122  if (getInstance().on_) {
123  getInstance().status_[name] = Execution::Active;
124  getInstance().record<bool>(name + "_active", Execution::Active,
125  Callback::DontCall);
126  getInstance().record<ValueT>(name, val, Callback::Call);
127  }
