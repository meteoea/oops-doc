88 template <typename T>
89 bool is_close(T a, T b, T epsilon) {
90  // if nan or inf values, always return false
91  if (std::isnan(a) || std::isnan(b) || std::isinf(a) || std::isinf(b))
92  return false;
93 
94  // otherwise, create a relative tolerance that is of the same type as a and b
95  // (which is what is_approximately_equal wants) and call
96  // is_approximately_equal.
97  T AbsA = fabs(a);
98  T AbsB = fabs(b);
99  T EpsAB = (AbsA < AbsB ? AbsB : AbsA) *
100  epsilon; // greater of AbsA, AbsB times epsilon
101  return eckit::types::is_approximately_equal(a, b, EpsAB);
102 }
