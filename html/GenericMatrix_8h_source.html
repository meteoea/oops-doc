112 template <typename MODEL>
113 GenericMatrix<MODEL> *GenericMatrixFactory<MODEL>::create(
114  const eckit::Configuration &conf) {
115  Log::trace() << "GenericMatrix<MODEL>::create starting" << std::endl;
116  const std::string id = conf.getString("matrix");
117  typename std::map<std::string, GenericMatrixFactory<MODEL> *>::iterator jerr =
118  getMakers().find(id);
119  if (jerr == getMakers().end()) {
120  Log::error() << id << " does not exist in matrix factory." << std::endl;
121  ABORT("Element does not exist in GenericMatrixFactory.");
122  }
123  GenericMatrix<MODEL> *ptr = jerr->second->make(conf);
124  Log::trace() << "GenericMatrix<MODEL>::create done" << std::endl;
125  return ptr;
126 }
