59 template <typename MODEL>
60 void JqTermTL<MODEL>::computeModelErrorTL(Increment4D_ &dx) {
61  // Compute x_i - M(x_{i-1})
62  for (unsigned jsub = 1; jsub < nsubwin_; ++jsub) {
63  dx[jsub] -= mxi_[jsub - 1];
64  Log::info() << "CostJbJq: x_" << jsub << " - M(x_" << jsub - 1 << ")"
65  << dx[jsub] << std::endl;
66  }
67  mxi_.clear();
