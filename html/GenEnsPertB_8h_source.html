54  int execute(const eckit::Configuration& fullConfig) const {
55  // Setup resolution
56  const eckit::LocalConfiguration resolConfig(fullConfig, "resolution");
57  const Geometry_ resol(resolConfig);
58 
59  // Setup variables
60  const eckit::LocalConfiguration varConfig(fullConfig, "variables");
61  const Variables_ vars(varConfig);
62 
63  // Setup Model
64  const eckit::LocalConfiguration modelConfig(fullConfig, "model");
65  const Model_ model(resol, modelConfig);
66 
67  // Setup initial state
68  const eckit::LocalConfiguration initialConfig(fullConfig, "initial");
69  const State_ xx(resol, model, initialConfig);
70  Log::test() << "Initial state: " << xx.norm() << std::endl;
71 
72  // Setup augmented state
73  const ModelAux_ moderr(resol, model, initialConfig);
74 
75  // Setup times
76  const util::Duration fclength(fullConfig.getString("forecast_length"));
77  const util::DateTime bgndate(xx.validTime());
78  const util::DateTime enddate(bgndate + fclength);
79  Log::info() << "Running forecast from " << bgndate << " to " << enddate
80  << std::endl;
81 
82  // Setup B matrix
83  const eckit::LocalConfiguration covar(fullConfig, "Covariance");
84  std::unique_ptr<ModelSpaceCovarianceBase<MODEL> > Bmat(
85  CovarianceFactory<MODEL>::create(covar, resol, vars, xx));
86  Bmat->linearize(xx, resol, covar);
87 
88  // Generate perturbed states
89  Increment_ dx(resol, vars, bgndate);
90  const int members = fullConfig.getInt("members");
91  for (int jm = 0; jm < members; ++jm) {
92  Bmat->randomize(dx);
93  Log::debug() << "before copy xx:" << xx << std::endl;
94  State_ xp(xx);
95  Log::debug() << "after copy xx:" << xx << std::endl;
96  Log::debug() << "after copy xp:" << xp << std::endl;
97  xp += dx;
98 
99  // Setup forecast outputs
100  PostProcessor<State_> post;
101 
102  eckit::LocalConfiguration outConfig(fullConfig, "output");
103  outConfig.set("member", long(jm + 1));
104 
105  post.enrollProcessor(new StateWriter<State_>(bgndate, outConfig));
106 
107  // Run forecast
108  model.forecast(xp, moderr, fclength, post);
109  Log::test() << "Member " << jm << " final state: " << xp.norm()
110  << std::endl;
111  }
112 
113  return 0;
114  }
115  // -----------------------------------------------------------------------------
116  private:
117  std::string appname() const {
118  return "oops::GenEnsPertB<" + MODEL::name() + ">";
119  }
120  // -----------------------------------------------------------------------------
121 };
