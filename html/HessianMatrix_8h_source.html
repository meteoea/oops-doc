65 template <typename MODEL>
66 void HessianMatrix<MODEL>::multiply(const CtrlInc_& dx, CtrlInc_& dz) const {
67  // Increment counter
68  iter_++;
69 
70  // Setup TL terms of cost function
71  PostProcessorTL<Increment_> costtl;
72  JqTermTL_* jqtl = j_.jb().initializeTL();
73  costtl.enrollProcessor(jqtl);
74  unsigned iq = 0;
75  if (jqtl) iq = 1;
76  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
77  costtl.enrollProcessor(j_.jterm(jj).setupTL(dx));
78  }
79 
80  // Run TLM
81  CtrlInc_ mdx(dx);
82  j_.runTL(mdx, costtl);
83 
84  // Finalize Jb+Jq
85 
86  // Get TLM outputs, multiply by covariance inverses and setup ADJ forcing
87  // terms
88  PostProcessorAD<Increment_> costad;
89  dz.zero();
90  CtrlInc_ dw(j_.jb());
91 
92  // Jb
93  CtrlInc_ tmp(j_.jb());
94  j_.jb().finalizeTL(jqtl, dx, dw);
95  j_.jb().multiplyBinv(dw, tmp);
96  JqTermAD_* jqad = j_.jb().initializeAD(dz, tmp);
97  costad.enrollProcessor(jqad);
98 
99  j_.zeroAD(dw);
100 
101  DualVector<MODEL> ww;
102  DualVector<MODEL> zz;
103 
104  // Jo + Jc
105  for (unsigned jj = 0; jj < j_.nterms(); ++jj) {
106  ww.append(costtl.releaseOutputFromTL(iq + jj));
107  zz.append(j_.jterm(jj).multiplyCoInv(*ww.getv(jj)));
108  costad.enrollProcessor(j_.jterm(jj).setupAD(zz.getv(jj), dw));
109  }
110 
111  // Run ADJ
112  j_.runAD(dw, costad);
113  dz += dw;
114  j_.jb().finalizeAD(jqad);
115 
116  if (test_) {
117  // <G dx, dy>, where dy = Rinv H dx
118  double adj_tst_fwd = dot_product(ww, zz);
119  // <dx, Gt dy> , where dy = Rinv H dx
120  double adj_tst_bwd = dot_product(dx, dw);
121 
122  Log::info() << "Online adjoint test, iteration: " << iter_ << std::endl
123  << util::PrintAdjTest(adj_tst_fwd, adj_tst_bwd, "G")
124  << std::endl;
125  }
126 }
