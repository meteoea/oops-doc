121 template <typename MODEL>
122 ObsOperatorBase<MODEL> *ObsOperatorFactory<MODEL>::create(
123  const ObsSpace_ &odb, const eckit::Configuration &conf) {
124  Log::trace() << "ObsOperatorBase<MODEL>::create starting" << std::endl;
125  Log::debug() << "ObsOperatorBase<MODEL>::create conf" << conf << std::endl;
126  const std::string id = conf.getString("ObsType");
127  typename std::map<std::string, ObsOperatorFactory<MODEL> *>::iterator jloc =
128  getMakers().find(id);
129  if (jloc == getMakers().end()) {
130  Log::error() << id << " does not exist in observation operator factory."
131  << std::endl;
132  ABORT("Element does not exist in ObsOperatorFactory.");
133  }
134  ObsOperatorBase<MODEL> *ptr = jloc->second->make(odb, conf);
135  Log::trace() << "ObsOperatorBase<MODEL>::create done" << std::endl;
136  return ptr;
