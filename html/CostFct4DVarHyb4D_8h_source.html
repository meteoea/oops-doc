90 template<typename MODEL>

119 template <typename MODEL>
120 void CostFct4DVarHyb4D<MODEL>::runTL(CtrlInc_ & dx,
121  PostProcessorTL<Increment_> & cost,
122  PostProcessor<Increment_> post,
123  const bool idModel) const {
124  PostProcessorTL<Increment_> no_cost;
125  PostProcessor<Increment_> no_post;
126 
130 
131  CostFct_::getTLM().forecastTL(dx.state()[0], dx.modVar(), util::Duration(0),
132  no_post, no_cost);
133  cost.initializeTL(dx.state()[0], dx.state()[ncontrol_].validTime(), util::Duration(0));
134 
135  for (int jsub = 1; jsub <= ncontrol_; ++jsub) {
136  util::DateTime now(windowBegin_ + jsub*windowSub_);
137 
138  ASSERT(dx.state()[jsub].validTime() == now);
139  CostFct_::getTLM().forecastTL(dx.state()[jsub], dx.modVar(), util::Duration(0),
140  no_post, no_cost);
141  cost.processTL(dx.state()[jsub]);
142  }
143  cost.finalizeTL(dx.state()[0]);
144 }

150  util::DateTime now(windowBegin_);
