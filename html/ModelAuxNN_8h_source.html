48  int execute(const eckit::Configuration& fullConfig) const {

50  const eckit::LocalConfiguration resolConfig(fullConfig, "resolution");
51  const Geometry_ resol(resolConfig);
52 
53  // Setup Model
54  const eckit::LocalConfiguration modelConfig(fullConfig, "model");
55  const Model_ model(resol, modelConfig);
56 
57  // Setup background state
58  const eckit::LocalConfiguration initialConfig(fullConfig, "initial");
59  const eckit::LocalConfiguration xbConfig(initialConfig, "xb");
60  State_ xb(resol, model, xbConfig);
61  Log::info() << "Background state: " << xb.norm() << std::endl;
62 
63  // Setup an state
64  const eckit::LocalConfiguration anConfig(initialConfig, "an");
65  State_ an(resol, model, anConfig);
66  Log::info() << "Previous Analysis state: " << an.norm() << std::endl;
67 
68  // Setup model error estimator
69  const ModelAuxEstimator_ moderrEstimator(resol, model, initialConfig);
70 
71  // Setup augmented state
72  ModelAux_ moderr(resol, model, initialConfig);
73 
74  // Estimate model error
75  moderrEstimator.estimate(xb, an, moderr);
76 
77  // Compute the model error norm
78  Log::test() << "ModelAuxControl state: " << moderr.norm() << std::endl;
79 
80  return 0;
