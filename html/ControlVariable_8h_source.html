161 template <typename MODEL>
162 ControlVariable<MODEL>::ControlVariable(const eckit::Configuration &conf,
163  const Geometry_ &resol,
164  const Model_ &model,
165  const ObsSpaces_ &odb)
166  : state4d_(eckit::LocalConfiguration(conf, "Jb.Background"), resol, model),
167  modbias_(),
168  obsbias_(odb, eckit::LocalConfiguration(conf, "Jo")) {
169  modbias_.reset(new ModelAuxCtrl_(resol, model,
170  conf.getSubConfiguration("Jb.Background")));
171  if (conf.has("Jb.Background.stateModAuxEstimator")) {
172  Log::info() << "ControlVariable<MODEL>::ControlVariable Estimate Model Bias"
173  << std::endl;
174  // Setup NN geometry
175  const eckit::LocalConfiguration resolConfig(
176  conf, "Jb.Background.stateModAuxEstimator.resolution");
177  const Geometry_ resolNN(resolConfig);
178 
179  // Setup model error estimator
180  const ModelAuxCtrlEstimator_ modauxEstimator(
181  resolNN, model, conf.getSubConfiguration("Jb.Background"));
182  // Setup MODERR at lowres
183  ModelAuxCtrl_ me(resolNN, *modbias_);
184 
185  // Setup BG predictor at lowres
186  State_ bg(resolNN, state4d_[0]);
187 
188  if (conf.has("Jb.Background.stateModAuxEstimator.an")) {
189  // Setup AN state predictor
190  const eckit::LocalConfiguration anConfig(
191  conf, "Jb.Background.stateModAuxEstimator.an");
192  State_ anhres(resol, model, anConfig);
193  State_ an(resolNN, anhres);
194 
195  // Estimate model error
196  modauxEstimator.estimate(bg, an, me);
197  } else {
198  // Estimate model error
199  modauxEstimator.estimate(bg, me);
200  }
201 
202  // Interpolate model error
203  modbias_.reset(new ModelAuxCtrl_(resol, me));
204  }
205  Log::trace() << "ControlVariable contructed" << std::endl;
206 }

280 double ControlVariable<MODEL>::norm() const {
281  double zz = state4d_.norm();
282  double zn = zz * zz;
283  zz = modbias_->norm();
284  zn += zz * zz;
285  zz = obsbias_.norm();
286  zn += zz * zz;
287  return sqrt(zn);
