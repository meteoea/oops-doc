95 template <typename MODEL>
96 CostFctWeak<MODEL>::CostFctWeak(const eckit::Configuration &config,
97  const Geometry_ &geom, const Model_ &model)
98  : CostFunction<MODEL>::CostFunction(config, geom, model),
99  tlforcing_(false),
100  geom_(geom),
101  ctlvars_(config) {
102  windowLength_ = util::Duration(config.getString("window_length"));
103  windowBegin_ = util::DateTime(config.getString("window_begin"));
104  windowEnd_ = windowBegin_ + windowLength_;
105  windowSub_ = util::Duration(config.getString("window_sub"));
106 
107  nsubwin_ = windowLength_.toSeconds() / windowSub_.toSeconds();
108  ASSERT(windowLength_.toSeconds() == windowSub_.toSeconds() * nsubwin_);
109 
110  if (config.has("tlforcing")) {
111  if (config.getString("tlforcing") == "on") tlforcing_ = true;
112  }
113 
114  this->setupTerms(config);
115  Log::trace() << "CostFctWeak constructed" << std::endl;
116 }

149 template <typename MODEL>
150 void CostFctWeak<MODEL>::runNL(CtrlVar_ &xx,
151  PostProcessor<State_> &post) const {
152  for (unsigned int jsub = 0; jsub < nsubwin_; ++jsub) {
153  util::DateTime bgn(windowBegin_ + jsub * windowSub_);
154  util::DateTime end(bgn + windowSub_);
155 
156  ASSERT(xx.state()[jsub].validTime() == bgn);
157  CostFct_::getModel().forecast(xx.state()[jsub], xx.modVar(), windowSub_,
158  post);
159  ASSERT(xx.state()[jsub].validTime() == end);
160  }
161 }

165 template <typename MODEL>
166 void CostFctWeak<MODEL>::runTL(CtrlInc_ &dx, PostProcessorTL<Increment_> &cost,
167  PostProcessor<Increment_> post,
168  const bool idModel) const {
169  for (int jsub = dx.state().first(); jsub <= dx.state().last(); ++jsub) {
170  util::DateTime bgn(windowBegin_ + jsub * windowSub_);
171  util::DateTime end(bgn + windowSub_);
172 
173  ASSERT(dx.state()[jsub].validTime() == bgn);
174  if (tlforcing_ && jsub > 0) dx.state()[jsub] += dx.state()[jsub - 1];
175  CostFct_::getTLM(jsub).forecastTL(dx.state()[jsub], dx.modVar(), windowSub_,
176  post, cost, idModel);
177  ASSERT(dx.state()[jsub].validTime() == end);
178  }
179 }

183 template <typename MODEL>
184 void CostFctWeak<MODEL>::runTL(CtrlInc_ &dx, const bool idModel) const {
185  PostProcessor<Increment_> post;
186  PostProcessorTL<Increment_> cost;
187  ASSERT(!tlforcing_);
188 
189  for (int jsub = dx.state().first(); jsub <= dx.state().last(); ++jsub) {
190  util::DateTime bgn(windowBegin_ + jsub * windowSub_);
191  util::DateTime end(bgn + windowSub_);
192  ASSERT(dx.state()[jsub].validTime() == bgn);
193 
194  if (idModel) {
195  dx.state()[jsub].updateTime(windowSub_);
196  } else {
197  CostFct_::getTLM(jsub).forecastTL(dx.state()[jsub], dx.modVar(),
198  windowSub_, post, cost);
199  }
200 
201  ASSERT(dx.state()[jsub].validTime() == end);
202  }
203 
204  dx.state().shift_forward();
205 }

221 template <typename MODEL>
222 void CostFctWeak<MODEL>::runAD(CtrlInc_ &dx, PostProcessorAD<Increment_> &cost,
223  PostProcessor<Increment_> post,
224  const bool idModel) const {
225  for (int jsub = dx.state().last(); jsub >= dx.state().first(); --jsub) {
226  util::DateTime bgn(windowBegin_ + jsub * windowSub_);
227  util::DateTime end(bgn + windowSub_);
228 
229  ASSERT(dx.state()[jsub].validTime() == end);
230  CostFct_::getTLM(jsub).forecastAD(dx.state()[jsub], dx.modVar(), windowSub_,
231  post, cost, idModel);
232  if (tlforcing_ && jsub > 0) dx.state()[jsub - 1] += dx.state()[jsub];
233  ASSERT(dx.state()[jsub].validTime() == bgn);

239 template <typename MODEL>
240 void CostFctWeak<MODEL>::runAD(CtrlInc_ &dx, const bool idModel) const {
241  PostProcessor<Increment_> post;
242  PostProcessorAD<Increment_> cost;
243  ASSERT(!tlforcing_);
244 
245  dx.state().shift_backward();
246 
247  for (int jsub = dx.state().last(); jsub >= dx.state().first(); --jsub) {
248  util::DateTime bgn(windowBegin_ + jsub * windowSub_);
249  util::DateTime end(bgn + windowSub_);
250  ASSERT(dx.state()[jsub].validTime() == end);
251 
252  if (idModel) {
253  dx.state()[jsub].updateTime(-windowSub_);
254  } else {
255  CostFct_::getTLM(jsub).forecastAD(dx.state()[jsub], dx.modVar(),
256  windowSub_, post, cost);
257  }
258 
259  ASSERT(dx.state()[jsub].validTime() == bgn);

265 template <typename MODEL>
266 void CostFctWeak<MODEL>::addIncr(CtrlVar_ &xx, const CtrlInc_ &dx,
267  PostProcessor<Increment_> &post) const {
268  if (tlforcing_) {
269  Increment_ xi(dx.state()[0]);
270  for (unsigned int jsub = 0; jsub < nsubwin_; ++jsub) {
271  if (jsub > 0) xi += dx.state()[jsub];
272  xx.state()[jsub] += xi;
273  if (jsub < nsubwin_ - 1) {
274  CostFct_::getTLM(jsub).forecastTL(xi, dx.modVar(), windowSub_, post);
275  }
276  }
277  } else {
278  for (unsigned int jsub = 0; jsub < nsubwin_; ++jsub) {
279  xx.state()[jsub] += dx.state()[jsub];
280  }
281  }
282 }
