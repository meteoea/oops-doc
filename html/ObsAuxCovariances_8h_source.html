99 template <typename MODEL>
100 ObsAuxCovariances<MODEL>::ObsAuxCovariances(const ObsSpaces_ &odb,
101  const eckit::Configuration &conf)
102  : covs_(0), odb_(), conf_(conf) {
103  Log::trace() << "ObsAuxCovariances<MODEL>::ObsAuxCovariances starting"
104  << std::endl;
105  odb_ = &odb;
106  std::vector<eckit::LocalConfiguration> obsconf;
107  conf.get("ObsTypes", obsconf);
108  for (std::size_t jobs = 0; jobs < obsconf.size(); ++jobs) {
109  std::shared_ptr<ObsAuxCovariance_> tmp(
110  new ObsAuxCovariance_(odb[jobs], obsconf[jobs]));
111  covs_.push_back(tmp);
112  }
113  Log::trace() << "ObsAuxCovariances<MODEL>::ObsAuxCovariances done"
114  << std::endl;
115 }
