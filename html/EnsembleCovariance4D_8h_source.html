106 template <typename MODEL>
107 void EnsembleCovariance4D<MODEL>::advectedLinearize(const State4D_ &xb,
108  const Geometry_ &resol,
109  const eckit::Configuration &) {
110  /*
111  * Build localization
112  */
113  geom_.reset(new Geometry_(resol));
114  const eckit::LocalConfiguration conf(config_, "localization");
115  loc_.reset(new Localization_(*geom_, conf));
116 
117  /*
118  * Build ensemble
119  */
120  std::vector<eckit::LocalConfiguration> confEnsemble;
121  config_.get("ensemble", confEnsemble);
122  nsubwin_ = confEnsemble.size();
123  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
124  EnsemblePtr_ ens_k(new Ensemble_(xb[jsub].validTime(), confEnsemble[jsub]));
125  ens_k->build(xb[jsub], resol);
126  EnsemblesCollection_::getInstance().put(xb[jsub].validTime(), ens_k);
127  }
128 
129  /*
130  * Read files
131  */
132  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
133  EnsemblePtr_ ens =
134  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
135  ens->read();
136  }
137 
138  /*
139  * Calculate mean
140  */
141  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
142  EnsemblePtr_ ens =
143  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
144  ens->to_perturbations();
145  }
146 
147  Log::trace() << "EnsembleCovariance4D linearized.";
148 }

150 template <typename MODEL>
151 void EnsembleCovariance4D<MODEL>::advectedMultiply(const Increment4D_ &dxi,
152  Increment4D_ &dxo) const {
153  Increment_ dx_loc(dxo[0]);
154  Increment_ tmp(dxo[0]);
155  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxi[0].validTime()];
156 
157  dxo.zero();
158  for (unsigned l = 0; l < e->size(); ++l) {
159  dx_loc.zero();
160  for (unsigned k2 = 0; k2 < nsubwin_; ++k2) {
161  EnsemblePtr_ e_k2 =
162  EnsemblesCollection_::getInstance()[dxi[k2].validTime()];
163  schur_add((*e_k2)[l], dxi[k2], dx_loc, tmp);
164  }
165 
166  loc_->multiply(dx_loc);
167 
168  for (unsigned k1 = 0; k1 < nsubwin_; ++k1) {
169  EnsemblePtr_ e_k1 =
170  EnsemblesCollection_::getInstance()[dxi[k1].validTime()];
171  schur_add((*e_k1)[l], dx_loc, dxo[k1], tmp);
172  }
173  }
174 }

207  EnsemblePtr_ e_k2 =
208  EnsemblesCollection_::getInstance()[dxi[k2].validTime()];
209  schur_add((*e_k2)[l], dxi[k2], dx, tmp);
210  }
211  loc_->multiplySqrtTrans(dx, dvo.ensCtlVec(l));
212  }
