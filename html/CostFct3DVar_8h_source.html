90 template <typename MODEL>
91 CostFct3DVar<MODEL>::CostFct3DVar(const eckit::Configuration &config,
92  const Geometry_ &geom, const Model_ &model)
93  : CostFunction<MODEL>::CostFunction(config, geom, model),
94  windowLength_(),
95  windowHalf_(),
96  zero_(0),
97  geom_(geom),
98  ctlvars_(config) {
99  windowLength_ = util::Duration(config.getString("window_length"));
100  windowBegin_ = util::DateTime(config.getString("window_begin"));
101  windowEnd_ = windowBegin_ + windowLength_;
102  windowHalf_ = windowBegin_ + windowLength_ / 2;
103  this->setupTerms(config);
104  Log::trace() << "CostFct3DVar constructed" << std::endl;

140  PostProcessor<State_> &post) const {
141  ASSERT(xx.state().checkStatesNumber(1));
142  ASSERT(xx.state()[0].validTime() == windowHalf_);
143  CostFct_::getModel().forecast(xx.state()[0], xx.modVar(), util::Duration(0),
