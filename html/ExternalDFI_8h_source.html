36 template <typename MODEL>
37 class ExternalDFI : public Application {
38  using Geometry_ = Geometry<MODEL>;
39  using Model_ = Model<MODEL>;
40  using ModelAux_ = ModelAuxControl<MODEL>;
41  using State_ = State<MODEL>;
42

44  // -----------------------------------------------------------------------------
45  ExternalDFI() {}

47  virtual ~ExternalDFI() {}

49  int execute(const eckit::Configuration& fullConfig) const {
50  // Setup resolution
51  const eckit::LocalConfiguration resolConfig(fullConfig, "resolution");
52  const Geometry_ resol(resolConfig);
53 
54  // Setup Model
55  const eckit::LocalConfiguration modelConfig(fullConfig, "model");
56  const Model_ model(resol, modelConfig);
57 
58  // Setup initial state
59  const eckit::LocalConfiguration initialConfig(fullConfig, "initial");
60  State_ xx(resol, model, initialConfig);
61  Log::test() << "Initial state: " << xx.norm() << std::endl;
62 
63  // Setup augmented state
64  const ModelAux_ moderr(resol, model, initialConfig);
65 
66  // Setup times
67  const util::Duration fclength(fullConfig.getString("forecast_length"));
68  const util::DateTime bgndate(xx.validTime());
69  const util::DateTime enddate(bgndate + fclength);
70  Log::info() << "Running forecast from " << bgndate << " to " << enddate
71  << std::endl;
72 
73  // Setup post-processing
74  PostProcessor<State_> post;
75 
76  eckit::LocalConfiguration prtConf;
77  fullConfig.get("prints", prtConf);
78  post.enrollProcessor(new StateInfo<State_>("fc", prtConf));
79 
80  // Setup DFI
81  PostProcessor<State_> pp(post);
82 
83  const eckit::LocalConfiguration dfiConf(fullConfig, "dfi");
84  const util::Duration dfispan(dfiConf.getString("filter_span"));
85  const util::DateTime dfitime(bgndate + dfispan / 2);
86  std::shared_ptr<WeightedMean<MODEL, State_> > pdfi(
87  new WeightedMean<MODEL, State_>(dfitime, dfispan, resol, dfiConf));
88  pp.enrollProcessor(pdfi);
89 
90  // Run DFI forecast
91  model.forecast(xx, moderr, dfispan, pp);
92 
93  // Retrieve initialized state
94  std::unique_ptr<State_> xdfi(pdfi->releaseMean());
95  Log::test() << "Filtered state: " << xdfi->norm() << std::endl;
96 
97  // Setup forecast outputs
98  const eckit::LocalConfiguration outConfig(fullConfig, "output");
99  post.enrollProcessor(new StateWriter<State_>(bgndate, outConfig));
100 
101  // Run forecast from initialized state
102  const util::Duration fclen = fclength - dfispan / 2;
103  if (fclength < util::Duration(0))
104  ABORT("DFI: fliter span longer than forecast");
105  State_ zz(*xdfi);
106  model.forecast(zz, moderr, fclen, post);
107  Log::test() << "Final state: " << zz.norm() << std::endl;
108 
109  return 0;
110  }
111  // -----------------------------------------------------------------------------
112  private:
113  std::string appname() const {
114  return "oops::ExternalDFI<" + MODEL::name() + ">";
115  }
116  // -----------------------------------------------------------------------------
117 };
