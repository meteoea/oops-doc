99 template <typename MODEL>
100 CostFct4DEnVar<MODEL>::CostFct4DEnVar(const eckit::Configuration &config,
101  const Geometry_ &geom,
102  const Model_ &model)
103  : CostFunction<MODEL>::CostFunction(config, geom, model),
104  zero_(0),
105  geom_(geom),
106  ctlvars_(config) {
107  windowLength_ = util::Duration(config.getString("window_length"));
108  windowBegin_ = util::DateTime(config.getString("window_begin"));
109  windowEnd_ = windowBegin_ + windowLength_;
110  windowSub_ = util::Duration(config.getString("window_sub"));
111 
112  ncontrol_ = windowLength_.toSeconds() / windowSub_.toSeconds();
113  ASSERT(windowLength_.toSeconds() == windowSub_.toSeconds() * ncontrol_);
114 
115  this->setupTerms(config);
116  Log::trace() << "CostFct4DEnVar constructed" << std::endl;
117 }

149 template <typename MODEL>
150 void CostFct4DEnVar<MODEL>::runNL(CtrlVar_ &xx,
151  PostProcessor<State_> &post) const {
152  PostProcessor<State_> no_post;
153  post.initialize(xx.state()[0], xx.state()[ncontrol_].validTime(),
154  util::Duration(0));
155 
156  for (int jsub = 1; jsub <= ncontrol_; ++jsub) {
157  util::DateTime now(windowBegin_ + jsub*windowSub_);
158 
159  State_ zz(xx.state()[jsub]);
160  ASSERT(zz.validTime() == now);
161  CostFct_::getModel().forecast(zz, xx.modVar(), util::Duration(0), no_post);
162  ASSERT(zz.validTime() == now);
163  post.process(zz);
164  }
165  post.finalize(xx.state()[0]);
166 }

170 template <typename MODEL>
171 void CostFct4DEnVar<MODEL>::runTL(CtrlInc_ &dx,
172  PostProcessorTL<Increment_> &cost,
173  PostProcessor<Increment_> post,
174  const bool idModel) const {
175  PostProcessorTL<Increment_> no_cost;
176  PostProcessor<Increment_> no_post;
177 
178  /* In the IFS, cost.finalizeTL has to be called once after all horizontal
179  * interpolations ( cost.processTL ). But a forecast is called at each
180  * timeslot in
181  * order to connect dx with the model object. */
182 
183  CostFct_::getTLM().forecastTL(dx.state()[0], dx.modVar(), util::Duration(0),
184  no_post, no_cost);
185  cost.initializeTL(dx.state()[0], dx.state()[ncontrol_].validTime(),
186  util::Duration(0));
187 
188  for (unsigned int jsub = 1; jsub <= ncontrol_; ++jsub) {
189  util::DateTime now(windowBegin_ + jsub * windowSub_);
190 
191  ASSERT(dx.state()[jsub].validTime() == now);
192  CostFct_::getTLM().forecastTL(dx.state()[jsub], dx.modVar(),
193  util::Duration(0), no_post, no_cost);
194  cost.processTL(dx.state()[jsub]);
195  }
196  cost.finalizeTL(dx.state()[0]);
197 }

201 template <typename MODEL>
202 void CostFct4DEnVar<MODEL>::zeroAD(CtrlInc_ &dx) const {
203  util::DateTime now(windowBegin_);
204  for (unsigned int jsub = 0; jsub <= ncontrol_; ++jsub) {
205  dx.state()[jsub].zero(now);
206  now += windowSub_;
207  }
208  dx.modVar().zero();
209  dx.obsVar().zero();
210 }

214 template <typename MODEL>
215 void CostFct4DEnVar<MODEL>::runAD(CtrlInc_ &dx,
216  PostProcessorAD<Increment_> &cost,
217  PostProcessor<Increment_> post,
218  const bool idModel) const {
219  PostProcessorAD<Increment_> no_cost;
220  PostProcessor<Increment_> no_post;
221 
222  /* In the IFS, cost.finalizeTL has to be called once after all horizontal
223  * interpolations ( cost.processTL ). But a forecast is called at each
224  * timeslot in
225  * order to connect dx with the model object. */
226 
227  CostFct_::getTLM().forecastAD(dx.state()[ncontrol_], dx.modVar(),
228  util::Duration(0), no_post, no_cost);
229  cost.initializeAD(dx.state()[ncontrol_], dx.state()[0].validTime(),
230  util::Duration(0));
231 
232  for (unsigned int jsub = ncontrol_; jsub >= 1; --jsub) {
233  util::DateTime now(windowBegin_ + jsub * windowSub_);
234 
235  ASSERT(dx.state()[jsub].validTime() == now);
236  CostFct_::getTLM().forecastAD(dx.state()[jsub], dx.modVar(),
237  util::Duration(0), no_post, no_cost);
238  cost.processAD(dx.state()[jsub]);
239  }
240  CostFct_::getTLM().forecastAD(dx.state()[0], dx.modVar(), util::Duration(0),
241  no_post, no_cost);
242  cost.finalizeAD(dx.state()[0]);
243 }
