136 template <typename MODEL>
137 void Ensemble<MODEL>::build(const State_ &xb, const Geometry_ &resol) {
138  /* Build objects for storing Ensemble datas */
139  unsigned int size = rank_;
140  if (is_distributed_) size = 2;
141 
142  for (unsigned int m = 0; m < size; ++m) {
143  Increment_ *dx = new Increment_(resol, vars_, validTime_);
144  dx->zero();
145  if (is_distributed_ && (m == 0)) {
146  dx->accumul(1.0, xb);
147  }
148  ensemblePerturbs_.emplace_back(dx);
149  }
150 }

167 template <typename MODEL>
168 void Ensemble<MODEL>::to_perturbations() {
169  double r = 1.0 / sqrt(((double)rank_ - 1.0));
170  double invrank = 1.0 / ((double)rank_);
171 
172  /* Update datas */
173  Increment_ sum(*ensemblePerturbs_[0]);
174  sum *= invrank;
175 
176  for (unsigned m = 1; m < rank_; ++m) {
177  sum.axpy(invrank, *ensemblePerturbs_[m]);
178  }
179 
180  Log::info() << "Ensemble.to_perturbations with inflation :" << infl_;
181  r *= infl_;
182  for (unsigned m = 0; m < rank_; ++m) {
183  *ensemblePerturbs_[m] -= sum;
184  *ensemblePerturbs_[m] *= r;
185  }
186 }

189 template <typename MODEL>
190 void Ensemble<MODEL>::to_perturbations(bool distributed, unsigned id) {
191  double r = 1.0 / sqrt(((double)rank_ - 1.0));
192  double invrank = 1.0 / ((double)rank_);
193 
194  /* Update datas */
195  Increment_ sum(*ensemblePerturbs_[0]);
196  sum *= invrank;
197 
198  ASSERT(distributed == is_distributed_);
199  util::MPIWrapper &mpiw = util::MPIWrapper::Instance();
200  Increment_ tmp(*ensemblePerturbs_[0]);
201  tmp.zero();
202 
203  if (mpiw.my_rank() == 0) {
204  int tag = 666 + 1;
205  eckit::mpi::Status st = mpiw.comm().probe(int(1), tag);
206  size_t size = eckit::round(mpiw.comm().getCount<char>(st),
207  8); //< round to nearest 8 bytes
208  eckit::Buffer b(size);
209 
210  // Get members
211  for (unsigned m = 1; m < rank_; ++m) {
212  tag = 666 + m;
213  b.zero();
214  mpiw.comm().receive(static_cast<char *>(b.data()), b.size(), m, tag);
215  eckit::ResizableMemoryStream sr(b);
216  sr >> tmp;
217  sum.axpy(invrank, tmp);
218  }
219  // Send the sum
220  b.zero();
221  eckit::ResizableMemoryStream ss(b);
222  ss << sum;
223  mpiw.comm().broadcast(static_cast<char *>(b.data()), ss.position(),
224  (size_t)0);
225  } else {
226  // Send my member
227  int tag = 666 + mpiw.my_rank();
228  eckit::Buffer b(32);
229  eckit::ResizableMemoryStream ss(b);
230  ss << (*ensemblePerturbs_[0]);
231  eckit::mpi::Request req = mpiw.comm().iSend(
232  static_cast<const char *>(b.data()), ss.position(), 0, tag);
233 
234  // Get the sum
235  b.zero();
236  mpiw.comm().broadcast(static_cast<char *>(b.data()), b.size(), (size_t)0);
237  eckit::ResizableMemoryStream sr(b);
238  sr >> sum;
239  mpiw.comm().wait(req);
240  }
241 
242  Log::info() << "Ensemble.to_perturbations with inflation :" << infl_;
243  r *= infl_;
244  int size = 1;
245  for (int m = 0; m < size; ++m) {
246  (*ensemblePerturbs_[m]) -= sum;
247  (*ensemblePerturbs_[m]) *= r;
248  }
249 
250  // Prepare package for MPI transfers
251  arch_.reset(new eckit::Buffer(32));
252  eckit::ResizableMemoryStream s(*arch_);
253  s << (*ensemblePerturbs_[0]);
254 }

256 template <typename MODEL>
257 void Ensemble<MODEL>::to_perturbations(int nb_sub_ensemble) {
258  int sub_rank = rank_ / nb_sub_ensemble;
259  double inv_sub_rank = 1.0 / ((double)sub_rank);
260  double r = 1.0 / sqrt(((double)rank_ - 1.0));
261 
262  Log::info() << "Ensemble.to_perturbations with inflation :" << infl_;
263  r *= infl_;
264 
265  /* Update datas */
266  for (int p = 0; p < nb_sub_ensemble; ++p) {
267  Log::info() << "Ensemble.to_perturbations, remove ensemble mean for "
268  "sub_ensemble # :"
269  << p + 1;
270 
271  Increment_ sum(*ensemblePerturbs_[p * sub_rank]);
272  sum *= inv_sub_rank;
273 
274  for (int m = 1; m < sub_rank; ++m) {
275  sum.axpy(inv_sub_rank, *ensemblePerturbs_[p * sub_rank + m]);
276  }
277 
278  for (int m = 0; m < sub_rank; ++m) {
279  *ensemblePerturbs_[p * sub_rank + m] -= sum;
280  *ensemblePerturbs_[p * sub_rank + m] *= r;
281  }
282  }
283 }
