105 template <typename MODEL>
106 void EnsembleCovariance4DScaleDependent<MODEL>::advectedLinearize(
107  const State4D_ &xb, const Geometry_ &resol, const eckit::Configuration &) {
108  const eckit::LocalConfiguration conf(config_, "localization");
109  const eckit::LocalConfiguration confScaleDecomposition(config_,
110  "scale_decomposition");
111 
112  // What type of SDL is demanded
113  SDLtype_ = confScaleDecomposition.getInt("formulation");
114 
115  // How many horizontal bands do we have?
116  nhorizBand_ = confScaleDecomposition.getInt("nb_wave_band_peaks");
117 
118  // How many localization lenghts do we have?
119  std::vector<eckit::LocalConfiguration> confHorizBand;
120  confScaleDecomposition.get("horiz_band", confHorizBand);
121  int nband = confHorizBand.size();
122  // Make sure that the above two numbers are equal
123  ASSERT(nband == nhorizBand_);
124 
125  geom_.reset(new Geometry_(resol));
126  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_ ; ++horizBandIndex) {
127  loc_.emplace_back(new Localization_(*geom_, confHorizBand[horizBandIndex]));
128  }
129 
130  /*
131  * Build ensemble
132  */
133  std::vector<eckit::LocalConfiguration> confEnsemble;
134  config_.get("ensemble", confEnsemble);
135  nsubwin_ = confEnsemble.size();
136  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
137  EnsemblePtr_ ens_k(new Ensemble_(xb[jsub].validTime(), confEnsemble[jsub]));
138  ens_k->build(xb[jsub], resol);
139  EnsemblesCollection_::getInstance().put(xb[jsub].validTime(), ens_k);
140  }
141 
142  /*
143  * Read files
144  */
145  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
146  EnsemblePtr_ ens =
147  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
148  ens->read();
149  }
150 
151  /*
152  * Calculate mean
153  */
154  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
155  EnsemblePtr_ ens =
156  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
157  ens->to_perturbations();
158  }
159 
160  /*
161  * Performed a horizontal scale decomposition of the perturbations
162  */
163  Log::info() << "Scale decomposition of the perturbation object";
164 
165  HorizScaleDecomposition_ fltbg(confScaleDecomposition);
166 
167  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) { // loop on ensemble time
168 
169  Log::info() << " > treating ensemble time # " << jsub + 1;
170  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
171 
172  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_;
173  ++horizBandIndex) {
174  EnsemblePtr_ ens(new Ensemble_(*e));
175  EnsembleKey k(xb[jsub].validTime(), horizBandIndex);
176  EnsemblesCollection_::getInstance().put(k, ens);
177  }
178 
179  for (unsigned l = 0; l < e->size(); ++l) { // loop on ensemble member
180  Log::info()
181  << " >> calling horiz_scale_decomposition for ensemble member # "
182  << l + 1;
183 
184  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_;
185  ++horizBandIndex) {
186  EnsembleKey k(xb[jsub].validTime(), horizBandIndex);
187  EnsemblePtr_ horizBandEns = EnsemblesCollection_::getInstance()[k];
188  fltbg.collect((*horizBandEns)[l], horizBandIndex);
189  }
190  fltbg.decompose((*e)[l], SDLtype_);
191  }
192  }
193 
194  Log::trace() << "EnsembleCovariance4DScaleDependent linearized.";
195 }

199 template <typename MODEL>
200 void EnsembleCovariance4DScaleDependent<MODEL>::advectedMultiply(
201  const Increment4D_ &dxi, Increment4D_ &dxo) const {
202  Increment_ ensAmplitude(dxo[0]);
203  Increment_ tmp(dxo[0]);
204  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxi[0].validTime()];
205  unsigned ksiIndex;
206 
207  // Note on the scale-dependent localization (SDL) formulation...
208  // We use here a B^1/2 approach to save computational cost, since:
209  // With B^1/2, the cost of the SDL approach is proportional to the number of
210  // scale-bands
211  // With B , the cost of the SDL approach is proportional to the SQUARE of
212  // the number of scale-bands
213 
214  //
215  // --- Apply B^T/2 ---
216  //
217 
218  Log::info() << "Aplying B^T/2";
219 
220  std::vector<std::unique_ptr<IncrEnsCtlVec_>> aksi;
221  if (SDLtype_ == 1) {
222  // Create one Ksi per ensemble member
223  for (unsigned l = 0; l < e->size(); ++l) {
224  IncrEnsCtlVec_ *ksi = new IncrEnsCtlVec_(*geom_, *loc_[0]);
225  aksi.emplace_back(ksi);
226  }
227  } else {
228  // Create one Ksi per ensemble member and per horizontal waveband
229  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_;
230  ++horizBandIndex) {
231  for (unsigned l = 0; l < e->size(); ++l) {
232  IncrEnsCtlVec_ *ksi = new IncrEnsCtlVec_(*geom_, *loc_[0]);
233  aksi.emplace_back(ksi);
234  }
235  }
236  }
237 
238  // loop on horizontal scale band
239  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
240  // loop on ensemble members
241  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
242  ensAmplitude.zero();
243  // loop on time
244  for (unsigned timeIndex = 0; timeIndex < nsubwin_; ++timeIndex) {
245  EnsembleKey k(dxi[timeIndex].validTime(), horizBandIndex);
246  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
247  schur_add((*e_t)[memberIndex], dxi[timeIndex], ensAmplitude, tmp);
248  }
249  if (SDLtype_ == 1) {
250  ksiIndex = memberIndex;
251  } else {
252  ksiIndex = horizBandIndex * e->size() + memberIndex;
253  }
254  loc_[horizBandIndex]->multiplySqrtTrans(ensAmplitude, *aksi[ksiIndex]);
255  }
256  }
257 
258  //
259  // --- Apply B^1/2 ---
260  //
261 
262  Log::info() << "Aplying B^1/2";
263 
264  dxo.zero();
265  // loop on horizontal scale band
266  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
267  // loop on ensemble members
268  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
269  ensAmplitude.zero();
270  if (SDLtype_ == 1) {
271  ksiIndex = memberIndex;
272  } else {
273  ksiIndex = horizBandIndex * e->size() + memberIndex;
274  }
275  loc_[horizBandIndex]->multiplySqrt(*aksi[ksiIndex], ensAmplitude);
276 
277  // loop on time
278  for (unsigned timeIndex = 0; timeIndex < nsubwin_; ++timeIndex) {
279  EnsembleKey k(dxi[timeIndex].validTime(), horizBandIndex);
280  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
281  schur_add((*e_t)[memberIndex], ensAmplitude, dxo[timeIndex], tmp);
282  }
283  }
284  }
285 }

288 template <typename MODEL>
289 void EnsembleCovariance4DScaleDependent<MODEL>::advectedMultiplySqrt(
290  const IncrCtlVec_ &dvi, Increment4D_ &dxo) const {
291  Increment_ ensAmplitude(dxo[0]);
292  Increment_ tmp(dxo[0]);
293  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxo[0].validTime()];
294 
295  Log::info() << "Aplying B^1/2";
296 
297  dxo.zero();
298  // loop on horizontal scale band
299  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
300  // loop on ensemble members
301  for (unsigned memberIndex = 0; memberIndex < e->size(); ++memberIndex) {
302  ensAmplitude.zero();
303  loc_[horizBandIndex]->multiplySqrt(dvi.ensCtlVec(memberIndex),
304  ensAmplitude);
305 
306  // loop on time
307  for (unsigned timeIndex = 0; timeIndex < nsubwin_; ++timeIndex) {
308  EnsembleKey k(dxo[timeIndex].validTime(), horizBandIndex);
309  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
310  schur_add((*e_t)[memberIndex], ensAmplitude, dxo[timeIndex], tmp);
311  }
312  }
313  }
