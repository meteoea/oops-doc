125 template<typename MODEL>
126 void EnsembleCovarianceSDL<MODEL>::linearize(const State_ & xb,
127  const Geometry_ & resol, const eckit::Configuration & config) {
128  geom_.reset(new Geometry_(resol));
129 
130  // Compute the ensemble of perturbations at time of xb.
131  ASSERT(xb.validTime() == time_);
132  EnsemblePtr_ ens(new Ensemble_(xb.validTime(), config));
133  ens->build(xb, resol);
134  EnsemblesCollection_::getInstance().put(xb.validTime(), ens);
135  ens->read();
136 
137  members_ = ens->size();
138  if (has_change_of_variable_) {
139  change_of_variable_.reset(
140  MatrixFactory_::create(eckit::LocalConfiguration(config, "change_of_variable")));
141  for (unsigned l = 0; l < members_; ++l) {
142  change_of_variable_->multiply((*ens)[l]);
143  }
144  }
145  ens->to_perturbations(1);
146 
147  /*
148  * Init Scale decomposition
149  */
150  const eckit::LocalConfiguration conf(config, "localization");
151  const eckit::LocalConfiguration confScaleDecomposition(conf, "scale_decomposition");
152 
153  // How many horizontal bands do we have?
154  nhorizBand_ = confScaleDecomposition.getInt("nb_wave_band_peaks");
155 
156  // How many localization lengths do we have?
157  std::vector < eckit::LocalConfiguration > confHorizBand;
158  confScaleDecomposition.get("horiz_band", confHorizBand);
159  int nband = confHorizBand.size();
160  // Make sure that the above two numbers are equal
161  ASSERT(nband == nhorizBand_);
162 
163  /*
164  * Build Localization
165  */
166  loc_.clear();
167  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
168  loc_.emplace_back(new Localization_(*geom_, confHorizBand[horizBandIndex]));
169  }
170 
171  /*
172  * Performed a horizontal scale decomposition of the perturbations
173  */
174 
175  Log::info() << "Scale decomposition of the perturbation object" << std::endl;
176 
177  HorizScaleDecomposition_ fltbg(confScaleDecomposition);
178 
179  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[xb.validTime()];
180  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
181  EnsemblePtr_ ens(new Ensemble_(*e));
182  EnsembleKey k(xb.validTime(),horizBandIndex);
183  EnsemblesCollection_::getInstance().put(k, ens);
184  }
185 
186  for (unsigned m = 0; m < members_; ++m) {
187  Log::info() << " >> calling horiz_scale_decomposition for ensemble member # " << m + 1 << std::endl;
188 
189  for (int horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
190  EnsembleKey k(xb.validTime(),horizBandIndex);
191  EnsemblePtr_ horizBandEns = EnsemblesCollection_::getInstance()[k];
192  fltbg.collect((*horizBandEns)[m], horizBandIndex);
193  }
194  int SDLtype = 1;
195  fltbg.decompose((*e)[m], SDLtype);
196  }
197 
198  Log::trace() << "EnsembleCovarianceSDL linearized." << std::endl;
199 }

201 template <typename MODEL>
202 void EnsembleCovarianceSDL<MODEL>::multiply(const Increment_ &dxi,
203  Increment_ &dxo) const {
204  Increment_ dxi_tmp(dxi);
205  dxo.zero();
206  if (has_change_of_variable_) change_of_variable_->transposeInverseMultiply(dxi_tmp);
207 
208  //
209  // --- B formulation ---
210  //
211 
212  // loop on horizontal scale band
213  for (unsigned horizBandIndex1 = 0; horizBandIndex1 < nhorizBand_; ++horizBandIndex1) {
214  EnsembleKey j1(dxi.validTime(),horizBandIndex1);
215  EnsemblePtr_ e_j1 = EnsemblesCollection_::getInstance()[j1];
216 
217  // loop on horizontal scale band
218  for (unsigned horizBandIndex2 = 0; horizBandIndex2 < nhorizBand_; ++horizBandIndex2) {
219  EnsembleKey j2(dxi.validTime(),horizBandIndex2);
220  EnsemblePtr_ e_j2 = EnsemblesCollection_::getInstance()[j2];
221 
222  for (unsigned k = 0; k < members_; ++k) {
223  std::unique_ptr<IncrEnsCtlVec<MODEL> > dv(this->newIncrEnsCtlVec());
224  Increment_ dx(dxi_tmp);
225 
226  dx.schur_product_with((*e_j1)[k]);
227  loc_[horizBandIndex1]->multiplySqrtTrans(dx, *dv);
228  loc_[horizBandIndex2]->multiplySqrt(*dv, dx);
229  dx.schur_product_with((*e_j2)[k]);
230  dxo.axpy(1.0, dx, false);
231  }
232  }
233  }
234 
235  /* No localization
236  for (unsigned horizBandIndex1 = 0; horizBandIndex1 < nhorizBand_; ++horizBandIndex1) {
237  EnsembleKey j1(dxi.validTime(),horizBandIndex1);
238  EnsemblePtr_ e_j1 = EnsemblesCollection_::getInstance()[j1];
239 
240  // loop on horizontal scale band
241  for (unsigned horizBandIndex2 = 0; horizBandIndex2 < nhorizBand_; ++horizBandIndex2) {
242  EnsembleKey j2(dxi.validTime(),horizBandIndex2);
243  EnsemblePtr_ e_j2 = EnsemblesCollection_::getInstance()[j2];
244 
245  for (unsigned int m = 0; m < members_; ++m) {
246  double zz = (*e_j2)[m].dot_product_with(dxi_tmp);
247  dxo.axpy(zz, (*e_j1)[m], false);
248  }
249  }
250  }*/
251 
252  if (has_change_of_variable_) change_of_variable_->inverseMultiply(dxo);
253 }

261 template <typename MODEL>
262 void EnsembleCovarianceSDL<MODEL>::multiplySqrt(const IncrCtlVec_ &dv,
263  Increment_ &dxo) const {
264  dxo.zero();
265  Increment_ dx(dxo);
266  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
267  EnsembleKey j(dx.validTime(),horizBandIndex);
268  EnsemblePtr_ e_j = EnsemblesCollection_::getInstance()[j];
269  for (unsigned int k = 0; k < members_; ++k) {
270  loc_[horizBandIndex]->multiplySqrt(dv.ensCtlVec(k), dx);
271  dx.schur_product_with((*e_j)[k]);
272  dxo.axpy(1.0, dx, false);
273  }
274  }
275  if (has_change_of_variable_) change_of_variable_->inverseMultiply(dxo);
276 }

278 template <typename MODEL>
279 void EnsembleCovarianceSDL<MODEL>::multiplySqrtTrans(const Increment_ &dxi,
280  IncrCtlVec_ &dv) const {
281  Increment_ dxi_tmp(dxi);
282  for (unsigned k = 0; k < members_; ++k) {
283  dv.ensCtlVec(k).zero();
284  }
285 
286  if (has_change_of_variable_) change_of_variable_->transposeInverseMultiply(dxi_tmp);
287  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
288  EnsembleKey j(dxi_tmp.validTime(),horizBandIndex);
289  EnsemblePtr_ e_j = EnsemblesCollection_::getInstance()[j];
290  for (unsigned int k = 0; k < members_; ++k) {
291  Increment_ dx(dxi_tmp);
292  dx.schur_product_with((*e_j)[k]);
293  loc_[horizBandIndex]->multiplySqrtTrans(dx, dv.ensCtlVec(k));
294  }
295  }

298 template<typename MODEL>
299 void EnsembleCovarianceSDL<MODEL>::randomize(Increment_ & dxo) const {
300  // Build dv
301  std::vector<std::unique_ptr<IncrEnsCtlVec<MODEL> > > dv;
302  for (unsigned int jj = 0; jj < members_; ++jj) {
303  dv.emplace_back(this->newIncrEnsCtlVec());
304  }
305 
306  // Initialize dv
307  for (unsigned int m = 0; m < members_; ++m) {
308  dv[m]->random();
309  }
310 
311  // dxo = B^1/2 dv
312  dxo.zero();
313  Increment_ dx(dxo);
314  for (unsigned horizBandIndex = 0; horizBandIndex < nhorizBand_; ++horizBandIndex) {
315  EnsembleKey k(dx.validTime(),horizBandIndex);
316  EnsemblePtr_ e_t = EnsemblesCollection_::getInstance()[k];
317  for (unsigned int m = 0; m < members_; ++m) {
318  loc_[horizBandIndex]->multiplySqrt(*dv[m], dx);
319  dx.schur_product_with((*e_t)[m]);
320  dxo.axpy(1.0, dx, false);
321  }
322  }
323  if (has_change_of_variable_) change_of_variable_->inverseMultiply(dxo);
324 }
