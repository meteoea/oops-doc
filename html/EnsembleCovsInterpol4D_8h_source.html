110 template <typename MODEL>
111 void EnsembleCovsInterpol4D<MODEL>::advectedLinearize(const State4D_ &xb,
112  const Geometry_ &resol,
113  const eckit::Configuration &) {
114  /*
115  * Build Geometry of the Ensemble
116  */
117  const eckit::LocalConfiguration confNewGeo(config_, "ensemble_resol");
118  resol_ens_.reset(new Geometry_(confNewGeo));
119 
120  /*
121  * Build localization
122  */
123  const eckit::LocalConfiguration conf(config_, "localization");
124  loc_.reset(new Localization_(*resol_ens_, conf));
125 
126  /*
127  * Build Interpolation operators
128  */
129  const eckit::LocalConfiguration confInterpol(config_, "interpol");
130  intdir_.reset(new Interpolator_(xb[0], *resol_ens_, confInterpol, false));
131  intadj_.reset(new Interpolator_(xb[0], *resol_ens_, confInterpol, true));
132 
133  /*
134  * Build ensemble
135  */
136  std::vector<eckit::LocalConfiguration> confEnsemble;
137  config_.get("ensemble", confEnsemble);
138  nsubwin_ = confEnsemble.size();
139  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
140  EnsemblePtr_ ens_k(new Ensemble_(xb[jsub].validTime(), confEnsemble[jsub]));
141  ens_k->build(xb[jsub], *resol_ens_);
142  EnsemblesCollection_::getInstance().put(xb[jsub].validTime(), ens_k);
143  }
144 
145  /*
146  * Read files
147  */
148  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
149  EnsemblePtr_ ens =
150  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
151  ens->read();
152  }
153 
154  /*
155  * Calculate mean
156  */
157  for (unsigned jsub = 0; jsub < nsubwin_; ++jsub) {
158  EnsemblePtr_ ens =
159  EnsemblesCollection_::getInstance()[xb[jsub].validTime()];
160  ens->to_perturbations();
161  }
162 
163  Log::trace() << "EnsembleCovsInterpol4D linearized.";
164 }

167 template <typename MODEL>
168 void EnsembleCovsInterpol4D<MODEL>::advectedMultiply(const Increment4D_ &dxi,
169  Increment4D_ &dxo) const {
170  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxi[0].validTime()];
171  dxo.zero();
172 
173  // Adjoint interpolation to the resolution of the perturbations
174  const eckit::LocalConfiguration confInterpol(config_, "interpol");
175  const Variables_ var(confInterpol);
176 
177  Increment4D_ dx_lr(*resol_ens_, dxi);
178  dx_lr.zero();
179 
180  for (unsigned k = 0; k < nsubwin_; ++k) {
181  Log::trace()
182  << "Adjoint Interpolation (HR -> LR) of the increment for timeslot #"
183  << k;
184  intadj_->interpolate(dxi[k], var, dx_lr[k]);
185  }
186 
187  Increment_ dx_lr_loc(dx_lr[0]);
188  Increment_ tmp(dx_lr[0]);
189  Increment4D_ dxo_lr(dx_lr);
190 
191  dxo_lr.zero();
192 
193  for (unsigned l = 0; l < e->size(); ++l) {
194  dx_lr_loc.zero();
195  for (unsigned k2 = 0; k2 < nsubwin_; ++k2) {
196  EnsemblePtr_ e_k2 =
197  EnsemblesCollection_::getInstance()[dx_lr[k2].validTime()];
198  schur_add((*e_k2)[l], dx_lr[k2], dx_lr_loc, tmp);
199  }
200 
201  loc_->multiply(dx_lr_loc);
202 
203  for (unsigned k1 = 0; k1 < nsubwin_; ++k1) {
204  EnsemblePtr_ e_k1 =
205  EnsemblesCollection_::getInstance()[dx_lr[k1].validTime()];
206  schur_add((*e_k1)[l], dx_lr_loc, dxo_lr[k1], tmp);
207  }
208  }
209 
210  // Back to HR
211  for (unsigned k = 0; k < nsubwin_; ++k) {
212  Log::trace()
213  << "Direct interpolation (LR -> HR) of the increment for timeslot #"
214  << k;
215  intdir_->interpolate(dxo_lr[k], var, dxo[k]);
216  }
217 }

221 void EnsembleCovsInterpol4D<MODEL>::advectedMultiplySqrt(
222  const IncrCtlVec_ &dvi, Increment4D_ &dxo) const {
223  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxo[0].validTime()];
224 
225  // Interpolation to the resolution of the increment
226  const eckit::LocalConfiguration confInterpol(config_, "interpol");
227  const Variables_ var(confInterpol);
228 
229  Increment4D_ dx_lr(*resol_ens_, dxo);
230  Increment_ dx_lr_loc(dx_lr[0]);
231  Increment_ tmp(dx_lr_loc);
232  Increment4D_ dxo_lr(dx_lr);
233 
234  dxo_lr.zero();
235  dxo.zero();
236  for (unsigned l = 0; l < e->size(); ++l) {
237  loc_->multiplySqrt(dvi.ensCtlVec(l), dx_lr_loc);
238 
239  for (unsigned k1 = 0; k1 < nsubwin_; ++k1) {
240  EnsemblePtr_ e_k1 =
241  EnsemblesCollection_::getInstance()[dxo[k1].validTime()];
242  schur_add((*e_k1)[l], dx_lr_loc, dxo_lr[k1], tmp);
243  }
244  }
245 
246  // Back to HR
247  for (unsigned k = 0; k < nsubwin_; ++k) {
248  Log::trace()
249  << "Direct interpolation (LR -> HR) of the increment for timeslot #"
250  << k;
251  intdir_->interpolate(dxo_lr[k], var, dxo[k]);
