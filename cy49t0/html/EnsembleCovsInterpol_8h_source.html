112 template<typename MODEL>
113 void EnsembleCovsInterpol<MODEL>::linearize(const State_ & xb,
114  const Geometry_ & resol, const eckit::Configuration & config) {
115 
116  /*
117  * Build Geometry of the Ensemble
118  */
119  const eckit::LocalConfiguration confNewGeo(config_, "ensemble_resol");
120  resol_ens_.reset(new Geometry_(confNewGeo));
121 
122  /*
123  * Build localization
124  */
125  const eckit::LocalConfiguration conf(config_, "localization");
126  loc_.reset(new Localization_(*resol_ens_, conf));
127 
128  /*
129  * Build Interpolation operators
130  */
131  const eckit::LocalConfiguration confInterpol(config_, "interpol");
132  intdir_.reset(new Interpolator_(xb,*resol_ens_,confInterpol, false));
133  intadj_.reset(new Interpolator_(xb,*resol_ens_,confInterpol, true));
134 
135  /*
136  * Build ensemble
137  */
138  std::vector < eckit::LocalConfiguration > confEnsemble;
139  config_.get("ensemble", confEnsemble);
140  EnsemblePtr_ ens_k(new Ensemble_(xb.validTime(), confEnsemble[0]));
141  ens_k->build(xb, *resol_ens_);
142  EnsemblesCollection_::getInstance().put(xb.validTime(), ens_k);
143 
144  /*
145  * Read files, calculate mean
146  */
147  EnsemblePtr_ ens = EnsemblesCollection_::getInstance()[xb.validTime()];
148  ens->read();
149  int nb_sub_ens=confEnsemble[0].getInt("sub_ens",1);
150  ens->to_perturbations(nb_sub_ens);
151 
152  Log::trace() << "EnsembleCovsInterpol linearized.";
153 
154 }

156 template<typename MODEL>
157 void EnsembleCovsInterpol<MODEL>::multiply(const Increment_ & dxi,
158  Increment_ & dxo) const {
159 
160  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[dxi.validTime()];
161  dxo.zero();
162 
163  // Adjoint interpolation to the resolution of the perturbations
164  const eckit::LocalConfiguration confInterpol(config_, "interpol");
165  const Variables_ var(confInterpol);
166 
167  Increment_ dx_lr(*resol_ens_, var, dxi.validTime(), dxi.validTime());
168  dx_lr.zero();
169 
170  Log::trace() << "Adjoint Interpolation (HR -> LR) of the increment";
171  intadj_->interpolate(dxi, var, dx_lr);
172 
173  Increment_ dx_lr_loc(dx_lr);
174  Increment_ tmp(dx_lr);
175  Increment_ dxo_lr(dx_lr);
176 
177  dxo_lr.zero();
178  for (unsigned l = 0; l < e->size(); ++l) {
179  dx_lr_loc.zero();
180  schur_add((*e)[l], dx_lr, dx_lr_loc, tmp);
181  loc_->multiply(dx_lr_loc);
182  schur_add((*e)[l], dx_lr_loc, dxo_lr, tmp);
183  }
184 
185  // Back to HR
186  Log::trace() << "Direct interpolation (LR -> HR) of the increment";
187  intdir_->interpolate(dxo_lr, var , dxo);
188 }

240  dv.ensCtlVec(l).zero();
241  dx_lr_loc.zero();

243  loc_->multiplySqrtTrans(dx_lr_loc, dv.ensCtlVec(l));
