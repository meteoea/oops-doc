241 template <typename MODEL>
242 double CostJbTotal<MODEL>::finalize(JqTerm_ *jqnl,
243  const eckit::Configuration &conf) const {
250  CtrlInc_ dx(*this);
251 
253  jb_->computeIncrement(xb_.state(), fg_->state(), dx.state());
254 
256  dx.modVar().diff(fg_->modVar(), xb_.modVar());
257  dx.obsVar().diff(fg_->obsVar(), xb_.obsVar());
258 
261  if (jqnl) jqnl->computeModelError(fg_->state(), dx.state());
262  Log::info() << "CostJb: FG-BG" << dx << std::endl;
263 
265  this->writeIncrement(dx, conf);
266 
268  if (conf.has("stateModAux")) {
269  fg_->modVar().write(conf);
270  }
271 
273  fg_ = nullptr;
274 
276  double zjb = this->evaluate(dx);
277  return zjb;
278 }

365 template <typename MODEL>
366 double CostJbTotal<MODEL>::finalizeTraj(JqTerm_ *jqnl,
367  const eckit::Configuration &conf) {
376  dxFG_.reset(new CtrlInc_(*this));
378 
379  // Compute x_0 - x_b for Jb
380  if (conf.has("explicitChangeResolution")) {
381  State4D_ xblr(resolution(), xb_.state());
382  jb_->computeIncrement(xblr, fg_->state(), dxFG_->state());
383  } else {
384  jb_->computeIncrement(xb_.state(), fg_->state(), dxFG_->state());
385  }
386 
387  // Model and Obs biases
388  dxFG_->modVar().diff(fg_->modVar(), xb_.modVar());
389  dxFG_->obsVar().diff(fg_->obsVar(), xb_.obsVar());
390 
392  if (jqnl) jqnl->computeModelError(fg_->state(), dxFG_->state());
393  Log::info() << "CostJb: FG-BG" << *dxFG_ << std::endl;
394 
397  if (conf.has("covariancesObsAux")) {
398  // Update ObsAuxCovariance
399  jbObsBias_.update(fg_->obsVar(), conf);
400  }
401 
402  // Reset first guess pointer
403  fg_ = nullptr;
404 
406  double zjb = this->evaluate(*dxFG_);
407  return zjb;
408 }

412 template <typename MODEL>
413 double CostJbTotal<MODEL>::evaluate(const CtrlInc_ &dx) const {
414  CtrlInc_ gg(dx, false);
415  this->multiplyBinv(dx, gg);
416 
417  double zjb = 0.0;
418  double zz = 0.5 * dot_product(dx.state(), gg.state());
419  Log::info() << "CostJb : Nonlinear Jb State = " << zz << std::endl;
420  zjb += zz;
421  zz = 0.5 * dot_product(dx.modVar(), gg.modVar());
422  Log::info() << "CostJb : Nonlinear Jb Model Aux = " << zz << std::endl;
423  zjb += zz;
424  zz = 0.5 * dot_product(dx.obsVar(), gg.obsVar());
425  Log::info() << "CostJb : Nonlinear Jb Obs Aux = " << zz << std::endl;
426  zjb += zz;
427 
428  Log::info() << "CostJb : Nonlinear Jb = " << zjb << std::endl;
429 
430  // Get rid of very small values for test
431  double ztest = zjb;
432  if (zjb >= 0.0 && zjb <= std::numeric_limits<double>::epsilon()) ztest = 0.0;
433  Log::test() << "CostJb : Nonlinear Jb = " << ztest << std::endl;
434 
435  return zjb;
436 }
