109 template <typename MODEL>
110 void EnsembleCovariance4DWithChangeOfVariable<MODEL>::advectedLinearize(
111  const State4D_ & xb, const Geometry_ & resol, const eckit::Configuration & config) {
112  unsigned nbtslots;
113  std::vector < eckit::LocalConfiguration > confEnsemble;
114  config.get("ensemble", confEnsemble);
115  nbtslots = confEnsemble.size();
116 
117  /*
118  * Build localization
119  */
120  geom_.reset(new Geometry_(resol));
121  const eckit::LocalConfiguration conf(config, "localization");
122  loc_.reset(new Localization_(*geom_, conf));
123 
124  /*
125  * Build ensemble
126  */
127  time_.clear();
128  for (unsigned jsub = 0; jsub < nbtslots; ++jsub) {
129  util::DateTime dt(confEnsemble[jsub].getString("reference_time"));
130  EnsembleKey id(dt, id_);
131 
132  EnsemblePtr_ ens_k(new Ensemble_(dt, confEnsemble[jsub]));
133  ens_k->build(xb[0], resol);
134  EnsemblesCollection_::getInstance().put(id, ens_k);
135  time_.push_back(dt);
136  }
137 
138  /*
139  * Read files
140  */
141  for (unsigned jsub = 0; jsub < ntslots(); ++jsub) {
142  EnsembleKey id(time_[jsub], id_);
143  EnsemblePtr_ ens = EnsemblesCollection_::getInstance()[id];
144  ens->read();
145  }
146 
147  change_of_variable_.reset(
148  MatrixFactory_::create(eckit::LocalConfiguration(config, "change_of_variable")));
149  for (unsigned jsub = 0; jsub < ntslots(); ++jsub) {
150  EnsembleKey id(time_[jsub], id_);
151  EnsemblePtr_ ens = EnsemblesCollection_::getInstance()[id];
152  for (unsigned l = 0; l < ens->size(); ++l) {
153  change_of_variable_->multiply((*ens)[l]);
154  }
155  }
156 
157  /*
158  * Calculate mean
159  */
160  int nb_sub_ens;
161  for (unsigned jsub = 0; jsub < ntslots(); ++jsub) {
162  EnsembleKey id(time_[jsub], id_);
163  nb_sub_ens = confEnsemble[jsub].getInt("sub_ens");
164  EnsemblePtr_ ens = EnsemblesCollection_::getInstance()[id];
165  ens->to_perturbations(nb_sub_ens);
166  }
167 
168  Log::trace() << "EnsembleCovariance4DWithChangeOfVariable linearized.";
169 }

180 template <typename MODEL>
181 void EnsembleCovariance4DWithChangeOfVariable<MODEL>::advectedMultiply(
182  const Increment4D_ &dxi, Increment4D_ &dxo) const {
183  dxo.zero();
184  Increment_ dx_loc(dxo[0]);
185  Increment4D_ dxi_tmp(dxi);
186  Increment4D_ dxo_tmp(dxo);
187  Increment_ tmp(dxo[0]);
188  EnsembleKey id0(time_[0], id_);
189  EnsemblePtr_ e = EnsemblesCollection_::getInstance()[id0];
190 
191  /*
192  * Change variable in dxi
193  */
194  toControlVariableTimes_ad(dxi, dxi_tmp);
195  for (unsigned k = 0; k < ntslots(); ++k) {
196  change_of_variable_->transposeInverseMultiply(dxi_tmp[k]);
197  }
198 
199  for (unsigned l = 0; l < e->size(); ++l) {
200  dx_loc.zero();
201  for (unsigned k2 = 0; k2 < ntslots(); ++k2) {
202  EnsembleKey idk2(time_[k2], id_);
203  EnsemblePtr_ e_k2 = EnsemblesCollection_::getInstance()[idk2];
204  schur_add((*e_k2)[l], dxi_tmp[k2], dx_loc, tmp);
205  }
206 
207  loc_->multiply(dx_loc);
208 
209  for (unsigned k1 = 0; k1 < ntslots(); ++k1) {
210  EnsembleKey idk1(time_[k1], id_);
211  EnsemblePtr_ e_k1 = EnsemblesCollection_::getInstance()[idk1];
212  schur_add((*e_k1)[l], dx_loc, dxo_tmp[k1], tmp);
213  }
214 
215  /*
216  * Version without any localization ... for testing and debugging
217  *
218  dxo.zero();
219  for (unsigned l = 0; l < e->size(); ++l) {
220  for (unsigned k1 = 0; k1 < ntslots(); ++k1) {
221  EnsemblePtr_ e_k1 =
222  EnsemblesCollection_::getInstance()[dxi[k1].validTime()];
223  for (unsigned k2 = 0; k2 < ntslots(); ++k2) {
224  EnsemblePtr_ e_k2 =
225  EnsemblesCollection_::getInstance()[dxi[k2].validTime()];
226  double zz = (*e_k2)[l].dot_product_with(dxi[k2]);
227  dxo[k1].axpy(zz, (*e_k1)[l]);
228  }
229  }*/
230  }
231 
232  /*
233  * Go back to the right variables
234  */
235  for (unsigned k = 0; k < ntslots(); ++k) {
236  change_of_variable_->inverseMultiply(dxo_tmp[k]);
237  }
238  toControlVariableTimes(dxo_tmp, dxo);
239 }

320 template <typename MODEL>
321 void EnsembleCovariance4DWithChangeOfVariable<MODEL>::toControlVariableTimes(
322  const Increment4D_ & dxi, Increment4D_ & dxo) const {
323 
324  if (ntslots() != (dxi.last()-dxi.first()+1)) {
325  assign_no_date_ctrl(dxi[0], dxo[0]); //dxo[0] = dxi[0];
326 
327  for (unsigned k = 1; k < ntslots(); ++k) {
328  assign_no_date_ctrl(dxi[k - 1],
329  dxo[2 * k - 1]); // dxo[2 * k - 1] = dxi[k - 1];
330  dxo[2 * k - 1] *= 0.5;
331  dxo[2 * k - 1].axpy(0.5, dxi[k], false);
332 
333  assign_no_date_ctrl(dxi[k], dxo[2 * k]); // dxo[2 * k] = dxi[k];
334  }
335  } else {
336  dxo = dxi;
337  }
338 }

341 template <typename MODEL>
342 void EnsembleCovariance4DWithChangeOfVariable<MODEL>::toControlVariableTimes_ad(
343  const Increment4D_ & dxi, Increment4D_ & dxo) const {
344 
345  // Check times
346  int envar_times = dxi.last()-dxi.first()+1;
347  unsigned nb = ntslots();
348 
349  if (nb != envar_times) {
350  ASSERT(envar_times == (2 * nb - 1));
351 
352  assign_no_date_ctrl(dxi[0], dxo[0]); // dxo[0] = dxi[0];
353  dxo[0].axpy(0.5, dxi[1], false);
354 
355  for (unsigned k = 1; k < (nb - 1); ++k) {
356  assign_no_date_ctrl(dxi[2 * k], dxo[k]); // dxo[k] = dxi[2 * k];
357  dxo[k].axpy(0.5, dxi[2 * k - 1], false);
358  dxo[k].axpy(0.5, dxi[2 * k + 1], false);
359  }
360 
361  assign_no_date_ctrl(dxi[2 * nb - 2],
362  dxo[nb - 1]); // dxo[nb - 1] = dxi[2 * nb - 2];
363  dxo[nb - 1].axpy(0.5, dxi[2 * nb - 3], false);
364 
365  } else {
366  dxo = dxi;
367  }
368 }
