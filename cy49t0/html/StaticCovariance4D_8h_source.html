105 template <typename MODEL>
106 void StaticCovariance4D<MODEL>::advectedMultiply(const Increment4D_ &dxi,
107  Increment4D_ &dxo) const {
108  Increment_ temp(dxi[0]);
109  temp.zero();
110  for (int k2 = dxi.first(); k2 <= dxi.last(); ++k2) {
111  temp.axpy(1.0, dxi[k2], false);
112  }
113  static_->multiply(temp, dxo[dxi.first()]);
114  for (int k1 = dxi.first() + 1; k1 <= dxi.last(); ++k1) {
115  dxo[k1].zero();
116  dxo[k1].axpy(1.0, dxo[0], false);
117  }
118 }
