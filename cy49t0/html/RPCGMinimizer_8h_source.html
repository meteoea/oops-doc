88 template <typename MODEL>
89 double RPCGMinimizer<MODEL>::solve(Dual_ &vv, double &vvp, Dual_ &rr,
90  const HBHt_ &HBHt, const Rinv_ &Rinv,
91  const int &maxiter, const double &tolerance,
92  Dual_ &dy, const double &sigma) {
93  IdentityMatrix<Dual_> precond;
94  IdentityMatrix<Dual_> precondt;
95 
96  Dual_ zz(vv);
97  Dual_ ww(vv);
98  Dual_ pp(vv);
99  Dual_ tt(vv);
100  Dual_ ll(vv);
101  Dual_ qq(vv);
102  Dual_ w(vv);
103  Dual_ v(vv);
104 
105  // augmented part of the vectors
106  double rrp = 1.0;
107  double llp;
108  double wwp;
109  double zzp;
110  double ppp = 0.0;
111  double ttp = 0.0;
112  double qqp;
113  double vp = 1.0;
114  double wp;
115 
116  std::vector<Dual_> vVEC; // required for re-orthogonalization
117  std::vector<Dual_> wVEC; // required for re-orthogonalization
118  std::vector<double> vpVEC;
119  std::vector<double> wpVEC;
120 
121  // llaug = HBHtaug rraug
122  HBHt.multiply(rr, ll);
123  ll.axpy(rrp, dy);
124  llp = dot_product(dy, rr) + sigma * rrp;
125 
126  // wwaug = Gtaug llaug
127  precondt.multiply(ll, ww); // UPDATE WHEN G IS NOT IDENTITY
128  wwp = llp;
129 
130  // zzaug = Gaug rraug
131  precond.multiply(rr, zz); // UPDATE WHEN G IS NOT IDENTITY
132  zzp = rrp;
133 
134  double dotwr = dot_product(rr, ww) + rrp * wwp;
135  double normReduction = 1.0;
136  double dotw0r0 = dotwr;
137  double dotwr_old = 0.0;
138 
139  v = rr;
140  v *= 1 / sqrt(dotwr);
141  vp = rrp;
142  vp *= 1 / sqrt(dotwr);
143  w = ww;
144  w *= 1 / sqrt(dotwr);
145  wp = wwp;
146  wp *= 1 / sqrt(dotwr);
147 
148  vVEC.clear();
149  vpVEC.clear();
150  wVEC.clear();
151  wpVEC.clear();
152 
153  vVEC.push_back(v);
154  vpVEC.push_back(vp);
155  wVEC.push_back(w);
156  wpVEC.push_back(wp);
157 
158  Log::info() << std::endl;
159  for (int jiter = 0; jiter < maxiter; ++jiter) {
160  Log::info() << "RPCG Starting Iteration " << jiter + 1 << std::endl;
161 
162  if (jiter > 0) {
163  double beta = dotwr / dotwr_old;
164  Log::debug() << "RPCG beta = " << beta << std::endl;
165 
166  pp *= beta;
167  ppp *= beta;
168 
169  tt *= beta;
170  ttp *= beta;
171  }
172 
173  pp += zz;
174  ppp += zzp; // ppaug = zzaug + beta*ppaug
175 
176  tt += ww;
177  ttp += wwp; // ttaug = wwaug + beta*ttaug
178 
179  // (RinvHBHt + I) pp
180  Rinv.multiply(tt, qq);
181  qq += pp;
182  qqp = ppp; // qqaug = ppaug + Rinv_aug ttaug
183 
184  double alpha = dotwr / (dot_product(qq, tt) + qqp * ttp);
185  Log::debug() << "RPCG alpha = " << alpha << std::endl;
186 
187  vv.axpy(alpha, pp);
188  vvp = vvp + alpha * ppp; // vvaug = vvaug + alpha*ppaug
189 
190  rr.axpy(-alpha, qq);
191  rrp = rrp - alpha * qqp; // rraug = rraug - alpha*qqaug
192 
193  // Re-orthogonalization
194  for (int iiter = 0; iiter < jiter; ++iiter) {
195  double proj = dot_product(rr, wVEC[iiter]) + rrp * wpVEC[iiter];
196  rr.axpy(-proj, vVEC[iiter]);
197  rrp -= proj * vpVEC[iiter];
198  }
199 
200  // llaug = HBHt_aug rraug
201  HBHt.multiply(rr, ll);
202  ll.axpy(rrp, dy);
203  llp = dot_product(dy, rr) + sigma * rrp;
204 
205  // wwaug = Gt_aug llaug
206  precondt.multiply(ll, ww); // UPDATE WHEN G IS NOT IDENTITY
207  wwp = llp;
208 
209  // zzaug = Gaug rraug
210  precond.multiply(rr, zz); // UPDATE WHEN G IS NOT IDENTITY
211  zzp = rrp;
212 
213  dotwr_old = dotwr;
214  dotwr = dot_product(ww, rr) + rrp * wwp;
215 
216  v = rr;
217  v *= 1 / sqrt(dotwr);
218  vp = rrp;
219  vp *= 1 / sqrt(dotwr);
220  w = ww;
221  w *= 1 / sqrt(dotwr);
222  wp = wwp;
223  wp *= 1 / sqrt(dotwr);
224 
225  vVEC.push_back(v);
226  vpVEC.push_back(vp);
227  wVEC.push_back(w);
228  wpVEC.push_back(wp);
229 
230  normReduction = sqrt(dotwr / dotw0r0);
231 
232  Log::info() << "RPCG end of iteration " << jiter + 1
233  << ". Norm reduction= " << util::full_precision(normReduction)
234  << std::endl
235  << std::endl;
236 
237  if (normReduction < tolerance) {
238  Log::info() << "RPCG: Achieved required reduction in residual norm."
239  << std::endl;
240  break;
241  }
242  }
243  return normReduction;
244 }
