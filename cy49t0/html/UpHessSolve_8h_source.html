29 void UpHessSolve(std::vector<std::vector<double> >& UpHess,
30  const std::vector<double>& rhs, std::vector<double>& sol) {
31  const double n = rhs.size();
32  ASSERT(UpHess.size() == n);
33  sol.resize(n);
34  std::vector<double> v(n);
35  std::vector<double> y(n);
36 
37  // Step 1: Compute the LU factorization
38  // Note that L = I + diag(v(2:n),-1)
39  v[0] = 0;
40  for (int ii = 0; ii <= n - 2; ++ii) {
41  v[ii + 1] = UpHess[ii][ii + 1] / UpHess[ii][ii];
42  for (int jj = ii; jj <= n - 1; ++jj) {
43  UpHess[jj][ii + 1] = UpHess[jj][ii + 1] - v[ii + 1] * UpHess[jj][ii];
44  }
45  }
46  // Extract the upper triangular part (U part)
47  for (int ii = 0; ii <= n - 1; ++ii) {
48  for (int jj = ii + 1; jj <= n - 1; ++jj) {
49  UpHess[ii][jj] = 0;
50  }
51  }
52 
53  // Step 2: Solve the n-by-n unit lower bidiagonal system L y = rhs
54  y[0] = rhs[0];
55  for (int ii = 1; ii <= n - 1; ++ii) {
56  y[ii] = rhs[ii] - v[ii] * y[ii - 1];
57  }
58 
59  // Step 3: Solve the upper triangular system U sol = y
60  UpTriSolve(UpHess, y, sol, n);
61 }
