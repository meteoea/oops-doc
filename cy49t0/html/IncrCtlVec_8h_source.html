345 template <typename MODEL>
346 double IncrCtlVec<MODEL>::dot_product_with(const IncrCtlVec &dv) const {
347  Log::trace() << "IncrCtlVec<MODEL>::dot_product_with starting" << std::endl;
348  util::Timer timer(classname(), "dot_product_with");
349 
350  double zz = incrModCtlVec_->dot_product_with(*dv.incrModCtlVec_);
351 
352  if (ensSize_ != 0) {
353  std::vector<double> zzloc;
354  for (unsigned int jj = 0; jj < ensSize_; ++jj) {
355  zzloc.push_back(incrEnsCtlVecCollection_[jj]->dot_product_with(
356  *dv.incrEnsCtlVecCollection_[jj],true));
357  }
358  eckit::mpi::comm().allReduceInPlace(zzloc.begin(), zzloc.end(), eckit::mpi::sum());
359  zz += std::accumulate(zzloc.begin(), zzloc.end(), double(0.0));
360  }
361 
362  Log::trace() << "IncrCtlVec<MODEL>::dot_product_with done" << std::endl;
363  return zz;
364 }
