98 template <typename MODEL>
99 CostFct4DEnsVar<MODEL>::CostFct4DEnsVar(const eckit::Configuration &config,
100  const Geometry_ &geom,
101  const Model_ &model)
102  : CostFunction<MODEL>::CostFunction(config, geom, model),
103  zero_(0),
104  geom_(geom),
105  ctlvars_(config) {
106  windowLength_ = util::Duration(config.getString("window_length"));
107  windowBegin_ = util::DateTime(config.getString("window_begin"));
108  windowEnd_ = windowBegin_ + windowLength_;
109  windowSub_ = util::Duration(config.getString("window_sub"));
110 
111  ncontrol_ = windowLength_.toSeconds() / windowSub_.toSeconds();
112  ASSERT(windowLength_.toSeconds() == windowSub_.toSeconds() * ncontrol_);
113 
114  this->setupTerms(config);
115  Log::trace() << "CostFct4DEnsVar constructed" << std::endl;
116 }

148 template <typename MODEL>
149 void CostFct4DEnsVar<MODEL>::runNL(CtrlVar_ &xx,
150  PostProcessor<State_> &post) const {
151  for (unsigned int jsub = 0; jsub <= ncontrol_; ++jsub) {
152  util::DateTime now(windowBegin_ + jsub * windowSub_);
153 
154  ASSERT(xx.state()[jsub].validTime() == now);
155  CostFct_::getModel().forecast(xx.state()[jsub], xx.modVar(),
156  util::Duration(0), post);
157  ASSERT(xx.state()[jsub].validTime() == now);
158  }
159 }

180 template <typename MODEL>
181 void CostFct4DEnsVar<MODEL>::zeroAD(CtrlInc_ &dx) const {
182  util::DateTime now(windowBegin_);
183  for (unsigned int jsub = 0; jsub <= ncontrol_; ++jsub) {
184  dx.state()[jsub].zero(now);
185  now += windowSub_;
186  }
187  dx.modVar().zero();
188  dx.obsVar().zero();
189 }

201  ASSERT(dx.state()[jsub].validTime() == now);
202  CostFct_::getTLM(jsub).forecastAD(dx.state()[jsub], dx.modVar(),
203  util::Duration(0), post, cost);
204  ASSERT(dx.state()[jsub].validTime() == now);
205  }
