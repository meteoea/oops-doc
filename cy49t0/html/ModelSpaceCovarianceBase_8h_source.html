159 template <typename MODEL>
160 ModelSpaceCovarianceBase<MODEL> *CovarianceFactory<MODEL>::create(
161  const eckit::Configuration &conf, const Geometry_ &resol,
162  const Variables_ &vars, const State_ &xb) {
163  const std::string id = conf.getString("covariance");
164  Log::trace() << "ModelSpaceCovarianceBase type = " << id << std::endl;
165  typename std::map<std::string, CovarianceFactory<MODEL> *>::iterator j =
166  makers_.find(id);
167  if (j == makers_.end()) {
168  Log::error() << id << " does not exist in covariance factory." << std::endl;
169  Log::error() << "Covariance factory contains " << makers_.size()
170  << " elements:" << std::endl;
171  for (typename std::map<std::string,
172  CovarianceFactory<MODEL> *>::const_iterator jj =
173  makers_.begin();
174  jj != makers_.end(); ++jj) {
175  Log::error() << "A" << jj->first << "B" << std::endl;
176  }
177  ABORT("Element does not exist in CovarianceFactory.");
178  }
179  return (*j).second->make(conf, resol, vars, xb);
180 }
