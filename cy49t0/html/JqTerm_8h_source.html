60 template <typename MODEL>
61 void JqTerm<MODEL>::computeModelError(const State4D_ &fg, Increment4D_ &dx) {
62  // Compute x_i - M(x_{i-1})
63  for (unsigned jsub = 1; jsub < nsubwin_; ++jsub) {
64  dx[jsub].diff(fg[jsub], mxi_[jsub - 1]);
65  Log::info() << "CostJbJq: x_" << jsub << " - M(x_" << jsub - 1 << ")"
66  << dx[jsub] << std::endl;
67  }
68  mxi_.clear();
69 }
