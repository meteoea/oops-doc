127 template<typename MODEL>
128 void EnsembleCovariance<MODEL>::linearize(const State_ & xb,
129  const Geometry_ & resol, const eckit::Configuration & config) {
130  geom_.reset(new Geometry_(resol));
131 
132  // Compute the ensemble of perturbations at time of xb.
133  ASSERT(xb.validTime() == time_);
134  EnsemblePtr_ ens(new Ensemble_(xb.validTime(), config));
135  ens->build(xb, resol);
136  EnsemblesCollection_::getInstance().put(xb.validTime(), ens);
137  ens->read();
138  if (has_change_of_variable_) {
139  change_of_variable_.reset(
140  MatrixFactory_::create(eckit::LocalConfiguration(config, "change_of_variable")));
141  for (unsigned l = 0; l < ens->size(); ++l) {
142  change_of_variable_->multiply((*ens)[l]);
143  }
144  }
145  ens->to_perturbations(1);
146 
147  // Build Localization
148  const eckit::LocalConfiguration conf(config, "localization");
149  loc_.reset(new Localization_(resol, conf));
150 
151  Log::trace() << "EnsembleCovariance linearized." << std::endl;
152 }

154 template <typename MODEL>
155 void EnsembleCovariance<MODEL>::multiply(const Increment_ &dxi,
156  Increment_ &dxo) const {
157  EnsemblePtr_ e_k2 = EnsemblesCollection_::getInstance()[dxi.validTime()];
158  EnsemblePtr_ e_k1 = EnsemblesCollection_::getInstance()[dxo.validTime()];
159  Increment_ dxi_tmp(dxi);
160 
161  // Apply to dxi the block B_k1k2 of the B matrix which contains the
162  // covariance of perturbations at time k1 with covariance at time k2
163  if (has_change_of_variable_)
164  change_of_variable_->transposeInverseMultiply(dxi_tmp);
165  dxo.zero();
166  for (unsigned int m = 0; m < e_k1->size(); ++m) {
167  Increment_ dx(dxi_tmp);
168  dx.schur_product_with((*e_k2)[m]);
169  loc_->multiply(dx);
170  dx.schur_product_with((*e_k1)[m]);
171 
172  // We can't use '+=' , dxo and dx aren't at the same time (QG)
173  dxo.axpy(1.0, dx, false);
174  }
175  if (has_change_of_variable_) change_of_variable_->inverseMultiply(dxo);
176 }

184 template <typename MODEL>
185 void EnsembleCovariance<MODEL>::multiplySqrt(const IncrCtlVec_ &dv,
186  Increment_ &dxo) const {
187  EnsemblePtr_ e_k = EnsemblesCollection_::getInstance()[dxo.validTime()];
188 
189  dxo.zero();
190  Increment_ dxi(dxo);
191  for (unsigned int m = 0; m < e_k->size(); ++m) {
192  loc_->multiplySqrt(dv.ensCtlVec(m), dxi);
193  dxi.schur_product_with((*e_k)[m]);
194  dxo.axpy(1.0, dxi, false);
195  }
196  if (has_change_of_variable_) change_of_variable_->inverseMultiply(dxo);
197 }

199 template <typename MODEL>
200 void EnsembleCovariance<MODEL>::multiplySqrtTrans(const Increment_ &dxi,
201  IncrCtlVec_ &dv) const {
202  EnsemblePtr_ e_k = EnsemblesCollection_::getInstance()[dxi.validTime()];
203  Increment_ dxi_tmp(dxi);
204 
205  if (has_change_of_variable_)
206  change_of_variable_->transposeInverseMultiply(dxi_tmp);
207  for (unsigned int m = 0; m < e_k->size(); ++m) {
208  dv.ensCtlVec(m).zero();
209  Increment_ dx(dxi_tmp);
210  dx.schur_product_with((*e_k)[m]);
211  loc_->multiplySqrtTrans(dx, dv.ensCtlVec(m));
212  }
213 }

215 template<typename MODEL>
216 void EnsembleCovariance<MODEL>::randomize(Increment_ & dxo) const {
217  // Build dv
218  EnsemblePtr_ e_k = EnsemblesCollection_::getInstance()[dxo.validTime()];
219  std::vector<std::unique_ptr<IncrEnsCtlVec<MODEL> > > dv;
220  for (unsigned int jj = 0; jj < e_k->size(); ++jj) {
221  dv.emplace_back(this->newIncrEnsCtlVec());
222  }
223 
224  // Initialize dv
225  for (unsigned int m = 0; m < e_k->size(); ++m) {
226  dv[m]->random();
227  }
228 
229  // dxo = B^1/2 dv
230  dxo.zero();
231  Increment_ dx(dxo);
232  for (unsigned int m = 0; m < e_k->size(); ++m) {
233  loc_->multiplySqrt(*dv[m], dx);
234  dx.schur_product_with((*e_k)[m]);
235  dxo.axpy(1.0, dx, false);
236  }
237  if (has_change_of_variable_) change_of_variable_->inverseMultiply(dxo);
238 }
